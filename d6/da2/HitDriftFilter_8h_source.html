26  std::function<float(float, float, float)>&& f)

28  // f is some function taking event time + z of hit and returns final "digit" time
29  LOG(DEBUG) << "BR NAME " << branchname;
30  auto br = chain.GetBranch(branchname);
31  if (!br) {
32  LOG(ERROR) << "No branch found";

34  }
35 
36  auto nentries = br->GetEntries();
37 
38  // do the filtering
39  for (int entry = 0; entry < nentries; ++entry) {
40  if (tmin > f(eventrecords[entry].timeNS, 0, 0)) {
41  continue;
42  }
43  if (tmax < f(eventrecords[entry].timeNS, 0, 250)) {
44  break;
45  }
46 
47  // This needs to be done only once for any entry per chain
48  if (hitvectors[entry] == nullptr) {
49  br->SetAddress(&hitvectors[entry]);
50  br->GetEntry(entry);
51  }
52 
53  int groupid = -1;
54  auto groups = hitvectors[entry];
55  for (auto& singlegroup : *groups) {
56  if (singlegroup.getSize() == 0) {
57  // there are not hits in this group .. so continue
58  // TODO: figure out why such a group would exist??
59  continue;
60  }
61  const auto& pos = singlegroup.getHit(0).getPos();
62  // std::cout << "This Group is in sector " << o2::TPC::Sector::ToSector(pos.X(), pos.Y(), pos.Z()) << "\n";
63  groupid++;
64  auto zmax = singlegroup.mZAbsMax;
65  auto zmin = singlegroup.mZAbsMin;
66  // in case of secondaries, the time ordering may be reversed
67  if (zmax < zmin) {
68  std::swap(zmax, zmin);
69  }
70  // auto tof = singlegroup.
71  float tmaxtrack = f(eventrecords[entry].timeNS, 0., zmin);
72  float tmintrack = f(eventrecords[entry].timeNS, 0., zmax);
73  if (tmin > tmaxtrack || tmax < tmintrack) {
74  // std::cout << "DISCARDING " << groupid << " OF ENTRY " << entry << "\n";
75  continue;
76  }
77  // need to record index of the group
78  hitids.emplace_back(entry, groupid);
79  }
80  }

119  const auto storeentry = eventID * maxpartspercollision + source;
