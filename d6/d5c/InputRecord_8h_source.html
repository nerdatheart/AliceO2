141  DataRef getByPos(int pos) const {
142  if (pos * 2 + 1 > mSpan.size() || pos < 0) {
143  throw std::runtime_error("Unknown message requested at position " + std::to_string(pos));
144  }
145  if (pos > mInputsSchema.size()) {
146  throw std::runtime_error("Unknown schema at position" + std::to_string(pos));
147  }
148  if (mSpan.get(pos * 2) != nullptr && mSpan.get(pos * 2 + 1) != nullptr) {
149  return DataRef{ &mInputsSchema[pos].matcher,
150  mSpan.get(pos * 2),
151  mSpan.get(pos * 2 + 1) };
152  } else {
153  return DataRef{ &mInputsSchema[pos].matcher, nullptr, nullptr };
154  }
155  }
