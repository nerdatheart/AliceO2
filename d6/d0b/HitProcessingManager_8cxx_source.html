26 bool HitProcessingManager::setupChain()
27 {
28  if (mBackgroundFileNames.size() == 0 && mSignalFileNames.size() == 0) {
29  // there are no files to be analysed/processed;
30  LOG(WARNING) << "No files to be analysed";
31  return false;
32  }
33 
34  // count max signal id
35  int sourcecounter = 0;
36  int maxsourceid = 0;
37  for (auto& pair : mSignalFileNames) {
38  sourcecounter++;
39  maxsourceid = std::max(pair.first, maxsourceid);
40  }
41  if (maxsourceid != sourcecounter) {
42  LOG(WARNING) << "max source id " << maxsourceid << " vs " << sourcecounter;
43  }
44  LOG(INFO) << "setting up " << maxsourceid + 1 << " chains";
45  mSimChains.resize(maxsourceid + 1);
46 
47  // allocate chains
48  for (int i = 0; i < mSimChains.size(); ++i) {
49  // make o2sim a parameter
50  mSimChains[i] = new TChain("o2sim");
51  }
52 
53  // background chain
54  auto& c = *mSimChains[0];
55  c.Reset();
56  for (auto& filename : mBackgroundFileNames) {
57  c.AddFile(filename.data());
58  }
59 
60  for (auto& pair : mSignalFileNames) {
61  const auto& signalid = pair.first;
62  const auto& filenamevector = pair.second;
63  auto& chain = *mSimChains[signalid];
64  for (auto& filename : filenamevector) {
65  chain.AddFile(filename.data());
66  }
67  }
68 
69  return true;
70 }
