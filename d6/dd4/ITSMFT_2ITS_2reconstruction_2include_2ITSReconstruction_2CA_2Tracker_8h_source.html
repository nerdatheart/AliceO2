110 template <bool IsGPU>
111 GPU_DEVICE const int4 TrackerTraits<IsGPU>::getBinsRect(const Cluster& currentCluster, const int layerIndex,
112  const float directionZIntersection, float maxdeltaz, float maxdeltaphi)
113 {
114  const float zRangeMin = directionZIntersection - 2 * maxdeltaz;
115  const float phiRangeMin = currentCluster.phiCoordinate - maxdeltaphi;
116  const float zRangeMax = directionZIntersection + 2 * maxdeltaz;
117  const float phiRangeMax = currentCluster.phiCoordinate + maxdeltaphi;
118 
119  if (zRangeMax < -Constants::ITS::LayersZCoordinate()[layerIndex + 1] ||
120  zRangeMin > Constants::ITS::LayersZCoordinate()[layerIndex + 1] || zRangeMin > zRangeMax) {
121 
122  return getEmptyBinsRect();
123  }
124 
125  return int4{ MATH_MAX(0, IndexTableUtils::getZBinIndex(layerIndex + 1, zRangeMin)),
126  IndexTableUtils::getPhiBinIndex(MathUtils::getNormalizedPhiCoordinate(phiRangeMin)),
127  MATH_MIN(Constants::IndexTable::ZBins - 1, IndexTableUtils::getZBinIndex(layerIndex + 1, zRangeMax)),
128  IndexTableUtils::getPhiBinIndex(MathUtils::getNormalizedPhiCoordinate(phiRangeMax)) };
129 }

169 template <bool IsGPU>
170 template <typename... T>
171 float Tracker<IsGPU>::evaluateTask(void (Tracker<IsGPU>::*task)(T...), const char* taskName, std::ostream& ostream,
172  T&&... args)
173 {
174  float diff{ 0.f };
175 
176  if (Constants::DoTimeBenchmarks) {
177  auto start = std::chrono::high_resolution_clock::now();
178  (this->*task)(std::forward<T>(args)...);
179  auto end = std::chrono::high_resolution_clock::now();
180 
181  std::chrono::duration<double, std::milli> diff_t{ end - start };
182  diff = diff_t.count();
183 
184  if (taskName == nullptr) {
185  ostream << diff << "\t";
186  } else {
187  ostream << std::setw(2) << " - " << taskName << " completed in: " << diff << " ms" << std::endl;
188  }
189  } else {
190  (this->*task)(std::forward<T>(args)...);
191  }
192 
193  return diff;
194 }
