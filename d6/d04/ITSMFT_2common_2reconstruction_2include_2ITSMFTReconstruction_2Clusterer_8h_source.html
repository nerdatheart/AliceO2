98  void addNewPrecluster(UInt_t ip, UShort_t row)
99  {
100  mPreClusterHeads.push_back(mPixels.size());
101  // new head does not point yet (-1) on other pixels, store just the entry of the pixel in the ChipPixelData
102  mPixels.emplace_back(-1, ip);
103  int lastIndex = mPreClusterIndices.size();
104  mPreClusterIndices.push_back(lastIndex);
105  mCurr[row] = lastIndex; // store index of the new precluster in the current column buffer
106  }

109  void expandPreCluster(UInt_t ip, UShort_t row, int preClusIndex)
110  {
111  auto& firstIndex = mPreClusterHeads[mPreClusterIndices[preClusIndex]];
112  mPixels.emplace_back(firstIndex, ip);
113  firstIndex = mPixels.size() - 1;
114  mCurr[row] = preClusIndex;
115  }
