206 template <typename Container>
207 void attachTMessage(Container const& hits, FairMQChannel& channel, FairMQParts& parts)
208 {
209  TMessage* tmsg = new TMessage();
210  tmsg->WriteObjectAny((void*)&hits, TClass::GetClass(typeid(hits)));
211  auto free_tmessage = [](void* data, void* hint) { delete static_cast<TMessage*>(hint); };
212 
213  std::unique_ptr<FairMQMessage> message(channel.NewMessage(tmsg->Buffer(), tmsg->BufferSize(), free_tmessage, tmsg));
214  parts.AddPart(std::move(message));
215 }

269  // generic implementation for the updateHitTrackIndices interface
270  // assumes Detectors have a GetHits(int) function that return some iterable
271  // hits which are o2::BaseHits
272  void updateHitTrackIndices(std::map<int, int> const& indexmapping) override
273  {
274  int probe = 0; // some Detectors have multiple hit vectors and we are probing
275  // them via a probe integer until we get a nullptr
276  while (auto hits = static_cast<Det*>(this)->Det::getHits(probe++)) {
277  for (auto& hit : *hits) {
278  auto iter = indexmapping.find(hit.GetTrackID());
279  hit.SetTrackID(iter->second);

285  {

294  void fillHitBranch(TTree& tr, FairMQParts& parts, int& index) override
295  {
296  int probe = 0;
297  using Hit_t = decltype(static_cast<Det*>(this)->Det::getHits(probe));
298  std::string name = static_cast<Det*>(this)->getHitBranchNames(probe++);
299  while (name.size() > 0) {
300  // for each branch name we extract/decode hits from the message parts ...
301  auto hitsptr = decodeTMessage<Hit_t>(parts, index++);
302 
303  // ... and fill the tree branch
304  auto br = getOrMakeBranch(tr, name.c_str(), hitsptr);
305  br->Fill();
306  br->ResetAddress();
307 
308  // next name
309  name = static_cast<Det*>(this)->getHitBranchNames(probe++);
310  }
311  }
