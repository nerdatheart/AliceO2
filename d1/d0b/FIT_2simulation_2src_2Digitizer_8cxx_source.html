31 void Digitizer::process(const std::vector<HitType>* hits, Digit* digit)
32 {
33  //parameters constants TO DO: move to class
34  constexpr Double_t TimeDiffAC = (Geometry::ZdetA - Geometry::ZdetC) * TMath::C();
35  constexpr Int_t nMCPs = (Geometry::NCellsA + Geometry::NCellsC) * 4;
36 
37  constexpr Double_t BC_clk = 25.; //ns event clk lenght
38  constexpr Double_t BC_clk_center = BC_clk / 2.; // clk center
39  constexpr Double_t C_side_cable_cmps = 2.8; //ns
40  constexpr Double_t A_side_cable_cmps = 11.; //ns
41  constexpr Double_t signal_width = 5.; // time gate for signal, ns
42  constexpr Double_t nPe_in_mip = 250.; // n ph. e. in one mip
43  constexpr Double_t CFD_trsh_mip = 0.4; // = 4[mV] / 10[mV/mip]
44  constexpr Double_t time_trg_gate = 4.; // ns
45 
46  constexpr Double_t trg_central_trh = 100.; // mip
47  constexpr Double_t trg_semicentral_trh = 50.; // mip
48  constexpr Double_t trg_vertex_min = BC_clk_center - 3.; //ns
49  constexpr Double_t trg_vertex_max = BC_clk_center + 3.; //ns
50 
51  //mDigits = digits;
52 
53  Int_t nlbl = 0; //number of MCtrues
54  Double_t digit_timeframe = mEventTime;
55  Double_t digit_bc = mEventID;
56  Int_t nClk = floor(mEventTime / BC_clk);
57  Double_t BCEventTime = mEventTime - BC_clk * nClk;
58 
59  // Counting photo-electrons, mean time --------------------------------------
60  Int_t ch_hit_nPe[nMCPs] = {};
61  Double_t ch_hit_mean_time[nMCPs] = {};
62 
63  for (auto& hit : *hits) {
64  Int_t hit_ch = hit.GetDetectorID();
65  Double_t hit_time = hit.GetTime();
66 
67  if (hit_time != 0.) {
68  ch_hit_nPe[hit_ch]++;
69  ch_hit_mean_time[hit_ch] += hit_time;
70  }
71  if (hit.GetEnergyLoss() > 0) {
72  o2::fit::MCLabel label(hit.GetTrackID(), mEventID, mSrcID, hit_ch);
73  // o2::MCCompLabel label(hit.GetTrackID(), mEventID, mSrcID);
74  int tr, ev, sr;
75  label.get(tr, ev, sr);
76  int lblCurrent;
77  if (mMCLabels) {
78  lblCurrent = mMCLabels->getIndexedSize(); // this is the size of mHeaderArray;
79  mMCLabels->addElement(lblCurrent, label);
80  nlbl++;
81  }
82  }
83  }
84 
85  for (Int_t ch_iter = 0; ch_iter < nMCPs; ch_iter++) {
86  if (ch_hit_nPe[ch_iter] != 0) {
87  ch_hit_mean_time[ch_iter] = ch_hit_mean_time[ch_iter] / (float)ch_hit_nPe[ch_iter];
88 
89  // LOG(DEBUG) << "nMCP: " << ch_iter << " n Ph. e. " << ch_hit_nPe[ch_iter] << " mean time " << ch_hit_mean_time[ch_iter] << FairLogger::endl;
90  }
91  }
92  // --------------------------------------------------------------------------
93 
94  //Calculating signal time, amplitude in mean_time +- time_gate --------------
95  Double_t ch_signal_nPe[nMCPs] = {};
96  Double_t ch_signal_MIP[nMCPs] = {};
97  Double_t ch_signal_time[nMCPs] = {};
98 
99  for (auto& hit : *hits) {
100  Int_t hit_ch = hit.GetDetectorID();
101  Double_t hit_time = hit.GetTime();
102  Bool_t is_A_side = (hit_ch <= 4 * Geometry::NCellsA);
103  Double_t time_compensate = is_A_side ? A_side_cable_cmps : C_side_cable_cmps;
104  Bool_t is_hit_in_signal_gate = (hit_time > ch_hit_mean_time[hit_ch] - signal_width * .5) &&
105  (hit_time < ch_hit_mean_time[hit_ch] + signal_width * .5);
106 
107  Double_t hit_time_corr = hit_time - time_compensate + BC_clk_center /* + BCEventTime*/;
108  Double_t is_time_in_gate = (hit_time != 0.); //&&(hit_time_corr > -BC_clk_center)&&(hit_time_corr < BC_clk_center);
109 
110  if (is_time_in_gate && is_hit_in_signal_gate) {
111  ch_signal_nPe[hit_ch]++;
112  ch_signal_time[hit_ch] += hit_time_corr;
113  }
114  }
115 
116  for (Int_t ch_iter = 0; ch_iter < nMCPs; ch_iter++) {
117  if (ch_signal_nPe[ch_iter] != 0) {
118  ch_signal_MIP[ch_iter] = ch_signal_nPe[ch_iter] / nPe_in_mip;
119  ch_signal_time[ch_iter] = ch_signal_time[ch_iter] / (float)ch_signal_nPe[ch_iter];
120  }
121 
122  //if ampl less than cfd trh adc and cfd has no signal
123  if (ch_signal_MIP[ch_iter] < CFD_trsh_mip) {
124  ch_signal_MIP[ch_iter] = 0.;
125  ch_signal_time[ch_iter] = 0.;
126  }
127  }
128 
129  // Calculating triggers -----------------------------------------------------
130  Int_t n_hit_A = 0., n_hit_C = 0.;
131  Double_t mean_time_A = 0.;
132  Double_t mean_time_C = 0.;
133  Double_t summ_ampl_A = 0.;
134  Double_t summ_ampl_C = 0.;
135  Double_t vertex_time;
136 
137  Bool_t is_A, is_C;
138  Bool_t is_Central;
139  Bool_t is_SemiCentral;
140  Bool_t is_Vertex;
141 
142  for (Int_t ch_iter = 0; ch_iter < nMCPs; ch_iter++) {
143  Bool_t is_A_side = (ch_iter <= 4 * Geometry::NCellsA);
144  Bool_t is_time_in_trg_gate = (ch_signal_time[ch_iter] > BC_clk_center - time_trg_gate * 0.5) && (ch_signal_time[ch_iter] < BC_clk_center + time_trg_gate * 0.5);
145  // if (ch_signal_time[ch_iter]>0)
146  if (ch_signal_MIP[ch_iter] == 0.)
147  continue;
148  if (ch_signal_MIP[ch_iter] < CFD_trsh_mip)
149  continue;
150  if (!is_time_in_trg_gate)
151  continue;
152 
153  if (is_A_side) {
154  n_hit_A++;
155  summ_ampl_A += ch_signal_MIP[ch_iter];
156  mean_time_A += ch_signal_time[ch_iter];
157  } else {
158  n_hit_C++;
159  summ_ampl_C += ch_signal_MIP[ch_iter];
160  mean_time_C += ch_signal_time[ch_iter];
161  }
162  }
163 
164  is_A = n_hit_A > 0;
165  is_C = n_hit_C > 0;
166  is_Central = summ_ampl_A + summ_ampl_C >= trg_central_trh;
167  is_SemiCentral = summ_ampl_A + summ_ampl_C >= trg_semicentral_trh;
168 
169  mean_time_A = is_A ? mean_time_A / n_hit_A : 0.;
170  mean_time_C = is_C ? mean_time_C / n_hit_C : 0.;
171  vertex_time = (mean_time_A + mean_time_C) * .5;
172  is_Vertex = (vertex_time > trg_vertex_min) && (vertex_time < trg_vertex_max);
173  // --------------------------------------------------------------------------
174 
175  //filling digit
176  digit->setTime(digit_timeframe);
177  digit->setBC(mEventID);
178  digit->setTriggers(is_A, is_C, is_Central, is_SemiCentral, is_Vertex);
179 
180  std::vector<ChannelData> mChDgDataArr;
181  // LOG(DEBUG) << "nMCP : :IsA : hit nPe : hit mTime : sig MIP : sig mTime" << FairLogger::endl;
182  for (Int_t ch_iter = 0; ch_iter < nMCPs; ch_iter++) {
183  if (ch_signal_MIP[ch_iter] > CFD_trsh_mip) {
184  Double_t smeared_time = gRandom->Gaus(ch_signal_time[ch_iter], 0.050) + BCEventTime;
185  mChDgDataArr.emplace_back(ChannelData{ ch_iter, smeared_time, ch_signal_MIP[ch_iter] });
186  LOG(DEBUG) << ch_iter << " : "
187  << " : " << ch_signal_time[ch_iter] << " : "
188  << ch_signal_MIP[ch_iter] << " : " << smeared_time << FairLogger::endl;
189  }
190  }
191 
192  digit->setChDgData(std::move(mChDgDataArr));
193 
194  // Debug output -------------------------------------------------------------
195  LOG(DEBUG) << "\n\nTest digizing data ===================" << FairLogger::endl;
196 
197  LOG(DEBUG) << "Event ID: " << mEventID << " Event Time " << mEventTime << FairLogger::endl;
198  LOG(DEBUG) << "nClk: " << nClk << " BC Event Time " << BCEventTime << FairLogger::endl;
199 
200  LOG(DEBUG) << "N hit A: " << n_hit_A << " N hit C: " << n_hit_C << " summ ampl A: " << summ_ampl_A
201  << " summ ampl C: " << summ_ampl_C << " mean time A: " << mean_time_A
202  << " mean time C: " << mean_time_C << FairLogger::endl;
203 
204  LOG(DEBUG) << "IS A " << is_A << " IS C " << is_C << " is Central " << is_Central
205  << " is SemiCentral " << is_SemiCentral << " is Vertex " << is_Vertex << FairLogger::endl;
206 
207  LOG(DEBUG) << "======================================\n\n"
208  << FairLogger::endl;
209  // --------------------------------------------------------------------------
210 }
