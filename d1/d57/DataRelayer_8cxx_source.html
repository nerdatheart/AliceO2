101 void DataRelayer::processDanglingInputs(std::vector<ExpirationHandler> const& expirationHandlers,
102  ServiceRegistry& services)
103 {
104  for (size_t ti = 0; ti < mTimesliceIndex.size(); ++ti) {
105  // FIXME: for the moment we need to have at least one data input before we
106  // can invoke the dangling inputs helpers.
107  // This is useful for stuff like conditions or histograms, but not
108  // (yet) for timers. For those we would need to have a mechanism
109  // to create valid timeslices.
110  TimesliceSlot slot{ ti };
111  if (mTimesliceIndex.isValid(slot) == false) {
112  continue;
113  }
114  assert(mDistinctRoutesIndex.empty() == false);
115  for (size_t ri = 0; ri < mDistinctRoutesIndex.size(); ++ri) {
116  auto& route = mInputRoutes[mDistinctRoutesIndex[ri]];
117  auto& expirator = expirationHandlers[mDistinctRoutesIndex[ri]];
118  auto timestamp = mTimesliceIndex.getTimesliceForSlot(slot);
119  auto& part = mCache[ti * mDistinctRoutesIndex.size() + ri];
120  if (part.header == nullptr && part.payload == nullptr && expirator.checker && expirator.checker(timestamp.value)) {
121  assert(ti * mDistinctRoutesIndex.size() + ri < mCache.size());
122  assert(expirator.handler);
123  expirator.handler(services, part, timestamp.value);
124  mTimesliceIndex.markAsDirty(slot, true);
125  assert(part.header != nullptr);
126  assert(part.payload != nullptr);
127  }
128  }
129  }
130 }

135 size_t
136  assignInputSpecId(void* data, std::vector<DataDescriptorMatcher> const& matchers)
137 {
140  std::vector<ContextElement> context{};
141 
142  for (size_t ri = 0, re = matchers.size(); ri < re; ++ri) {
143  auto& matcher = matchers[ri];
144  const DataHeader* h = o2::header::get<DataHeader*>(data);
145  if (h == nullptr) {

156 DataRelayer::RelayChoice
157 DataRelayer::relay(std::unique_ptr<FairMQMessage> &&header,
158  std::unique_ptr<FairMQMessage> &&payload) {
159  // STATE HOLDING VARIABLES
160  // This is the class level state of the relaying. If we start supporting
161  // multithreading this will have to be made thread safe before we can invoke
162  // relay concurrently.
163  auto const& inputRoutes = mInputRoutes;
164  auto& index = mTimesliceIndex;
165 
166  auto& cache = mCache;
167  auto const& readonlyCache = mCache;
168  auto& metrics = mMetrics;
169  auto numInputTypes = mDistinctRoutesIndex.size();
170 
171  // IMPLEMENTATION DETAILS
172  //
173  // This returns the identifier for the given input. We use a separate
174  // function because while it's trivial now, the actual matchmaking will
175  // become more complicated when we will start supporting ranges.
176  auto getInput = [&matchers = mInputMatchers ,&header] () -> int {
177  return assignInputSpecId(header->GetData(), matchers);
178  };
179 
180  // This will check if the input is valid. We hide the details so that
181  // in principle the outer code will work regardless of the actual
182  // implementation.
183  auto isValidInput = [](int inputIdx) {
184  // If this is true, it means the message we got does
185  // not match any of the expected inputs.
186  return inputIdx != INVALID_INPUT;
187  };
188 
189  // The timeslice is embedded in the DataProcessingHeader header of the O2
190  // header stack. This is an extension to the DataHeader, because apparently
191  // we do have data which comes without a timestamp, although I am personally
192  // not sure what that would be.
193  const auto getTimeslice = [&header, &index]() -> TimesliceId {
194  const DataProcessingHeader* dph = o2::header::get<DataProcessingHeader*>(header->GetData());
195  if (dph == nullptr) {
196  return TimesliceId{ TimesliceId::INVALID };
197  }
198  size_t timesliceId = dph->startTime;
199  assert(index.size());
200  return TimesliceId{ timesliceId };
201  };
202 
203  // A cache line is obsolete if the incoming one has a greater timestamp or
204  // if the slot contains an invalid timeslice.
205  const auto isCacheEntryObsoleteFor = [&index](TimesliceId timeslice) {
206  auto current = index.getTimesliceForSlot(index.getSlotForTimeslice(timeslice));
207  return TimesliceId::isValid(current) == false || (current.value < timeslice.value);
208  };
209 
210  // We need to prune the cache from the old stuff, if any. Otherwise we
211  // simply store the payload in the cache and we mark relevant bit in the
212  // hence the first if.
213  auto pruneCacheSlotFor = [&cache, &numInputTypes, &index, &metrics](TimesliceId timeslice) {
214  assert(cache.empty() == false);
215  assert(index.size() * numInputTypes == cache.size());
216  auto slot = index.getSlotForTimeslice(timeslice);
217  // Prune old stuff from the cache, hopefully deleting it...
218  // We set the current slot to the timeslice value, so that old stuff
219  // will be ignored.
220  assert(numInputTypes * slot.index < cache.size());
221  index.bookTimeslice(timeslice);
222  for (size_t ai = slot.index * numInputTypes, ae = ai + numInputTypes; ai != ae; ++ai) {
223  cache[ai].header.reset(nullptr);
224  cache[ai].payload.reset(nullptr);
225  metrics.send({ 0, sMetricsNames[ai] });
226  }
227  };
228 
229  // We need to check if the slot for the current input is already taken for
230  // the current timeslice.
231  // This should never happen, however given this is dependent on the input
232  // we want to protect again malicious / bad upstream source.
233  auto hasCacheInputAlreadyFor = [&cache, &index, &numInputTypes](TimesliceId timeslice, int input) {
234  auto slot = index.getSlotForTimeslice(timeslice);
235  PartRef& currentPart = cache[numInputTypes * slot.index + input];
236  return (currentPart.payload != nullptr) || (currentPart.header != nullptr);
237  };
238 
239  // Actually save the header / payload in the slot
240  auto saveInSlot = [&header, &payload, &cache, &index, &numInputTypes, &metrics](TimesliceId timeslice, int input) {
241  auto slot = index.bookTimeslice(timeslice);
242  auto cacheIdx = numInputTypes * slot.index + input;
243  PartRef& currentPart = cache[cacheIdx];
244  metrics.send({ 1, sMetricsNames[cacheIdx] });
245  PartRef ref{std::move(header), std::move(payload)};
246  currentPart = std::move(ref);
247  assert(header.get() == nullptr && payload.get() == nullptr);
248  };
249 
250  // OUTER LOOP
251  //
252  // This is the actual outer loop processing input as part of a given
253  // timeslice. All the other implementation details are hidden by the lambdas
254  auto input = getInput();
255 
256  if (isValidInput(input) == false) {
257  LOG(ERROR) << "A malformed message just arrived";
258  return WillNotRelay;
259  }
260 
261  auto timeslice = getTimeslice();
262  auto slot = index.getSlotForTimeslice(timeslice);
263  LOG(DEBUG) << "Received timeslice" << timeslice.value;
264  if (TimesliceId::isValid(timeslice) == false) {
265  LOG(ERROR) << "Could not determine the timeslice for input";
266  return WillNotRelay;
267  }
268 
269  if (index.isObsolete(timeslice)) {
270  LOG(ERROR) << "An entry for timeslice " << timeslice.value << " just arrived but too late to be processed";
271  return WillNotRelay;
272  }
273 
274  if (isCacheEntryObsoleteFor(timeslice)) {
275  pruneCacheSlotFor(timeslice);
276  }
277  if (hasCacheInputAlreadyFor(timeslice, input)) {
278  LOG(ERROR) << "Got a message with the same header and timeslice twice!!";
279  return WillNotRelay;
280  }
281  saveInSlot(timeslice, input);
282  return WillRelay;
283 }

286 std::vector<DataRelayer::RecordAction>
287 DataRelayer::getReadyToProcess() {
288  // THE STATE
289  std::vector<RecordAction> completed;
290  const auto &cache = mCache;
291  const auto numInputTypes = mDistinctRoutesIndex.size();
292  //
293  // THE IMPLEMENTATION DETAILS
294  //
295  // We use this to bail out early from the check as soon as we find something
296  // which we know is not complete.
297  auto getPartialRecord = [&cache, &numInputTypes](int li) -> gsl::span<const PartRef> {
298  auto offset = li * numInputTypes;
299  assert(cache.size() >= offset + numInputTypes);
300  auto const start = cache.data() + offset;
301  auto const end = cache.data() + offset + numInputTypes;
302  return gsl::span<const PartRef>(start, end);
303  };
304 
305  // These two are trivial, but in principle the whole loop could be parallelised
306  // or vectorised so "completed" could be a thread local variable which needs
307  // merging at the end.
308  auto updateCompletionResults = [&completed](TimesliceSlot li, CompletionPolicy::CompletionOp op) {
309  completed.push_back({li, op});
310  };
311 
312  auto completionResults = [&completed]() -> std::vector<RecordAction> {
313  return completed;
314  };
315 
316  // THE OUTER LOOP
317  //
318  // To determine if a line is complete, we iterate on all the arguments
319  // and check if they are ready. We do it this way, because in the end
320  // the number of inputs is going to be small and having a more complex
321  // structure will probably result in a larger footprint in any case.
322  // Also notice that ai == inputsNumber only when we reach the end of the
323  // iteration, that means we have found all the required bits.
324  assert(numInputTypes != 0);
325  size_t cacheLines = cache.size() / numInputTypes;
326  assert(cacheLines * numInputTypes == cache.size());
327 
328  for (size_t li = 0; li < cacheLines; ++li) {
329  TimesliceSlot slot{ li };
330  // We only check the cachelines which have been updated by an incoming
331  // message.
332  if (mTimesliceIndex.isDirty(slot) == false) {
333  continue;
334  }
335  auto partial = getPartialRecord(li);
336  auto action = mCompletionPolicy.callback(partial);
337  switch (action) {
338  case CompletionPolicy::CompletionOp::Consume:
339  case CompletionPolicy::CompletionOp::Process:
340  case CompletionPolicy::CompletionOp::Discard:
341  updateCompletionResults(slot, action);
342  break;
343  case CompletionPolicy::CompletionOp::Wait:
344  break;
345  }
346  // Given we have created an action for this cacheline, we need to wait for
347  // a new message before we look again into the given cacheline.
348  mTimesliceIndex.markAsDirty(slot, false);
349  }
350  return completionResults();
351 }

353 std::vector<std::unique_ptr<FairMQMessage>>
354  DataRelayer::getInputsForTimeslice(TimesliceSlot slot)
355 {
356  const auto numInputTypes = mDistinctRoutesIndex.size();
357  // State of the computation
358  std::vector<std::unique_ptr<FairMQMessage>> messages;
359  messages.reserve(numInputTypes*2);
360  auto& cache = mCache;
361  auto& index = mTimesliceIndex;
362  auto& metrics = mMetrics;
363 
364  // Nothing to see here, this is just to make the outer loop more understandable.
365  auto jumpToCacheEntryAssociatedWith = [](TimesliceSlot) {
366  return;
367  };
368 
369  // We move ownership so that the cache can be reused once the computation is
370  // finished. We bump by one the timeslice for the given cache entry, so that
371  // in case we get (for whatever reason) an old input, it will be
372  // automatically discarded by the relay method.
373  auto moveHeaderPayloadToOutput = [&messages, &cache, &index, &numInputTypes, &metrics](TimesliceSlot s, size_t arg) {
374  auto cacheId = s.index * numInputTypes + arg;
375  metrics.send({ 2, sMetricsNames[cacheId] });
376  messages.emplace_back(std::move(cache[cacheId].header));
377  messages.emplace_back(std::move(cache[cacheId].payload));
378  index.markAsObsolete(s);
379  };
380 
381  // An invalid set of arguments is a set of arguments associated to an invalid
382  // timeslice, so I can simply do that. I keep the assertion there because in principle
383  // we should have dispatched the timeslice already!
384  // FIXME: what happens when we have enough timeslices to hit the invalid one?
385  auto invalidateCacheFor = [&numInputTypes, &index, &cache](TimesliceSlot s) {
386  for (size_t ai = s.index * numInputTypes, ae = ai + numInputTypes; ai != ae; ++ai) {
387  assert(cache[ai].header.get() == nullptr);
388  assert(cache[ai].payload.get() == nullptr);
389  }
390  index.markAsInvalid(s);
391  };
392 
393  // Outer loop here.
394  jumpToCacheEntryAssociatedWith(slot);
395  for (size_t ai = 0, ae = numInputTypes; ai != ae; ++ai) {
396  moveHeaderPayloadToOutput(slot, ai);
397  }
398  invalidateCacheFor(slot);
399 
400  return std::move(messages);
401 }
