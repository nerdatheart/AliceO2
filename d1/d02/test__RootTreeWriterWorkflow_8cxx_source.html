37 constexpr int kTreeSize = 10; // elemants to send and write
38 DataProcessorSpec getSourceSpec()
39 {
40  auto initFct = [](InitContext& ic) {
41  auto counter = std::make_shared<int>();
42  *counter = 0;
43 
44  auto processingFct = [counter](ProcessingContext& pc) {
45  if (*counter >= kTreeSize) {
46  // don't publish more
47  return;
48  }
49  o2::test::Polymorphic a(*counter);
50  pc.outputs().snapshot(OutputRef{ "output" }, a);
51  int& metadata = pc.outputs().make<int>(Output{ "TST", "METADATA", 0, Lifetime::Timeframe });
52  metadata = *counter;
53  *counter = *counter + 1;
54  };
55 
56  return processingFct;
57  };
58  return DataProcessorSpec{ "source", // name of the processor
59  {},
60  { OutputSpec{ { "output" }, "TST", "SOMEOBJECT", 0, Lifetime::Timeframe },
61  OutputSpec{ { "meta" }, "TST", "METADATA", 0, Lifetime::Timeframe } },
62  AlgorithmSpec(initFct) };

68 {
69  auto initFct = [](InitContext& ic) {
70  std::string fileName = gSystem->TempDirectory();
71  fileName += "/test_RootTreeWriter.root";
72  using Polymorphic = o2::test::Polymorphic;
73  using WriterType = RootTreeWriter;
74  auto writer = std::make_shared<WriterType>(fileName.c_str(), // output file name
75  "testtree", // tree name
76  WriterType::BranchDef<Polymorphic>{ "input", "polyobject" },
77  WriterType::BranchDef<int>{ "meta", "counter" });
78  auto counter = std::make_shared<int>();
79  *counter = 0;
80 
81  // the callback to be set as hook at stop of processing for the framework
82  auto finishWriting = [writer]() { writer->close(); };
83  ic.services().get<CallbackService>().set(CallbackService::Id::Stop, finishWriting);
84 
85  auto processingFct = [writer, counter](ProcessingContext& pc) {
86  (*writer)(pc.inputs());
87  *counter = *counter + 1;
88  if (*counter >= kTreeSize) {
89  pc.services().get<ControlService>().readyToQuit(true);
90  }
91  };
92 
93  return processingFct;
94  };
95 
96  return DataProcessorSpec{ "sink",
97  { InputSpec{ "input", "TST", "SOMEOBJECT" }, //
98  InputSpec{ "meta", "TST", "METADATA" } }, //
99  Outputs{},
100  AlgorithmSpec(initFct) };
