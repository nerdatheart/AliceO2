55 void Dispatcher::run(ProcessingContext& ctx)
56 {
57  for (const auto& input : ctx.inputs()) {
58  if (input.header != nullptr && input.spec != nullptr) {
59 
60  for (auto& policy : mPolicies) {
61  // todo: consider getting the outputSpec in match to improve performance
62  // todo: consider matching (and deciding) in completion policy to save some time
63  if (policy->match(*input.spec) && policy->decide(input)) {
64 
65  if (!policy->getFairMQOutputChannel().empty()) {
66  sendFairMQ(ctx.services().get<RawDeviceService>().device(), input, policy->getFairMQOutputChannelName());
67  } else {
68  send(ctx.outputs(), input, policy->prepareOutput(*input.spec));
69  }
70  }
71  }
72  }
73  }

76 void Dispatcher::send(DataAllocator& dataAllocator, const DataRef& inputData, const Output& output) const
77 {
78  //todo: support other serialization methods
79  const auto* inputHeader = header::get<header::DataHeader*>(inputData.header);
80  if (inputHeader->payloadSerializationMethod == header::gSerializationMethodInvalid) {
81  LOG(WARNING) << "DataSampling::dispatcherCallback: input of origin'" << inputHeader->dataOrigin.str
82  << "', description '" << inputHeader->dataDescription.str
83  << "' has gSerializationMethodInvalid.";
84  } else if (inputHeader->payloadSerializationMethod == header::gSerializationMethodROOT) {
85  dataAllocator.adopt(output, DataRefUtils::as<TObject>(inputData).release());
86  } else { // POD
87  // todo: do it non-copy, when API is available
88  auto outputMessage = dataAllocator.newChunk(output, inputHeader->payloadSize);
89  memcpy(outputMessage.data, inputData.payload, inputHeader->payloadSize);
90  }

94 void Dispatcher::sendFairMQ(const FairMQDevice* device, const DataRef& inputData, const std::string& fairMQChannel) const
95 {
96  const auto* dh = header::get<header::DataHeader*>(inputData.header);
97  assert(dh);
98  const auto* dph = header::get<DataProcessingHeader*>(inputData.header);
99  assert(dph);
100 
101  header::DataHeader dhout{ dh->dataDescription, dh->dataOrigin, dh->subSpecification, dh->payloadSize };
102  dhout.payloadSerializationMethod = dh->payloadSerializationMethod;
103  DataProcessingHeader dphout{ dph->startTime, dph->duration };
104  o2::header::Stack headerStack{ dhout, dphout };
105 
106  auto channelAlloc = o2::memory_resource::getTransportAllocator(device->Transport());
107  FairMQMessagePtr msgHeaderStack = o2::memory_resource::getMessage(std::move(headerStack), channelAlloc);
108 
109  char* payloadCopy = new char[dh->payloadSize];
110  memcpy(payloadCopy, inputData.payload, dh->payloadSize);
111  auto cleanupFcn = [](void* data, void*) { delete[] reinterpret_cast<char*>(data); };
112  FairMQMessagePtr msgPayload(device->NewMessage(payloadCopy, dh->payloadSize, cleanupFcn, payloadCopy));
113 
114  FairMQParts message;
115  message.AddPart(move(msgHeaderStack));
116  message.AddPart(move(msgPayload));
117 
118  int64_t bytesSent = device->Send(message, fairMQChannel);
119 }

121 void Dispatcher::registerPath(const std::pair<InputSpec, OutputSpec>& path)
122 {
123  //todo: take care of inputs inclusive in others, when subSpec matchers are supported
124  auto cmp = [a = path.first](const InputSpec b)
125  {
126  return a.origin == b.origin && a.description == b.description && a.subSpec == b.subSpec && a.lifetime == b.lifetime;
127  };
128 
129  if (std::find_if(inputs.begin(), inputs.end(), cmp) == inputs.end()) {
130  inputs.push_back(path.first);
131  LOG(DEBUG) << "Registering input " << path.first.origin.str << " " << path.first.description.str << " "
132  << path.first.subSpec;
133  } else {
134  LOG(DEBUG) << "Input " << path.first.origin.str << " " << path.first.description.str << " " << path.first.subSpec
135  << " already registered";
136  }
137 
138  outputs.push_back(path.second);
139 }
