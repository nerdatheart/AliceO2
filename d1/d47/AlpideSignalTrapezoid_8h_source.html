64 inline float AlpideSignalTrapezoid::getCollectedCharge(float totalNEle, float tMin, float tMax) const
65 {
66  // calculate max number of electrons seen by the strobe from tMin to tMax (in nanosec),
67  // provided that the total injected charge was totalNEle electrons
68 
69  // estimate rise time for given charge
70  float riseTime = totalNEle > mChargeRise0 ? 0. : mMaxRiseTime * (1.f - totalNEle * mChargeRise0Inv);
71 
72  if (tMax >= riseTime && tMin <= mDuration) { // strobe overlaps flat top
73  return totalNEle;
74  }
75  if (tMax > 0. && tMin < riseTime) { // strobe overlaps with rise
76  return totalNEle * tMax / riseTime;
77  }
78  return 0;
79 }
80 }
