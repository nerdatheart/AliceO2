17 void InteractionSampler::init()
18 {
19  // (re-)initialize sample and check parameters consistency
20 
21  int nBCSet = mBCFilling.getNBunches();
22  if (!nBCSet) {
23  LOG(WARNING) << "No bunch filling provided, impose default one" << FairLogger::endl;
24  mBCFilling.setDefault();
25  nBCSet = mBCFilling.getNBunches();
26  }
27 
28  if (mMuBC < 0. && mIntRate < 0.) {
29  LOG(WARNING) << "No IR or muBC is provided, setting default IR" << FairLogger::endl;
30  mIntRate = DefIntRate;
31  }
32 
33  if (mMuBC > 0.) {
34  mIntRate = mMuBC * nBCSet * o2::constants::lhc::LHCRevFreq;
35  LOG(INFO) << "Deducing IR=" << mIntRate << "Hz from " << nBCSet << " BCs at mu=" << mMuBC << FairLogger::endl;
36  } else {
37  mMuBC = mIntRate / (nBCSet * o2::constants::lhc::LHCRevFreq);
38  LOG(INFO) << "Deducing mu=" << mMuBC << " per BC from IR=" << mIntRate << " with " << nBCSet << " BCs"
39  << FairLogger::endl;
40  }
41 
42  mBCMin = 0;
43  mBCMax = -1;
44  for (int i = 0; i < o2::constants::lhc::LHCMaxBunches; i++) {
45  if (mBCFilling.testBC(i)) {
46  if (mBCMin > i) {
47  mBCMin = i;
48  }
49  if (mBCMax < i) {
50  mBCMax = i;
51  }
52  }
53  }
54  double muexp = TMath::Exp(-mMuBC);
55  mProbNoInteraction = 1. - muexp;
56  mMuBCZTRed = mMuBC * muexp / mProbNoInteraction;
57  mBCCurrent = mBCMin + gRandom->Integer(mBCMax - mBCMin + 1);
58  mIntBCCache = 0;
59  mOrbit = 0;
60 }

75 o2::InteractionRecord InteractionSampler::generateCollisionTime()
76 {
77  // generate single interaction record
78  if (mIntRate < 0)
79  init();
80 
81  if (mIntBCCache < 1) { // do we still have interaction in current BC?
82  mIntBCCache = simulateInteractingBC(); // decide which BC interacts and N collisions
83  }
84  double timeInt = mTimeInBC.back() + o2::InteractionRecord::bc2ns(mBCCurrent, mOrbit);
85  mTimeInBC.pop_back();
86  mIntBCCache--;
87 
88  o2::InteractionRecord tmp(timeInt);
89 
90  return o2::InteractionRecord(timeInt);
91 }

94 int InteractionSampler::simulateInteractingBC()
95 {
96  // in order not to test random numbers for too many BC's with low mu,
97  // we estimate from very beginning how many BC's happen w/o interaction
98  // Returns number of collisions assigned to selected BC
99  double prob = gRandom->Rndm();
100  while (prob > 0.) { // skip BCs w/o interaction
101  nextCollidingBC(); // pick next interacting bunch
102  prob -= mProbNoInteraction;
103  }
104  // once BC is decided, enforce at least one interaction
105  int ncoll = genPoissonZT();
106  // assign random time withing a bunch
107  for (int i = ncoll; i--;) {
108  double tInBC = 0; // tInBC should be in the vicinity of the BC
109  do {
110  tInBC = gRandom->Gaus(mBCTimeRMS);
111  } while (std::abs(tInBC) > o2::constants::lhc::LHCBunchSpacingNS / 2.1);
112  mTimeInBC.push_back(tInBC);
113  }
114  if (ncoll > 1) { // sort in DECREASING time order (we are reading vector from the end)
115  std::sort(mTimeInBC.begin(), mTimeInBC.end(), [](const double a, const double b) { return a > b; });
116  }
117  return ncoll;
118 }
