42 void Clusterer::process(PixelReader& reader, std::vector<Cluster>* fullClus,
43  std::vector<CompClusterExt>* compClus, MCTruth* labelsCl)
44 {
45 
46 #ifdef _PERFORM_TIMING_
47  mTimer.Start(kFALSE);
48 #endif
49 
50  UInt_t prevROF = o2::ITSMFT::PixelData::DummyROF;
51  mClustersCount = compClus ? compClus->size() : (fullClus ? fullClus->size() : 0);
52 
53  while ((mChipData = reader.getNextChipData(mChips))) { // read next chip data to corresponding
54  // vector in the mChips and return the pointer on it
55 
56  mCurrROF = mChipData->getROFrame();
57  if (prevROF != mCurrROF && prevROF != o2::ITSMFT::PixelData::DummyROF) {
58  LOG(INFO) << "ITS: clusterizing new ROFrame " << mCurrROF << FairLogger::endl;
59  if (mClusTree) { // if necessary, flush existing data
60  flushClusters(fullClus, compClus, labelsCl);
61  }
62  }
63  prevROF = mCurrROF;
64 
65  mCurrChipID = mChipData->getChipID();
66  // LOG(DEBUG) << "ITSClusterer got Chip " << mCurrChipID << " ROFrame " << mChipData->getROFrame()
67  // << " Nhits " << mChipData->getData().size() << FairLogger::endl;
68 
69  if (mMaskOverflowPixels) { // mask pixels fired from the previous ROF
70  if (mChipsOld.size() < mChips.size()) {
71  mChipsOld.resize(mChips.size()); // expand buffer of previous ROF data
72  }
73  const auto& chipInPrevROF = mChipsOld[mCurrChipID];
74  if (chipInPrevROF.getROFrame() + 1 == mCurrROF) {
75  mChipData->maskFiredInSample(mChipsOld[mCurrChipID]);
76  }
77  }
78  auto validPixID = mChipData->getFirstUnmasked();
79  if (validPixID < mChipData->getData().size()) { // chip data may have all of its pixels masked!
80  initChip(validPixID++);
81  for (; validPixID < mChipData->getData().size(); validPixID++) {
82  if (!mChipData->getData()[validPixID].isMasked()) {
83  updateChip(validPixID);
84  }
85  }
86  finishChip(fullClus, compClus, reader.getDigitsMCTruth(), labelsCl);
87  }
88  if (mMaskOverflowPixels) { // current chip data will be used in the next ROF to mask overflow pixels
89  mChipsOld[mCurrChipID].swap(*mChipData);
90  }
91  }
92 
93  // if asked, flush last ROF
94  if (mClusTree && prevROF != o2::ITSMFT::PixelData::DummyROF) { // if necessary, flush existing data
95  flushClusters(fullClus, compClus, labelsCl);
96  }
97 
98 #ifdef _PERFORM_TIMING_
99  mTimer.Stop();
100  printf("Clusterization timing (w/o disk IO): ");
101  mTimer.Print();
102 #endif
103 }

106 void Clusterer::initChip(UInt_t first)
107 {
108  // init chip with the 1st unmasked pixel (entry "from" in the mChipData)
109  mPrev = mColumn1 + 1;
110  mCurr = mColumn2 + 1;
111  resetColumn(mCurr);
112 
113  mPixels.clear();
114  mPreClusterHeads.clear();
115  mPreClusterIndices.clear();
116  auto pix = mChipData->getData()[first];
117  mCol = pix.getCol();
118 
119  //addNewPrecluster(first, pix.getRowDirect()); // save on .size() calls ?
120  mCurr[pix.getRowDirect()] = 0; // can use getRowDirect since the pixel is not masked
121  // start the first pre-cluster
122  mPreClusterHeads.push_back(0);
123  mPreClusterIndices.push_back(0);
124  mPixels.emplace_back(-1, first); // id of current pixel
125  mNoLeftColumn = true; // flag that there is no column on the left to check yet
126 }

129 void Clusterer::updateChip(UInt_t ip)
130 {
131  const auto pix = mChipData->getData()[ip];
132  UShort_t row = pix.getRowDirect(); // can use getRowDirect since the pixel is not masked
133  if (mCol != pix.getCol()) { // switch the buffers
134  swapColumnBuffers();
135  resetColumn(mCurr);
136  mNoLeftColumn = false;
137  if (pix.getCol() > mCol + 1) {
138  // no connection with previous column, this pixel cannot belong to any of the
139  // existing preclusters, create a new precluster and flag to check only the row above for next pixels of this column
140  mCol = pix.getCol();
141  addNewPrecluster(ip, row);
142  mNoLeftColumn = true;
143  return;
144  }
145  mCol = pix.getCol();
146  }
147 
148  Bool_t orphan = true;
149 
150  if (mNoLeftColumn) { // check only the row above
151  if (mCurr[row - 1] >= 0) {
152  expandPreCluster(ip, row, mCurr[row - 1]); // attach to the precluster of the previous row
153  return;
154  }
155  } else {
156  int neighbours[]{ mCurr[row - 1], mPrev[row], mPrev[row + 1], mPrev[row - 1] };
157  for (auto pci : neighbours) {
158  if (pci < 0) {
159  continue;
160  }
161  if (orphan) {
162  expandPreCluster(ip, row, pci); // attach to the adjascent precluster
163  orphan = false;
164  continue;
165  }
166  // reassign precluster index to smallest one
167  if (mPreClusterIndices[pci] < mPreClusterIndices[mCurr[row]]) {
168  mPreClusterIndices[mCurr[row]] = mPreClusterIndices[pci];
169  } else {
170  mPreClusterIndices[pci] = mPreClusterIndices[mCurr[row]];
171  }
172  }
173  }
174  if (orphan) {
175  addNewPrecluster(ip, row); // start new precluster
176  }
177 }

180 void Clusterer::finishChip(std::vector<Cluster>* fullClus, std::vector<CompClusterExt>* compClus,
181  const MCTruth* labelsDig, MCTruth* labelsClus)
182 {
183  constexpr Float_t SigmaX2 = Segmentation::PitchRow * Segmentation::PitchRow / 12.; // FIXME
184  constexpr Float_t SigmaY2 = Segmentation::PitchCol * Segmentation::PitchCol / 12.; // FIXME
185 
186  const auto& pixData = mChipData->getData();
187 
188  for (int i1 = 0; i1 < mPreClusterHeads.size(); ++i1) {
189  const auto ci = mPreClusterIndices[i1];
190  if (ci < 0) {
191  continue;
192  }
193  UShort_t rowMax = 0, rowMin = 65535;
194  UShort_t colMax = 0, colMin = 65535;
195  int nlab = 0, npix = 0;
196  int next = mPreClusterHeads[i1];
197  while (next >= 0) {
198  const auto& pixEntry = mPixels[next];
199  const auto pix = pixData[pixEntry.second];
200  if (npix < mPixArrBuff.size()) {
201  mPixArrBuff[npix++] = pix; // needed for cluster topology
202  adjustBoundingBox(pix, rowMin, rowMax, colMin, colMax);
203  if (labelsClus) { // the MCtruth for this pixel is at mChipData->startID+pixEntry.second
204  fetchMCLabels(pixEntry.second + mChipData->getStartID(), labelsDig, nlab);
205  }
206  next = pixEntry.first;
207  } else {
208  LOG(ERROR) << "Cluster size " << npix + 1 << " exceeds the buffer size" << FairLogger::endl;
209  }
210  }
211  mPreClusterIndices[i1] = -1;
212  for (int i2 = i1 + 1; i2 < mPreClusterHeads.size(); ++i2) {
213  if (mPreClusterIndices[i2] != ci) {
214  continue;
215  }
216  next = mPreClusterHeads[i2];
217  while (next >= 0) {
218  const auto& pixEntry = mPixels[next];
219  const auto pix = pixData[pixEntry.second]; // PixelData
220  if (npix < mPixArrBuff.size()) {
221  mPixArrBuff[npix++] = pix; // needed for cluster topology
222  adjustBoundingBox(pix, rowMin, rowMax, colMin, colMax);
223  if (labelsClus) { // the MCtruth for this pixel is at mChipData->startID+pixEntry.second
224  fetchMCLabels(pixEntry.second + mChipData->getStartID(), labelsDig, nlab);
225  }
226  next = pixEntry.first;
227  } else {
228  LOG(ERROR) << "Cluster size " << npix + 1 << " exceeds the buffer size" << FairLogger::endl;
229  }
230  }
231  mPreClusterIndices[i2] = -1;
232  }
233  UShort_t rowSpan = rowMax - rowMin + 1, colSpan = colMax - colMin + 1;
234  Cluster clus;
235  clus.setROFrame(mChipData->getROFrame());
236  clus.setSensorID(mChipData->getChipID());
237  clus.setNxNzN(rowSpan, colSpan, npix);
238 #ifdef _ClusterTopology_
239  UShort_t colSpanW = colSpan, rowSpanW = rowSpan;
240  if (colSpan * rowSpan > Cluster::kMaxPatternBits) { // need to store partial info
241  // will curtail largest dimension
242  if (colSpan > rowSpan) {
243  if ((colSpanW = Cluster::kMaxPatternBits / rowSpan) == 0) {
244  colSpanW = 1;
245  rowSpanW = Cluster::kMaxPatternBits;
246  }
247  } else {
248  if ((rowSpanW = Cluster::kMaxPatternBits / colSpan) == 0) {
249  rowSpanW = 1;
250  colSpanW = Cluster::kMaxPatternBits;
251  }
252  }
253  }
254  clus.setPatternRowSpan(rowSpanW, rowSpanW < rowSpan);
255  clus.setPatternColSpan(colSpanW, colSpanW < colSpan);
256  clus.setPatternRowMin(rowMin);
257  clus.setPatternColMin(colMin);
258  for (int i = 0; i < npix; i++) {
259  const auto pix = mPixArrBuff[i];
260  unsigned short ir = pix.getRowDirect() - rowMin, ic = pix.getCol() - colMin;
261  if (ir < rowSpanW && ic < colSpanW) {
262  clus.setPixel(ir, ic);
263  }
264  }
265 #endif //_ClusterTopology_
266  if (fullClus) { // do we need conventional clusters with full topology and coordinates?
267  fullClus->push_back(clus);
268  Cluster& c = fullClus->back();
269  Float_t x = 0., z = 0.;
270  for (int i = npix; i--;) {
271  x += mPixArrBuff[i].getRowDirect();
272  z += mPixArrBuff[i].getCol();
273  }
274  Point3D<float> xyzLoc(Segmentation::getFirstRowCoordinate() + x * Segmentation::PitchRow / npix, 0.f,
275  Segmentation::getFirstColCoordinate() + z * Segmentation::PitchCol / npix);
276  auto xyzTra =
277  mGeometry->getMatrixT2L(mChipData->getChipID()) ^ (xyzLoc); // inverse transform from Local to Tracking frame
278  c.setPos(xyzTra);
279  c.setErrors(SigmaX2, SigmaY2, 0.f);
280  }
281 
282  if (compClus) { // store compact clusters
283  UShort_t pattID = clus.getNPix(); // HERE we need to attach calculated pattern ID, right now use just Ncl
284  compClus->emplace_back(rowMin, colMin, pattID, mChipData->getChipID(), mChipData->getROFrame());
285  }
286 
287  if (labelsClus) { // MC labels were requested
288  for (int i = nlab; i--;) {
289  labelsClus->addElement(mClustersCount, mLabelsBuff[i]);
290  }
291  }
292 
293  mClustersCount++;
294  }
295 }

298 void Clusterer::fetchMCLabels(int digID, const MCTruth* labelsDig, int& nfilled)
299 {
300  // transfer MC labels to cluster
301  if (nfilled >= Cluster::maxLabels) {
302  return;
303  }
304  const auto& lbls = labelsDig->getLabels(digID);
305  for (int i = lbls.size(); i--;) {
306  int ic = nfilled;
307  for (; ic--;) { // check if the label is already present
308  if (mLabelsBuff[ic] == lbls[i]) {
309  return; // label is found, do nothing
310  }
311  }
312  mLabelsBuff[nfilled++] = lbls[i];
313  if (nfilled >= Cluster::maxLabels) {
314  break;
315  }
316  }
317  //
318 }
