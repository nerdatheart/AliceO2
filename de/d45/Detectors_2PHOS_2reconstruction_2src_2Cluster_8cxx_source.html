107 void Cluster::Purify(double threshold)
108 {
109  // Removes digits below threshold
110 
111  std::vector<float>::iterator itE = mEnergyList.begin();
112  std::vector<int>::iterator itId = mDigitsIdList.begin();
113  std::vector<int>::iterator itTime = mTimeList.begin();
114  std::vector<int>::iterator jtId;
115  while (itE != mEnergyList.end()) {
116  if (*itE < threshold) {
117  itE = mEnergyList.erase(itE);
118  itId = mDigitsIdList.erase(itId);
119  itTime = mTimeList.erase(itTime);
120  } else {
121  itId++;
122  itE++;
123  itTime++;
124  }
125  }
126 
127  mMulDigit = mEnergyList.size();
128 
129  if (mMulDigit == 0) { // too soft cluster
130  mFullEnergy = 0.;
131  return;
132  }
133 
134  // Remove non-connected cells
135  if (mMulDigit > 1) {
136  mFullEnergy = 0.; // Recalculate total energy
137  itE = mEnergyList.begin();
138  for (itId = mDigitsIdList.begin(); itId != mDigitsIdList.end();) {
139  bool hasNeighbours = false;
140  for (jtId = mDigitsIdList.begin(); jtId != mDigitsIdList.end(); jtId++) {
141  if (itId == jtId) {
142  continue;
143  }
144  if (mPHOSGeom->AreNeighbours(*itId, *jtId) == 1) {
145  hasNeighbours = true;
146  break;
147  }
148  }
149  if (!hasNeighbours) {
150  itE = mEnergyList.erase(itE);
151  itId = mDigitsIdList.erase(itId);
152  itTime = mTimeList.erase(itId);
153  } else {
154  mFullEnergy += *itE;
155  ++itId;
156  ++itTime;
157  ++itE;
158  }
159  }
160  } else {
161  mFullEnergy = mEnergyList[0];
162  }
163 
164  mDigitsIdList.shrink_to_fit();
165  mTimeList.shrink_to_fit();
166  mEnergyList.shrink_to_fit();
167 }

169 void Cluster::EvalCoreEnergy(double coreRadius)
170 {
171  // This function calculates energy in the core,
172  // i.e. within a radius coreRadius (~3cm) around the center. Beyond this radius
173  // in accordance with shower profile the energy deposition
174  // should be less than 2%
175 
176  if (mLocalPosX < -900.) // local position was not calculated yiet
177  {
178  EvalLocalPosition();
179  }
180 
181  std::vector<float>::iterator itE = mEnergyList.begin();
182  for (std::vector<int>::iterator i = mDigitsIdList.begin(); i != mDigitsIdList.end(); i++) {
183  double xi = 0., zi = 0.;
184  mPHOSGeom->AbsIdToRelPosInModule(*i, xi, zi);
185  Float_t distance = std::sqrt((xi - mLocalPosX) * (xi - mLocalPosX) + (zi - mLocalPosZ) * (zi - mLocalPosX));
186  if (distance < coreRadius) {
187  mCoreEnergy += *itE;
188  }
189  ++itE;
190  }
191 }

193 void Cluster::EvalLocalPosition()
194 {
195  // Calculates the center of gravity in the local PHOS-module coordinates
196  // Note that correction for non-perpendicular incidence will be applied later
197  // when vertex will be known.
198 
199  if (mFullEnergy <= 0.) { // zero energy cluster, position undefined
200  mLocalPosX = -999.;
201  mLocalPosZ = -999.;
202  return;
203  }
204 
205  // find module number
206  mModule = mPHOSGeom->AbsIdToModule(mDigitsIdList[0]);
207 
208  double wtot = 0.;
209  mLocalPosX = 0.;
210  mLocalPosZ = 0.;
211  std::vector<float>::iterator itE = mEnergyList.begin();
212  for (std::vector<int>::iterator i = mDigitsIdList.begin(); i != mDigitsIdList.end(); i++) {
213  double xi = 0., zi = 0.;
214  mPHOSGeom->AbsIdToRelPosInModule(*i, xi, zi);
215  if (*itE > 0) {
216  double w = std::max(0., kLogWeight + std::log(*itE / mFullEnergy));
217  mLocalPosX += xi * w;
218  mLocalPosZ += zi * w;
219  wtot += w;
220  }
221  ++itE;
222  }
223  if (wtot > 0) {
224  mLocalPosX /= wtot;
225  mLocalPosZ /= wtot;
226  }
227 }

229 void Cluster::EvalDispersion()
230 { // computes the dispersion of the shower
231 
232  mDispersion = 0.;
233  if (mFullEnergy <= 0.) { // zero energy cluster, dispersion undefined
234  return;
235  }
236  if (mLocalPosX < -900.) // local position was not calculated yiet
237  {
238  EvalLocalPosition();
239  }
240 
241  double wtot = 0.;
242  std::vector<float>::iterator itE = mEnergyList.begin();
243  for (std::vector<int>::iterator i = mDigitsIdList.begin(); i != mDigitsIdList.end(); i++) {
244  double xi = 0., zi = 0.;
245  mPHOSGeom->AbsIdToRelPosInModule(*i, xi, zi);
246  if (*itE > 0) {
247  double w = std::max(0., kLogWeight + std::log(*itE / mFullEnergy));
248  mDispersion += w * ((xi - mLocalPosX) * (xi - mLocalPosX) + (zi - mLocalPosZ) * (zi - mLocalPosZ));
249  wtot += w;
250  }
251  ++itE;
252  }
253 
254  if (wtot > 0) {
255  mDispersion /= wtot;
256  }
257 
258  if (mDispersion >= 0)
259  mDispersion = std::sqrt(mDispersion);
260  else
261  mDispersion = 0.;
262 }

264 void Cluster::EvalElipsAxis()
265 { // computes the axis of shower ellipsoide
266  // Calculates the axis of the shower ellipsoid
267 
268  if (mFullEnergy <= 0.) { // zero energy cluster, dispersion undefined
269  mLambdaLong = mLambdaShort = 0.;
270  return;
271  }
272 
273  double wtot = 0., x = 0., z = 0., dxx = 0., dxz = 0., dzz = 0.;
274  std::vector<float>::iterator itE = mEnergyList.begin();
275  for (std::vector<int>::iterator i = mDigitsIdList.begin(); i != mDigitsIdList.end(); i++) {
276  double xi = 0., zi = 0.;
277  mPHOSGeom->AbsIdToRelPosInModule(*i, xi, zi);
278  if (*itE > 0) {
279  double w = std::max(0., kLogWeight + std::log(*itE / mFullEnergy));
280  dxx += w * xi * xi;
281  x += w * xi;
282  dzz += w * zi * zi;
283  z += w * zi;
284  dxz += w * xi * zi;
285  wtot += w;
286  }
287  ++itE;
288  }
289  if (wtot > 0) {
290  dxx /= wtot;
291  x /= wtot;
292  dxx -= x * x;
293  dzz /= wtot;
294  z /= wtot;
295  dzz -= z * z;
296  dxz /= wtot;
297  dxz -= x * z;
298 
299  mLambdaLong = 0.5 * (dxx + dzz) + std::sqrt(0.25 * (dxx - dzz) * (dxx - dzz) + dxz * dxz);
300  if (mLambdaLong > 0)
301  mLambdaLong = std::sqrt(mLambdaLong);
302 
303  mLambdaShort = 0.5 * (dxx + dzz) - std::sqrt(0.25 * (dxx - dzz) * (dxx - dzz) + dxz * dxz);
304  if (mLambdaShort > 0) // To avoid exception if numerical errors lead to negative lambda.
305  mLambdaShort = std::sqrt(mLambdaShort);
306  else
307  mLambdaShort = 0.;
308  } else {
309  mLambdaLong = mLambdaShort = 0.;
310  }
311 }

313 void Cluster::EvalPrimaries(const std::vector<Digit>* digits)
314 {
315  // Constructs the list of primary particles (tracks) which have contributed to this RecPoint
316 
317  if (mFullEnergy <= 0.) { // zero energy cluster, skip primary calculation
318  return;
319  }
320  std::vector<float>::iterator itE = mEnergyList.begin();
321  for (std::vector<int>::iterator i = mDigitsIdList.begin(); i != mDigitsIdList.end(); i++) {
322 
323  Digit testDigit(*i, 0., 0., 0); // Digit with correct AbsId
324  std::vector<Digit>::const_iterator foundIt = BinarySearch(digits, testDigit); // Find this digit in the total list
325  if (foundIt == digits->end()) { // should not happen
326  continue;
327  }
328 
329  for (int iPrim = 0; iPrim < (*foundIt).getNLabels(); iPrim++) {
330  Label lab = (*foundIt).getLabel(iPrim);
331  double eProp = (*foundIt).getLabelEProp(iPrim);
332  // look through list of primaries, add proportion of energy if already there, else create new entry
333  int primListSize = mLabels.size();
334  bool found = false;
335  for (int inPrimList = 0; inPrimList < primListSize; inPrimList++) {
336  if (lab == mLabels[inPrimList]) { // already exist, increase proportin of energy
337  mLabelsEProp[inPrimList] += eProp * foundIt->getAmplitude() / mFullEnergy;
338  found = true;
339  break;
340  }
341  }
342  if (!found) {
343  mLabelsEProp.push_back(lab);
344  mLabelsEProp.push_back(eProp * ((*foundIt).getAmplitude()) / mFullEnergy);
345  }
346  }
347  }
348  // Vectors will not be modified any more.
349  mLabels.shrink_to_fit();
350  mLabelsEProp.shrink_to_fit();
351 }

370 std::vector<Digit>::const_iterator Cluster::BinarySearch(const std::vector<Digit>* container, Digit& element)
371 {
372  const std::vector<Digit>::const_iterator endIt = container->end();
373 
374  std::vector<Digit>::const_iterator left = container->begin();
375  std::vector<Digit>::const_iterator right = endIt;
376 
377  if (container->size() == 0 || container->front() > element || container->back() < element) {
378  return endIt;
379  }
380 
381  while (distance(left, right) > 0) {
382  const std::vector<Digit>::const_iterator mid = left + distance(left, right) / 2;
383 
384  if (element > *mid) {
385  left = mid + 1;
386  } else {
387  right = mid;
388  }
389  }
390 
391  if (*right == element) {
392  return right;
393  }
394 
395  return endIt;
396 }
