147  // add clusters from a flattened buffer starting with ClusterGroupAttribute followed by the
148  // array of ClusterNative, the number of clusters is determined from the size of the buffer
149  // FIXME: add mc labels
150  template <typename AttributeT = ClusterGroupAttribute>
151  void addFlatBuffer(unsigned char* buffer, size_t size)
152  {
153  if (buffer == nullptr || size < sizeof(AttributeT) || (size - sizeof(AttributeT)) % sizeof(ClusterNative) != 0) {
154  // this is not a valid message
155  throw std::runtime_error("incompatible message size");
156  }
157  const auto& groupAttribute = *reinterpret_cast<AttributeT*>(buffer);
158  auto nofClusters = (size - sizeof(AttributeT)) / sizeof(ClusterNative);
159  auto ptrClusters = reinterpret_cast<ClusterNative*>(buffer + sizeof(groupAttribute));
160  clusters[groupAttribute.sector][groupAttribute.globalPadRow] = ptrClusters;
161  nClusters[groupAttribute.sector][groupAttribute.globalPadRow] = nofClusters;
162  }
163 };
