49 ExpirationHandler::Checker LifetimeHelpers::expireTimed(std::chrono::milliseconds period)
50 {
51  auto start = getCurrentTime();
52  auto last = std::make_shared<decltype(start)>(start);
53  return [last, period](int64_t) -> bool {
54  auto current = getCurrentTime();
55  auto delta = current - *last;
56  if (delta > period.count()) {
57  *last = current;
58  return true;
59  }
60  return false;
61  };

110 ExpirationHandler::Handler LifetimeHelpers::enumerate(InputSpec const& matcher, std::string const& sourceChannel)
111 {
112  auto counter = std::make_shared<int64_t>(0);
113  auto f = [matcher, counter, sourceChannel](ServiceRegistry& services, PartRef& ref, uint64_t timestamp) -> void {
114  // We should invoke the handler only once.
115  assert(!ref.header);
116  assert(!ref.payload);
117  auto& rawDeviceService = services.get<RawDeviceService>();
118 
119  DataHeader dh;
120  dh.dataOrigin = matcher.origin;
121  dh.dataDescription = matcher.description;
122  dh.subSpecification = matcher.subSpec;
123  dh.payloadSize = 8;
124  dh.payloadSerializationMethod = gSerializationMethodNone;
125 
126  DataProcessingHeader dph{ timestamp, 1 };
127 
128  auto&& transport = rawDeviceService.device()->GetChannel(sourceChannel, 0).Transport();
129  auto channelAlloc = o2::memory_resource::getTransportAllocator(transport);
130  auto header = o2::memory_resource::getMessage(o2::header::Stack{ channelAlloc, dh, dph });
131  ref.header = std::move(header);
132 
133  auto payload = rawDeviceService.device()->NewMessage(*counter);
134  ref.payload = std::move(payload);
135  (*counter)++;
136  };
137  return f;
138 }
