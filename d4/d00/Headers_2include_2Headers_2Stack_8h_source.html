128  template <typename T, typename... Args>
129  static o2::byte* inject(o2::byte* here, T&& h, Args&&... args) noexcept
130  {
131  auto alsohere = inject(here, h);
132  // the type might be a stack itself, loop through headers and set the flag in the last one
133  if (h.size() > 0) {
134  BaseHeader* next = BaseHeader::get(here);
135  while (next->flagsNextHeader) {
136  next = next->next();
137  }
138  next->flagsNextHeader = hasNonEmptyArg(args...);
139  }
140  return inject(alsohere, args...);
141  }
