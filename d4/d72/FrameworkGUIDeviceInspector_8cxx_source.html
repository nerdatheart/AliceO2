46 void optionsTable(const DeviceSpec& spec, const DeviceControl& control)
47 {
48  if (spec.options.empty()) {
49  return;
50  }
51  if (ImGui::CollapsingHeader("Options:", ImGuiTreeNodeFlags_DefaultOpen)) {
52  ImGui::Columns(2);
53  auto labels = { "Name", "Value" };
54  for (auto& label : labels) {
55  ImGui::TextUnformatted(label);
56  ImGui::NextColumn();
57  }
58  for (auto& option : spec.options) {
59  ImGui::TextUnformatted(option.name.c_str());
60  ImGui::NextColumn();
61  auto currentValueIt = control.options.find(option.name);
62 
63  // Did not find the option
64  if (currentValueIt == control.options.end()) {
65  switch (option.type) {
66  case VariantType::String:
67  ImGui::Text(R"("%s" (default))", option.defaultValue.get<const char*>());
68  break;
69  case VariantType::Int:
70  ImGui::Text("%d (default)", option.defaultValue.get<int>());
71  break;
72  case VariantType::Float:
73  ImGui::Text("%f (default)", option.defaultValue.get<float>());
74  break;
75  case VariantType::Double:
76  ImGui::Text("%f (default)", option.defaultValue.get<double>());
77  break;
78  case VariantType::Empty:
79  ImGui::TextUnformatted(""); // no default value
80  default:
81  ImGui::TextUnformatted("unknown");
82  }
83  } else {
84  ImGui::TextUnformatted(currentValueIt->second.c_str());
85  }
86  ImGui::NextColumn();
87  }
88  }
89  ImGui::Columns(1);
90 }
