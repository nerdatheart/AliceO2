304  bool match(header::DataHeader const& d, std::vector<ContextElement>& context) const
305  {
306  bool leftValue = false, rightValue = false;
307 
308  // FIXME: Using std::visit is not API compatible due to a new
309  // exception being thrown. This is the ABI compatible version.
310  // Replace with:
311  //
312  // auto eval = [&d](auto&& arg) -> bool {
313  // using T = std::decay_t<decltype(arg)>;
314  // if constexpr (std::is_same_v<T, std::unique_ptr<DataDescriptorMatcher>>) {
315  // return arg->match(d, context);
316  // if constexpr (std::is_same_v<T, ConstantValueMatcher>) {
317  // return arg->match(d);
318  // } else {
319  // return arg.match(d, context);
320  // }
321  // };
322  // switch (mOp) {
323  // case Op::Or:
324  // return std::visit(eval, mLeft) || std::visit(eval, mRight);
325  // case Op::And:
326  // return std::visit(eval, mLeft) && std::visit(eval, mRight);
327  // case Op::Xor:
328  // return std::visit(eval, mLeft) ^ std::visit(eval, mRight);
329  // case Op::Just:
330  // return std::visit(eval, mLeft);
331  // }
332  // When we drop support for macOS 10.13
333  if (auto pval0 = std::get_if<OriginValueMatcher>(&mLeft)) {
334  leftValue = pval0->match(d, context);
335  } else if (auto pval1 = std::get_if<DescriptionValueMatcher>(&mLeft)) {
336  leftValue = pval1->match(d, context);
337  } else if (auto pval2 = std::get_if<SubSpecificationTypeValueMatcher>(&mLeft)) {
338  leftValue = pval2->match(d, context);
339  } else if (auto pval3 = std::get_if<std::unique_ptr<DataDescriptorMatcher>>(&mLeft)) {
340  leftValue = (*pval3)->match(d, context);
341  } else if (auto pval4 = std::get_if<ConstantValueMatcher>(&mLeft)) {
342  leftValue = pval4->match(d);
343  } else {
344  throw std::runtime_error("Bad parsing tree");
345  }
346  // Common speedup.
347  if (mOp == Op::And && leftValue == false) {
348  return false;
349  }
350  if (mOp == Op::Or && leftValue == true) {
351  return true;
352  }
353  if (mOp == Op::Just) {
354  return leftValue;
355  }
356 
357  if (auto pval0 = std::get_if<OriginValueMatcher>(&mRight)) {
358  rightValue = pval0->match(d, context);
359  } else if (auto pval1 = std::get_if<DescriptionValueMatcher>(&mRight)) {
360  rightValue = pval1->match(d, context);
361  } else if (auto pval2 = std::get_if<SubSpecificationTypeValueMatcher>(&mRight)) {
362  rightValue = pval2->match(d, context);
363  } else if (auto pval3 = std::get_if<std::unique_ptr<DataDescriptorMatcher>>(&mRight)) {
364  rightValue = (*pval3)->match(d, context);
365  } else if (auto pval4 = std::get_if<ConstantValueMatcher>(&mRight)) {
366  rightValue = pval4->match(d);
367  }
368  // There are cases in which not having a rightValue might be legitimate,
369  // so we do not throw an exception.
370  switch (mOp) {
371  case Op::Or:
372  return leftValue || rightValue;
373  case Op::And:
374  return leftValue && rightValue;
375  case Op::Xor:
376  return leftValue ^ rightValue;
377  case Op::Just:
378  return leftValue;
379  }
380  };
