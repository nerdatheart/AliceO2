137 void someDataProducerAlgorithm(ProcessingContext& ctx)
138 {
139  size_t index = ctx.services().get<ParallelContext>().index1D();
140  sleep(1);
141  // Creates a new message of size collectionChunkSize which
142  // has "TPC" as data origin and "CLUSTERS" as data description.
143  auto tpcClusters = ctx.outputs().make<FakeCluster>(Output{ "TPC", "CLUSTERS", index }, collectionChunkSize);
144  int i = 0;
145 
146  for (auto& cluster : tpcClusters) {
147  assert(i < collectionChunkSize);
148  cluster.x = index;
149  cluster.y = i;
150  cluster.z = i;
151  cluster.q = rand() % 1000;
152  i++;
153  }

156 void someProcessingStageAlgorithm(ProcessingContext& ctx)
157 {
158  size_t index = ctx.services().get<ParallelContext>().index1D();
159 
160  const FakeCluster* inputDataTpc = reinterpret_cast<const FakeCluster*>(ctx.inputs().get("dataTPC").payload);

162  ctx.outputs().make<FakeCluster>(Output{ "TPC", "CLUSTERS_P", index }, collectionChunkSize);
163 
164  int i = 0;
165  for (auto& cluster : processedTpcClusters) {
166  assert(i < collectionChunkSize);

168  cluster.y = 2 * inputDataTpc[i].y;
169  cluster.z = inputDataTpc[i].z * inputDataTpc[i].q;
170  cluster.q = inputDataTpc[i].q;
171  i++;
172  }
