79 void displayHistory(const DeviceInfo& info, DeviceControl& control)
80 {
81  if (info.history.empty()) {
82  return;
83  }
84  int startPos = info.historyPos;
85  const int historySize = info.history.size();
86 
87  int triggerStartPos = startPos + 1 % historySize;
88  int triggerStopPos = startPos % historySize;
89 
90  int j = startPos;
91  // We look for a stop trigger, so that we know where to stop the search for
92  // out start search. If no stop trigger is found, we search until the end
93  if (control.logStopTrigger[0]) {
94  while ((j % historySize) != ((startPos + 1) % historySize)) {
95  assert(j >= 0);
96  assert(j < historySize);
97  auto& line = info.history[j];
98  if (strstr(line.c_str(), control.logStopTrigger)) {
99  triggerStopPos = (j + 1) % historySize;
100  break;
101  }
102  // Wrap in case we end up below 0
103  j = (j == 0) ? historySize - 1 : j - 1;
104  }
105  }
106 
107  // Look for the last instance of the start trigger before the
108  // last stop trigger.
109  j = startPos + 1;
110  if (control.logStartTrigger[0]) {
111  while ((j % historySize) != triggerStopPos) {
112  assert(historySize > j);
113  assert(historySize == 1000);
114  auto& line = info.history[j];
115  if (strstr(line.c_str(), control.logStartTrigger)) {
116  triggerStartPos = j;
117  }
118  j = (j + 1) % historySize;
119  }
120  }
121 
122  // We start from the last trigger found. Eventually this is the first
123  // line in the ring buffer, if no trigger is specified.
124  size_t ji = triggerStartPos % historySize;
125  size_t je = triggerStopPos % historySize;
126  size_t iterations = 0;
127  while (historySize && ((ji % historySize) != je)) {
128  assert(iterations < historySize);
129  iterations++;
130  assert(historySize == 1000);
131  assert(ji < historySize);
132  auto& line = info.history[ji];
133  auto logLevel = info.historyLevel[ji];
134 
135  // Skip empty lines
136  if (line.empty()) {
137  ji = (ji + 1) % historySize;
138  continue;
139  }
140  // Print matching lines
141  if (strstr(line.c_str(), control.logFilter) != nullptr) {
142  auto color = colorForLogLevel(logLevel);
143  // We filter twice, once on input, to reduce the
144  // stream, a second time at display time, to avoid
145  // showing unrelevant messages from past.
146  if (logLevel >= control.logLevel) {
147  ImGui::TextColored(color, line.c_str(), line.c_str() + line.size());
148  }
149  }
150  ji = (ji + 1) % historySize;
151  }
152 }

162 void displayDeviceMetrics(const char* label, ImVec2 canvasSize, std::string const& selectedMetricName,
163  size_t rangeBegin, size_t rangeEnd, size_t bins, bool useHistos,
164  std::vector<DeviceSpec> const& specs, std::vector<DeviceMetricsInfo> const& metricsInfos)
165 {
166  std::vector<void const*> metricsToDisplay;
167  std::vector<const char*> deviceNames;
168  std::vector<MultiplotData> userData;
169  std::vector<ImColor> colors;
170  MetricType metricType;
171  size_t metricSize = 0;
172  assert(specs.size() == metricsInfos.size());
173  float maxValue = std::numeric_limits<float>::lowest();
174  float minValue = 0;
175 
176  for (int mi = 0; mi < metricsInfos.size(); ++mi) {
177  auto vi = DeviceMetricsHelper::metricIdxByName(selectedMetricName, metricsInfos[mi]);
178  if (vi == metricsInfos[mi].metricLabelsIdx.size()) {
179  continue;
180  }
181  auto& metric = metricsInfos[mi].metrics[vi];
182  deviceNames.push_back(specs[mi].name.c_str());
183  colors.emplace_back(220, 220, 220);
184  metricType = metric.type;
185  MultiplotData data;
186  data.mod = metricsInfos[mi].timestamps[vi].size();
187  data.first = metric.pos - data.mod;
188  minValue = std::min(minValue, metricsInfos[mi].min[vi]);
189  maxValue = std::max(maxValue, metricsInfos[mi].max[vi]);
190  switch (metric.type) {
191  case MetricType::Int: {
192  data.size = metricsInfos[mi].intMetrics[metric.storeIdx].size();
193  data.points = metricsInfos[mi].intMetrics[metric.storeIdx].data();
194  data.type = MetricType::Int;
195  metricType = MetricType::Int;
196  metricSize = metricsInfos[mi].intMetrics[metric.storeIdx].size();
197  } break;
198  case MetricType::Float: {
199  data.size = metricsInfos[mi].floatMetrics[metric.storeIdx].size();
200  data.points = metricsInfos[mi].floatMetrics[metric.storeIdx].data();
201  data.type = MetricType::Float;
202  metricType = MetricType::Float;
203  metricSize = metricsInfos[mi].floatMetrics[metric.storeIdx].size();
204  } break;
205  default:
206  break;
207  }
208  userData.emplace_back(data);
209  }
210  for (size_t ui = 0; ui < userData.size(); ++ui) {
211  metricsToDisplay.push_back(&(userData[ui]));
212  }
213  auto getter = [](const void* hData, int idx) -> float {
214  auto histoData = reinterpret_cast<const MultiplotData*>(hData);
215  size_t pos = (histoData->first + static_cast<size_t>(idx)) % histoData->mod;
216  assert(pos >= 0 && pos < 1024);
217  if (histoData->type == MetricType::Int) {
218  return static_cast<const int*>(histoData->points)[pos];
219  }
220  return static_cast<const float*>(histoData->points)[pos];
221  };
222  if (useHistos) {
223  ImGui::PlotMultiHistograms(
224  label,
225  userData.size(),
226  deviceNames.data(),
227  colors.data(),
228  getter,
229  metricsToDisplay.data(),
230  metricSize,
231  minValue,
232  maxValue * 1.2f,
233  canvasSize);
234  } else {
235  ImGui::PlotMultiLines(
236  label,
237  userData.size(),
238  deviceNames.data(),
239  colors.data(),
240  getter,
241  metricsToDisplay.data(),
242  metricSize,
243  minValue,
244  maxValue * 1.2f,
245  canvasSize);
246  }
247 }
