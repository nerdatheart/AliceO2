36 DataProcessorSpec getSimReaderSpec(int fanoutsize, std::shared_ptr<std::vector<int>> tpcsectors,
37  std::shared_ptr<std::vector<int>> tpcsubchannels)
38 {
39  // this container will contain the TPC sector assignment per subchannel per invocation
40  // it will allow that we snapshot/send exactly one sector assignment per algorithm invocation
41  // to ensure that they all have different timeslice ids
42  auto tpcsectormessages = std::make_shared<std::vector<std::vector<int>>>();
43  tpcsectormessages->resize(tpcsubchannels->size());
44  int tpcchannelcounter = 0;
45  uint64_t activeSectors = 0;
46  for (const auto& tpcsector : *tpcsectors.get()) {
47  activeSectors |= (uint64_t)0x1 << tpcsector;
48  auto actualchannel = (*tpcsubchannels.get())[tpcchannelcounter % tpcsubchannels->size()];
49  LOG(DEBUG) << " WILL ASSIGN SECTOR " << tpcsector << " to subchannel " << actualchannel;
50  tpcsectormessages->operator[](actualchannel).emplace_back(tpcsector);
51  tpcchannelcounter++;
52  }
53 
54  // this is the number of invocations of the algorithm needed for the TPC
55  size_t tpcinvocations = 0;
56  for (int i = 0; i < tpcsubchannels->size(); ++i) {
57  tpcinvocations = std::max(tpcinvocations, tpcsectormessages->operator[](i).size());
58  }
59  // in principle each channel needs to be invoked exactly the same number of times (sigh)
60  // so I am adding some kind of "NOP" sectors in case needed
61  for (int i = 0; i < tpcsubchannels->size(); ++i) {
62  auto size = tpcsectormessages->operator[](i).size();
63  if (size < tpcinvocations) {
64  for (int k = 0; k < tpcinvocations - size; ++k) {
65  tpcsectormessages->operator[](i).emplace_back(-2); // -2 is NOP
66  LOG(INFO) << "ADDING NOP TO CHANNEL " << i << "\n";
67  }
68  assert(tpcsectormessages->operator[](i).size() == tpcinvocations);
69  }
70  }
71  // at this moment all tpc digitizers should receive the exact same number of messages/invocations
72 
73  auto doit = [fanoutsize, tpcsectormessages, tpcinvocations, tpcsubchannels, activeSectors](ProcessingContext& pc) {
74  auto& mgr = steer::HitProcessingManager::instance();
75  auto eventrecords = mgr.getRunContext().getEventRecords();
76  const auto& context = mgr.getRunContext();
77 
78  // counter to make sure we are sending the data only once
79  static int counter = 0;
80 
81  static bool finished = false;
82  if (finished) {
83  // we need to send this in a different time slice
84  // send message telling tpc workers that they can terminate
85  for (const auto& channel : *tpcsubchannels.get()) {
86  // -1 is marker for end of work
87  o2::TPC::TPCSectorHeader header{ -1 };
88  header.activeSectors = activeSectors;
89  pc.outputs().snapshot(
90  OutputRef{ "collisioncontext", static_cast<SubSpecificationType>(channel), { header } },
91  context);
92  }
93  // do this only one
94  pc.services().get<ControlService>().readyToQuit(false);
95  return;
96  }
97 
98  for (int tpcchannel = 0; tpcchannel < tpcsubchannels->size(); ++tpcchannel) {
99  auto& sectors = tpcsectormessages->operator[](tpcchannel);
100  if (counter < sectors.size()) {
101  auto sector = sectors[counter];
102  // send the sectorassign as header with the collision context data
103  o2::TPC::TPCSectorHeader header{ sector };
104  header.activeSectors = activeSectors;
105  pc.outputs().snapshot(
106  OutputRef{ "collisioncontext", static_cast<SubSpecificationType>(tpcchannel), { header } },
107  context);
108  }
109  }
110 
111  // everything not done previously treat here (this is to be seen how since other things than TPC will have
112  // a different number of invocations)
113  for (int subchannel = 0; subchannel < fanoutsize; ++subchannel) {
114  // TODO: this is temporary ... we should find a more clever+ faster + scalable mechanism
115  if (std::find(tpcsubchannels->begin(), tpcsubchannels->end(), subchannel) != tpcsubchannels->end()) {
116  continue;
117  }
118  LOG(INFO) << "SENDING SOMETHING TO OTHERS";
119  pc.outputs().snapshot(
120  OutputRef{ "collisioncontext", static_cast<SubSpecificationType>(subchannel) },
121  context);
122  }
123  counter++;
124  if (tpcinvocations == 0 || counter == tpcinvocations) {
125  finished = true;
126  }
127  };
128 
129  // init function return a lambda taking a ProcessingContext
130  auto initIt = [doit](InitContext& ctx) {
131  // initialize fundamental objects
132  auto& mgr = steer::HitProcessingManager::instance();
133  mgr.addInputFile(ctx.options().get<std::string>("simFile").c_str());
134  if (ctx.options().get<std::string>("simFileS").size() > 0) {
135  mgr.addInputSignalFile(ctx.options().get<std::string>("simFileS").c_str());
136  }
137 
138  // do we start from an existing context
139  auto incontextstring = ctx.options().get<std::string>("incontext");
140  LOG(INFO) << "INCONTEXTSTRING " << incontextstring;
141  if (incontextstring.size() > 0) {
142  auto success = mgr.setupRunFromExistingContext(incontextstring.c_str());
143  if (!success) {
144  LOG(FATAL) << "Could not read collision context from " << incontextstring;
145  }
146  } else {
147  // number of collisions asked?
148  auto col = ctx.options().get<int>("ncollisions");
149  if (col != 0) {
150  mgr.setupRun(col);
151  } else {
152  mgr.setupRun();
153  }
154  LOG(INFO) << "Initializing Spec ... have " << mgr.getRunContext().getEventRecords().size() << " times ";
155  LOG(INFO) << "Serializing Context for later reuse";
156  mgr.writeRunContext(ctx.options().get<std::string>("outcontext").c_str());
157  }
158 
159  return doit;
160  };

163  for (int subchannel = 0; subchannel < fanoutsize; ++subchannel) {
164  outputs.emplace_back(
