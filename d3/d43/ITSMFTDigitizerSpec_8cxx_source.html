44  void init(framework::InitContext& ic)
45  {
46  // setup the input chain for the hits
47  mSimChains.emplace_back(new TChain("o2sim"));
48  // add the main (background) file
49  mSimChains.back()->AddFile(ic.options().get<std::string>("simFile").c_str());
50  // maybe add a particular signal file
51  auto signalfilename = ic.options().get<std::string>("simFileS");
52  if (signalfilename.size() > 0) {
53  mSimChains.emplace_back(new TChain("o2sim"));
54  mSimChains.back()->AddFile(signalfilename.c_str());
55  }
56  // init optional QED chain
57  auto qedfilename = ic.options().get<std::string>("simFileQED");
58  if (qedfilename.size() > 0) {
59  mQEDChain.AddFile(qedfilename.c_str());
60  LOG(INFO) << "Attach QED Tree: " << mQEDChain.GetEntries() << FairLogger::endl;
61  }
62  std::string roString = mID.getName();
63  auto triggeredMode = (ic.options().get<bool>((roString + "triggered").c_str()));
64  if (!triggeredMode) {
65  mROMode = o2::parameters::GRPObject::CONTINUOUS;
66  }
67  LOG(INFO) << mID.getName() << " simulated in "
68  << ((mROMode == o2::parameters::GRPObject::CONTINUOUS) ? "CONTINUOUS" : "TRIGGERED")
69  << " RO mode";
70  // make sure that the geometry is loaded (TODO will this be done centrally?)
71  if (!gGeoManager) {
72  o2::Base::GeometryManager::loadGeometry();
73  }
74 
75  // configure digitizer
76  o2::ITSMFT::GeometryTGeo* geom = nullptr;
77  if (mID == o2::detectors::DetID::ITS) {
78  geom = o2::ITS::GeometryTGeo::Instance();
79  } else {
80  geom = o2::MFT::GeometryTGeo::Instance();
81  }

83  mDigitizer.setGeometry(geom);
84 
85  // defaults (TODO we need a way to pass particular configuration parameters)
86  mDigitizer.getParams().setContinuous(!triggeredMode);
87  mDigitizer.getParams().setROFrameLength(6000); // RO frame in ns
88  mDigitizer.getParams().setStrobeDelay(6000); // Strobe delay wrt beginning of the RO frame, in ns

98  void run(framework::ProcessingContext& pc)
99  {
100  if (mFinished) {
101  return;
102  }
103  std::string detStr = mID.getName();
104  // read collision context from input
105  auto context = pc.inputs().get<o2::steer::RunContext*>("collisioncontext");
106  auto& timesview = context->getEventRecords();
107  LOG(DEBUG) << "GOT " << timesview.size() << " COLLISSION TIMES" << FairLogger::endl;
108  // if there is nothing to do ... return
109  if (timesview.size() == 0) {
110  return;
111  }
112  TStopwatch timer;
113  timer.Start();
114  LOG(INFO) << " CALLING ITS DIGITIZATION " << FairLogger::endl;
115 
116  mDigitizer.setDigits(&mDigits);
117  mDigitizer.setROFRecords(&mROFRecords);
118  mDigitizer.setMCLabels(&mLabels);
119 
120  // attach optional QED digits branch
121  setupQEDChain();
122 
123  auto& eventParts = context->getEventParts();
124  // loop over all composite collisions given from context (aka loop over all the interaction records)
125  for (int collID = 0; collID < timesview.size(); ++collID) {
126  auto eventTime = timesview[collID].timeNS;
127 
128  if (mQEDChain.GetEntries()) { // QED must be processed before other inputs since done in small time steps
129  processQED(eventTime);
130  }
131 
132  mDigitizer.setEventTime(eventTime);
133  mDigitizer.resetEventROFrames(); // to estimate min/max ROF for this collID
134  // for each collision, loop over the constituents event and source IDs
135  // (background signal merging is basically taking place here)
136  for (auto& part : eventParts[collID]) {
137 
138  // get the hits for this event and this source
139  mHits.clear();
140  retrieveHits(part.sourceID, part.entryID);
141 
142  LOG(INFO) << "For collision " << collID << " eventID " << part.entryID
143  << " found " << mHits.size() << " hits " << FairLogger::endl;
144 
145  mDigitizer.process(&mHits, part.entryID, part.sourceID); // call actual digitization procedure
146  }
147  mMC2ROFRecordsAccum.emplace_back(collID, -1, mDigitizer.getEventROFrameMin(), mDigitizer.getEventROFrameMax());
148  accumulate();
149  }
150  // finish digitization ... stream any remaining digits/labels
151  if (mQEDChain.GetEntries()) { // fill last slots from QED input
152  processQED(mDigitizer.getEndTimeOfROFMax());
153  }
154 
155  mDigitizer.fillOutputContainer();
156  accumulate();
157 
158  // here we have all digits and labels and we can send them to consumer (aka snapshot it onto output)
159  pc.outputs().snapshot(Output{ mOrigin, "DIGITS", 0, Lifetime::Timeframe }, mDigitsAccum);
160  pc.outputs().snapshot(Output{ mOrigin, "DIGITSROF", 0, Lifetime::Timeframe }, mROFRecordsAccum);
161  pc.outputs().snapshot(Output{ mOrigin, "DIGITSMC2ROF", 0, Lifetime::Timeframe }, mMC2ROFRecordsAccum);
162  pc.outputs().snapshot(Output{ mOrigin, "DIGITSMCTR", 0, Lifetime::Timeframe }, mLabelsAccum);
163  LOG(INFO) << mID.getName() << ": Sending ROMode= " << mROMode << " to GRPUpdater";
164  pc.outputs().snapshot(Output{ mOrigin, "ROMode", 0, Lifetime::Timeframe }, mROMode);
165 
166  timer.Stop();
167  LOG(INFO) << "Digitization took " << timer.CpuTime() << "s" << FairLogger::endl;
168 
169  // we should be only called once; tell DPL that this process is ready to exit
170  pc.services().get<ControlService>().readyToQuit(false);
171 
172  mFinished = true;
173  }

178  void processQED(double tMax)
179  {
180  auto tQEDNext = mLastQEDTimeNS + mQEDEntryTimeBinNS; // timeslice to retrieve
181  std::string detStr = mID.getName();
182  auto br = mQEDChain.GetBranch((detStr + "Hit").c_str());
183  while (tQEDNext < tMax) {
184  mLastQEDTimeNS = tQEDNext; // time used for current QED slot
185  tQEDNext += mQEDEntryTimeBinNS; // prepare time for next QED slot
186  if (++mLastQEDEntry >= mQEDChain.GetEntries()) {
187  mLastQEDEntry = 0; // wrapp if needed
188  }
189  br->GetEntry(mLastQEDEntry);
190  mDigitizer.setEventTime(mLastQEDTimeNS);
191  mDigitizer.process(&mHits, mLastQEDEntry, QEDSourceID);
192  //
193  }
194  }

196  void setupQEDChain()
197  {
198  if (!mQEDChain.GetEntries()) {
199  return;
200  }
201  std::string detStr = mID.getName();
202  auto qedBranch = mQEDChain.GetBranch((detStr + "Hit").c_str());
203  assert(qedBranch != nullptr);
204  assert(mQEDEntryTimeBinNS >= 1.0);
205  assert(QEDSourceID < o2::MCCompLabel::maxSourceID());
206  mLastQEDTimeNS = -mQEDEntryTimeBinNS / 2; // time will be assigned to the middle of the bin
207  qedBranch->SetAddress(&mHitsP);
208  LOG(INFO) << "Attaching QED ITS hits as sourceID=" << int(QEDSourceID) << ", entry integrates "
209  << mQEDEntryTimeBinNS << " ns" << FairLogger::endl;
210  }

214  {
215  std::string detStr = mID.getName();
216  auto br = mSimChains[sourceID]->GetBranch((detStr + "Hit").c_str());

225  void accumulate()
226  {
227  // accumulate result of single event processing, called after processing every event supplied
228  // AND after the final flushing via digitizer::fillOutputContainer
229  if (!mDigits.size()) {
230  return; // no digits were flushed, nothing to accumulate
231  }
232  static int fixMC2ROF = 0; // 1st entry in mc2rofRecordsAccum to be fixed for ROFRecordID
233  auto ndigAcc = mDigitsAccum.size();
234  std::copy(mDigits.begin(), mDigits.end(), std::back_inserter(mDigitsAccum));
235 
236  // fix ROFrecords references on ROF entries
237  auto nROFRecsOld = mROFRecordsAccum.size();
238 
239  for (int i = 0; i < mROFRecords.size(); i++) {
240  auto& rof = mROFRecords[i];
241  rof.getROFEntry().shiftIndex(ndigAcc);
242  rof.print();
243 
244  if (mFixMC2ROF < mMC2ROFRecordsAccum.size()) { // fix ROFRecord entry in MC2ROF records
245  for (int m2rid = mFixMC2ROF; m2rid < mMC2ROFRecordsAccum.size(); m2rid++) {
246  // need to register the ROFRecors entry for MC event starting from this entry
247  auto& mc2rof = mMC2ROFRecordsAccum[m2rid];
248  if (rof.getROFrame() == mc2rof.minROF) {
249  mFixMC2ROF++;
250  mc2rof.rofRecordID = nROFRecsOld + i;
251  mc2rof.print();
252  }
253  }
254  }
255  }
256  std::copy(mROFRecords.begin(), mROFRecords.end(), std::back_inserter(mROFRecordsAccum));
257  mLabelsAccum.mergeAtBack(mLabels);
258  LOG(INFO) << "Added " << mDigits.size() << " digits " << FairLogger::endl;
259  // clean containers from already accumulated stuff
260  mLabels.clear();
261  mDigits.clear();
262  mROFRecords.clear();
263  }
