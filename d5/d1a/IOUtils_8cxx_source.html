57 std::vector<Event> IOUtils::loadEventData(const std::string& fileName)
58 {
59  std::vector<Event> events{};
60  std::ifstream inputStream{};
61  std::string line{}, unusedVariable{};
62  int layerId{}, monteCarlo{};
63  int clusterId{ EventLabelsSeparator };
64  float xCoordinate{}, yCoordinate{}, zCoordinate{}, alphaAngle{};
65  float varZ{ -1.f }, varY{ -1.f };
66 
67  inputStream.open(fileName);
68 
70  dataformats::MCTruthContainer<MCCompLabel>* mcLabels = nullptr;
71  while (std::getline(inputStream, line)) {
72 
73  std::istringstream inputStringStream(line);
74  if (inputStringStream >> layerId >> xCoordinate >> yCoordinate >> zCoordinate) {
75 
76  if (layerId == PrimaryVertexLayerId) {
77 
78  if (clusterId != 0) {
79  events.emplace_back(events.size());
80  }
81 
82  events.back().addPrimaryVertex(xCoordinate, yCoordinate, zCoordinate);
83  clusterId = 0;
84 
85  } else {
86 
87  if (inputStringStream >> varY >> varZ >> unusedVariable >> alphaAngle >> monteCarlo) {
88  events.back().addClusterToLayer(layerId, xCoordinate, yCoordinate, zCoordinate,
89  events.back().getLayer(layerId).getClustersSize());
90  const float sinAlpha = std::sin(alphaAngle);
91  const float cosAlpha = std::cos(alphaAngle);
92  const float xTF = xCoordinate * cosAlpha - yCoordinate * sinAlpha;
93  const float yTF = xCoordinate * sinAlpha + yCoordinate * cosAlpha;
94  events.back().addTrackingFrameInfoToLayer(layerId, xTF, alphaAngle, std::array<float, 2>{ yTF, zCoordinate },
95  std::array<float, 3>{ varY, 0.f, varZ });
96  events.back().addClusterLabelToLayer(layerId, MCCompLabel(monteCarlo));
97 
98  ++clusterId;
99  }
100  }
101  }
102  }
103 
104  return events;
105 }

107 void IOUtils::loadEventData(Event& event, const std::vector<ITSMFT::Cluster>* clusters,
108  const dataformats::MCTruthContainer<MCCompLabel>* mcLabels)
109 {
110  if (!clusters) {
111  std::cerr << "Missing clusters." << std::endl;
112  return;
113  }
114  event.clear();
115  GeometryTGeo* geom = GeometryTGeo::Instance();
116  geom->fillMatrixCache(utils::bit2Mask(TransformType::T2GRot));
117  int clusterId{ 0 };
118 
119  for (auto& c : *clusters) {
120  int layer = geom->getLayer(c.getSensorID());
121 
123  event.addTrackingFrameInfoToLayer(layer, c.getX(), geom->getSensorRefAlpha(c.getSensorID()),
124  std::array<float, 2>{ c.getY(), c.getZ() },
125  std::array<float, 3>{ c.getSigmaY2(), c.getSigmaYZ(), c.getSigmaZ2() });
126 
128  auto xyz = c.getXYZGloRot(*geom);
129  event.addClusterToLayer(layer, xyz.x(), xyz.y(), xyz.z(), event.getLayer(layer).getClustersSize());
130  event.addClusterLabelToLayer(layer, *(mcLabels->getLabels(clusterId).begin()));
131  clusterId++;
132  }
133 }

167 std::vector<std::unordered_map<int, Label>> IOUtils::loadLabels(const int eventsNum, const std::string& fileName)
168 {
169  std::vector<std::unordered_map<int, Label>> labelsMap{};
170  std::unordered_map<int, Label> currentEventLabelsMap{};
171  std::ifstream inputStream{};
172  std::string line{};
173  int monteCarloId{}, pdgCode{}, numberOfClusters{};
174  float transverseMomentum{}, phiCoordinate{}, pseudorapidity{};
175 
176  labelsMap.reserve(eventsNum);
177 
178  inputStream.open(fileName);
179  std::getline(inputStream, line);
180 
181  while (std::getline(inputStream, line)) {
182 
183  std::istringstream inputStringStream(line);
184 
185  if (inputStringStream >> monteCarloId) {
186 
187  if (monteCarloId == EventLabelsSeparator) {
188 
189  labelsMap.emplace_back(currentEventLabelsMap);
190  currentEventLabelsMap.clear();
191 
192  } else {
193 
194  if (inputStringStream >> transverseMomentum >> phiCoordinate >> pseudorapidity >> pdgCode >> numberOfClusters) {
195 
196  if (std::abs(pdgCode) == Constants::PDGCodes::PionCode && numberOfClusters == 7) {
197 
198  currentEventLabelsMap.emplace(std::piecewise_construct, std::forward_as_tuple(monteCarloId),
199  std::forward_as_tuple(monteCarloId, transverseMomentum, phiCoordinate,
200  pseudorapidity, pdgCode, numberOfClusters));
201  }
202  }
203  }
204  }
205  }
206 
207  labelsMap.emplace_back(currentEventLabelsMap);
208 
209  return labelsMap;
210 }

212 void IOUtils::writeRoadsReport(std::ofstream& correctRoadsOutputStream, std::ofstream& duplicateRoadsOutputStream,
213  std::ofstream& fakeRoadsOutputStream, const std::vector<std::vector<Road>>& roads,
214  const std::unordered_map<int, Label>& labelsMap)
215 {
216  const int numVertices{ static_cast<int>(roads.size()) };
217  std::unordered_set<int> foundMonteCarloIds{};
218 
219  correctRoadsOutputStream << EventLabelsSeparator << std::endl;
220  fakeRoadsOutputStream << EventLabelsSeparator << std::endl;
221 
222  for (int iVertex{ 0 }; iVertex < numVertices; ++iVertex) {
223 
224  const std::vector<Road>& currentVertexRoads{ roads[iVertex] };
225  const int numRoads{ static_cast<int>(currentVertexRoads.size()) };
226 
227  for (int iRoad{ 0 }; iRoad < numRoads; ++iRoad) {
228 
229  const Road& currentRoad{ currentVertexRoads[iRoad] };
230  const int currentRoadLabel{ currentRoad.getLabel() };
231 
232  if (!labelsMap.count(currentRoadLabel)) {
233 
234  continue;
235  }
236 
237  const Label& currentLabel{ labelsMap.at(currentRoadLabel) };
238 
239  if (currentRoad.isFakeRoad()) {
240 
241  fakeRoadsOutputStream << currentLabel << std::endl;
242 
243  } else {
244 
245  if (foundMonteCarloIds.count(currentLabel.monteCarloId)) {
246 
247  duplicateRoadsOutputStream << currentLabel << std::endl;
248 
249  } else {
250 
251  correctRoadsOutputStream << currentLabel << std::endl;
252  foundMonteCarloIds.emplace(currentLabel.monteCarloId);
253  }
254  }
255  }
256  }
