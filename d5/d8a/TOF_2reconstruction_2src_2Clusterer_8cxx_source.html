46 void Clusterer::processStrip(std::vector<Cluster>& clusters, MCLabelContainer const* digitMCTruth)
47 {
48  // method to clusterize the current strip
49 
50  Int_t detId[5];
51  Int_t chan, chan2, chan3;
52  Int_t strip1, strip2;
53  Int_t iphi, iphi2, iphi3;
54  Int_t ieta, ieta2, ieta3; // it is the number of padz-row increasing along the various strips
55 
56  for (int idig = 0; idig < mStripData.digits.size(); idig++) {
57  // LOG(DEBUG) << "Checking digit " << idig;
58  Digit* dig = &mStripData.digits[idig];
59  if (dig->isUsedInCluster())
60  continue; // the digit was already used to build a cluster
61 
62  mNumberOfContributingDigits = 0;
63  dig->getPhiAndEtaIndex(iphi, ieta);
64  if (mStripData.digits.size() > 1)
65  LOG(DEBUG) << "idig = " << idig;
66 
67  // first we make a cluster out of the digit
68  int noc = clusters.size();
69  // LOG(DEBUG) << "noc = " << noc << "\n";
70  clusters.emplace_back();
71  Cluster& c = clusters[noc];
72  addContributingDigit(dig);
73  float timeDig = dig->getTDC() * Geo::TDCBIN;
74 
75  for (int idigNext = idig + 1; idigNext < mStripData.digits.size(); idigNext++) {
76  Digit* digNext = &mStripData.digits[idigNext];
77  if (digNext->isUsedInCluster())
78  continue; // the digit was already used to build a cluster
79  // check if the TOF time are close enough to be merged; if not, it means that nothing else will contribute to the cluster (since digits are ordered in time)
80  float timeDigNext = digNext->getTDC() * Geo::TDCBIN; // we assume it calibrated (for now); in ps
81  LOG(DEBUG) << "Time difference = " << timeDigNext - timeDig;
82  if (timeDigNext - timeDig > 500 /*in ps*/)
83  break;
84  digNext->getPhiAndEtaIndex(iphi2, ieta2);
85 
86  // check if the fired pad are close in space
87  LOG(DEBUG) << "phi difference = " << iphi - iphi2;
88  LOG(DEBUG) << "eta difference = " << ieta - ieta2;
89  if ((TMath::Abs(iphi - iphi2) > 1) || (TMath::Abs(ieta - ieta2) > 1))
90  continue;
91 
92  // if we are here, the digit contributes to the cluster
93  addContributingDigit(digNext);
94 
95  } // loop on the second digit
96 
97  buildCluster(c, digitMCTruth);
98 
99  } // loop on the first digit
100 }

118 void Clusterer::buildCluster(Cluster& c, MCLabelContainer const* digitMCTruth)
119 {
120 
121  // here we finally build the cluster from all the digits contributing to it
122 
123  Digit* temp;
124  for (int idig = 1; idig < mNumberOfContributingDigits; idig++) {
125  // the digit[0] will be the main one
126  if (mContributingDigit[idig]->getTOT() > mContributingDigit[0]->getTOT()) {
127  temp = mContributingDigit[0];
128  mContributingDigit[0] = mContributingDigit[idig];
129  mContributingDigit[idig] = temp;
130  }
131  }
132 
133  c.setMainContributingChannel(mContributingDigit[0]->getChannel());
134  c.setTime(mContributingDigit[0]->getTDC() * Geo::TDCBIN); // time in ps (for now we assume it calibrated)
135  c.setTot(mContributingDigit[0]->getTOT() * Geo::TOTBIN * 1E-3); // TOT in ns (for now we assume it calibrated)
136  //setL0L1Latency(); // to be filled (maybe)
137  //setDeltaBC(); // to be filled (maybe)
138 
139  int chan1, chan2;
140  int phi1, phi2;
141  int eta1, eta2;
142  int deltaPhi, deltaEta;
143  int mask;
144 
145  mContributingDigit[0]->getPhiAndEtaIndex(phi1, eta1);
146  // now set the mask with the secondary digits
147  for (int idig = 1; idig < mNumberOfContributingDigits; idig++) {
148  mContributingDigit[idig]->getPhiAndEtaIndex(phi2, eta2);
149  deltaPhi = phi1 - phi2;
150  deltaEta = eta1 - eta2;
151  if (deltaPhi == 1) { // the digit is to the LEFT of the cluster; let's check about UP/DOWN/Same Line
152  if (deltaEta == 1) { // the digit is DOWN LEFT wrt the cluster
153  mask = Cluster::kDownLeft;
154  } else if (deltaEta == -1) { // the digit is UP LEFT wrt the cluster
155  mask = Cluster::kUpLeft;
156  } else { // the digit is LEFT wrt the cluster
157  mask = Cluster::kLeft;
158  }
159  } else if (deltaPhi == -1) { // the digit is to the RIGHT of the cluster; let's check about UP/DOWN/Same Line
160  if (deltaEta == 1) { // the digit is DOWN RIGHT wrt the cluster
161  mask = Cluster::kDownRight;
162  } else if (deltaEta == -1) { // the digit is UP RIGHT wrt the cluster
163  mask = Cluster::kUpRight;
164  } else { // the digit is RIGHT wrt the cluster
165  mask = Cluster::kRight;
166  }
167  } else if (deltaPhi == 0) { // the digit is on the same column as the cluster; is it UP or Down?
168  if (deltaEta == 1) { // the digit is DOWN wrt the cluster
169  mask = Cluster::kDown;
170  } else if (deltaEta == -1) { // the digit is UP wrt the cluster
171  mask = Cluster::kUp;
172  } else { // impossible!!
173  LOG(DEBUG) << " Check what is going on, the digit you are trying to merge to the cluster must be in a different channels... ";
174  }
175  } else { // impossible!!! We checked above...
176  LOG(DEBUG) << " Check what is going on, the digit you are trying to merge to the cluster is too far from the cluster, you should have not got here... ";
177  }
178  c.addBitInContributingChannels(mask);
179  }
180 
181  // filling the MC labels of this cluster; the first will be those of the main digit; then the others
182  if (digitMCTruth != nullptr) {
183  int lbl = mClsLabels->getIndexedSize(); // this should correspond to the number of digits also;
184  for (int i = 0; i < mNumberOfContributingDigits; i++) {
185  int digitLabel = mContributingDigit[i]->getLabel();
186  gsl::span<const o2::MCCompLabel> mcArray = digitMCTruth->getLabels(digitLabel);
187  for (int j = 0; j < static_cast<int>(mcArray.size()); ++j) {
188  auto label = digitMCTruth->getElement(digitMCTruth->getMCTruthHeader(digitLabel).index + j);
189  mClsLabels->addElement(lbl, label);
190  }
191  }
192  }
193 
194  return;
195 }
