147  auto persist(std::vector<std::string> const& columnNames)
148  {
149  constexpr int nColumns = sizeof...(ARGS);
150  if (nColumns != columnNames.size()) {
151  throw std::runtime_error("Mismatching number of column types and names");
152  }
153  if (mBuilders != nullptr) {
154  throw std::runtime_error("TableBuilder::persist can only be invoked once per instance");
155  }
156  mArrays.resize(nColumns);
157 
158  using BuildersTuple = typename std::tuple<std::unique_ptr<typename BuilderTraits<ARGS>::BuilderType>...>;
159  mSchema = std::make_shared<arrow::Schema>(TableBuilderHelper::makeFields<ARGS...>(columnNames));
160 
161  BuildersTuple* builders = new BuildersTuple(std::make_unique<typename BuilderTraits<ARGS>::BuilderType>(mMemoryPool)...);
162  mBuilders = builders; // We store the builders
164  mFinalizer = [schema = mSchema, &arrays = mArrays, builders = builders]() -> std::shared_ptr<arrow::Table> {
165  Finalizer<nColumns, nColumns>::finalize(arrays, *builders);
166  return arrow::Table::Make(schema, arrays);
167  };
168  // Callback used to fill the builders
169  return [builders = builders](unsigned int slot, ARGS... args) -> void {
170  Appender<nColumns, nColumns>::append(*builders, args...);
171  };
172  }
