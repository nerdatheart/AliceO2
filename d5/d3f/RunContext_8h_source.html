24 // a structure describing EventPart
25 // (an elementary constituent of a collision)
26 struct EventPart {
27  EventPart() = default;
28  EventPart(int s, int e) : sourceID(s), entryID(e) {}
29  int sourceID = 0; // the ID of the source (0->backGround; > 1 signal source)
30  // the sourceID should correspond to the chain ID
31  int entryID = 0; // the event/entry ID inside the chain corresponding to sourceID
32 
33  static bool isSignal(EventPart e) { return e.sourceID > 1; }
34  static bool isBackGround(EventPart e) { return !isSignal(e); }
