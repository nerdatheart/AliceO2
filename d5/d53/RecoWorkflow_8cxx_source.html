61 framework::WorkflowSpec getWorkflow(bool propagateMC, int nLanes, std::string cfgInput, std::string cfgOutput)
62 {
63  InputType inputType;
64  OutputType outputType;
65  try {
66  inputType = InputMap.at(cfgInput);
67  } catch (std::out_of_range&) {
68  throw std::runtime_error(std::string("invalid input type: ") + cfgInput);
69  }
70  try {
71  outputType = OutputMap.at(cfgOutput);
72  } catch (std::out_of_range&) {
73  throw std::runtime_error(std::string("invalid output type: ") + cfgOutput);
74  }
75 
76  WorkflowSpec specs;
77 
78  // there is no MC info when starting from raw data
79  // but maybe the warning can be dropped
80  if (propagateMC && outputType == OutputType::Raw) {
81  LOG(WARNING) << "input type 'raw' selected, switch off MC propagation";
82  propagateMC = false;
83  }
84 
85  // note: converter does not touch MC, this is routed directly to downstream consumer
86  if (inputType == InputType::Digits) {
87  specs.emplace_back(o2::TPC::getDigitReaderSpec(nLanes));
88  for (int n = 0; n < nLanes; ++n) {
89  specs.emplace_back(o2::TPC::getClustererSpec(propagateMC, (nLanes > 1 ? n : -1)));
90  specs.emplace_back(o2::TPC::getClusterConverterSpec(propagateMC, (nLanes > 1 ? n : -1)));
91  }
92  } else if (inputType == InputType::Digitizer) {
93  for (int n = 0; n < nLanes; ++n) {
94  specs.emplace_back(o2::TPC::getClustererSpec(propagateMC, (nLanes > 1 ? n : -1)));
95  specs.emplace_back(o2::TPC::getClusterConverterSpec(propagateMC, (nLanes > 1 ? n : -1)));
96  }
97  } else if (inputType == InputType::Clusters) {
98  specs.emplace_back(o2::TPC::getClusterReaderSpec(/*propagateMC*/));
99  for (int n = 0; n < nLanes; ++n) {
100  specs.emplace_back(o2::TPC::getClusterConverterSpec(propagateMC, (nLanes > 1 ? n : -1)));
101  }
102  } else if (inputType == InputType::Raw) {
103  throw std::runtime_error(std::string("input type 'raw' not yet implemented"));
104  }
105 
106  if (outputType == OutputType::Clusters || outputType == OutputType::Raw) {
107  throw std::runtime_error(std::string("output types 'clusters' and 'raw' not yet implemented"));
108  }
109  // also add a binary writer
110  auto writerFunction = [](InitContext& ic) {
111  auto filename = ic.options().get<std::string>("outfile");
112  if (filename.empty()) {
113  throw std::runtime_error("output file missing");
114  }
115  auto output = std::make_shared<std::ofstream>(filename.c_str(), std::ios_base::binary);
116  return [output](ProcessingContext& pc) {
117  LOG(INFO) << "processing data set with " << pc.inputs().size() << " entries";
118  for (const auto& entry : pc.inputs()) {
119  LOG(INFO) << " " << *(entry.spec);
120  output->write(entry.payload, o2::framework::DataRefUtils::getPayloadSize(entry));
121  LOG(INFO) << "wrote data, size " << o2::framework::DataRefUtils::getPayloadSize(entry);
122  }
123  };
124  };
125 
126  for (int n = 0; n < nLanes; ++n) {
127  specs.emplace_back(o2::TPC::getClusterDecoderRawSpec(propagateMC, (nLanes > 1 ? n : -1)));
128  }
129 
130  auto createInputSpec = []() {
131  o2::framework::Inputs inputs;
140  inputs.emplace_back(InputSpec{ "input", "TPC", "CLUSTERNATIVE", 0, framework::Lifetime::Timeframe });
141 
142  return std::move(inputs);
143  };
144 
145  if (outputType == OutputType::Tracks) {
146  specs.emplace_back(o2::TPC::getCATrackerSpec(propagateMC, nLanes));
147  specs.emplace_back(o2::TPC::getRootFileWriterSpec(propagateMC));
148  } else if (outputType == OutputType::DecodedClusters) {
149  specs.emplace_back(DataProcessorSpec{ "writer",
150  { createInputSpec() },
151  {},
152  AlgorithmSpec(writerFunction),
153  Options{
154  { "outfile", VariantType::String, { "Name of the output file" } },
155  } });
156  }
157  return std::move(specs);
158 }
