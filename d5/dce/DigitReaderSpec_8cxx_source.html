47  auto initFunction = [fanOut](InitContext& ic) {
48  // get the option from the init context
49  auto filename = ic.options().get<std::string>("infile");
50  auto treename = ic.options().get<std::string>("treename");
51  auto clbrName = ic.options().get<std::string>("digitbranch");
52  auto mcbrName = ic.options().get<std::string>("mcbranch");
53  auto nofEvents = ic.options().get<int>("nevents");
54 
55  auto processAttributes = std::make_shared<ProcessAttributes>();
56  {
57  processAttributes->nParallelReaders = fanOut;
58  processAttributes->terminateOnEod = ic.options().get<bool>("terminate-on-eod");
59  auto& sectors = processAttributes->sectors;
60  auto& activeSectors = processAttributes->activeSectors;
61  auto& readers = processAttributes->readers;
62 
63  sectors = std::move(o2::RangeTokenizer::tokenize<int>(ic.options().get<std::string>("tpc-sectors")));
64  for (auto const& s : sectors) {
65  // set the mask of active sectors
66  if (s >= NSectors) {
67  std::string message = std::string("invalid sector range specified, allowed 0-") + std::to_string(NSectors - 1);
68  // FIXME should probably be FATAL, but this doesn't seem to be handled in the DPL control flow
69  // at least the process is not marked dead in the DebugGUI
70  LOG(ERROR) << message;
71  throw std::invalid_argument(message);
72  }
73  activeSectors |= (uint64_t)0x1 << s;
74  }
75 
76  // set up the tree interface
77  // TODO: parallelism on sectors needs to be implemented as selector in the reader
78  // the data is now in parallel branches, as first attempt use an array of readers
79  constexpr auto persistency = Lifetime::Timeframe;
80  o2::header::DataHeader::SubSpecificationType lane = 0;
81  for (size_t sector = 0; sector < NSectors; ++sector) {
82  if ((activeSectors & ((uint64_t)0x1 << sector)) == 0) {
83  continue;
84  }
85  std::string clusterbranchname = clbrName + "_" + std::to_string(sector);
86  std::string mcbranchname = mcbrName + "_" + std::to_string(sector);
87  readers[sector] = std::make_shared<RootTreeReader>(treename.c_str(), // tree name
88  nofEvents, // number of entries to publish
89  filename.c_str(), // input file name
90  Output{ gDataOriginTPC, "DIGITS", lane, persistency },
91  clusterbranchname.c_str(), // name of digit branch
92  Output{ gDataOriginTPC, "DIGITSMCTR", lane, persistency },
93  mcbranchname.c_str() // name of mc label branch
94  );
95  if (++lane >= processAttributes->nParallelReaders) {
96  lane = 0;
97  }
98  }
99  }
100 
101  // set up the processing function
102  // using by-copy capture of the worker instance shared pointer
103  // the shared pointer makes sure to clean up the instance when the processing
104  // function gets out of scope
105  // FIXME: wanted to use it = sectors.begin() in the variable capture but the iterator
106  // is const and can not be incremented
107  auto processingFct = [ processAttributes, index = std::make_shared<int>(0) ](ProcessingContext & pc)
108  {
109  if (processAttributes->finished) {
110  return;
111  }
112 
113  auto publish = [&processAttributes, &index, &pc]() {
114  auto& sectors = processAttributes->sectors;
115  auto& activeSectors = processAttributes->activeSectors;
116  auto& readers = processAttributes->readers;
117  if (*index >= sectors.size()) {
118  *index = 0;
119  }
120  while (*index < sectors.size() && !readers[sectors[*index]]) {
121  // probably more efficient to use a vector of valid readers instead of the fixed array with
122  // possibly invalid entries
123  ++(*index);
124  }
125  if (*index == sectors.size()) {
126  // there is no valid reader at all
127  return false;
128  }
129  auto sector = sectors[*index];
130  o2::TPC::TPCSectorHeader header{ sector };
131  header.activeSectors = activeSectors;
132  auto& r = *(readers[sector].get());
133 
134  // increment the reader and invoke it for the processing context
135  if (r.next()) {
136  // there is data, run the reader
137  r(pc, header);
138  } else {
139  // no more data, delete the reader
140  readers[sector].reset();
141  return false;
142  }
143  ++(*index);
144  return true;
145  };
146 
147  int operation = -2;
148  for (size_t lane = 0; lane < processAttributes->nParallelReaders; ++lane) {
149  if (!publish()) {
150  // need to publish a dummy packet
151  // FIXME define and use flags in the TPCSectorHeader, for now using the same schema as
152  // in digitizer workflow, -1 -> end of data, -2 noop
153  if (lane == 0) {
154  operation = -1;
155  }
156  o2::TPC::TPCSectorHeader header{ operation };
157  pc.outputs().snapshot(OutputRef{ "output", lane, { header } }, lane);
158  pc.outputs().snapshot(OutputRef{ "outputMC", lane, { header } }, lane);
159  }
160  }
161 
162  if ((processAttributes->finished = (operation == -1)) && processAttributes->terminateOnEod) {
163  pc.services().get<ControlService>().readyToQuit(false);
164  }
165  };
166 
167  // return the actual processing function as a lambda function using variables
168  // of the init function
169  return processingFct;
170  };

178  outputSpecs.emplace_back(OutputSpec{ { "outputMC" }, gDataOriginTPC, datadescMC, n, Lifetime::Timeframe });
179  }
180  return std::move(outputSpecs);
181  };
