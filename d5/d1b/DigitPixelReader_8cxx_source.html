30 ChipPixelData* DigitPixelReader::getNextChipData(std::vector<ChipPixelData>& chipDataVec)

44 bool DigitPixelReader::getNextChipData(ChipPixelData& chipData)
45 {
46  // decode data of single chip to chipData
47  if (!mLastDigit) {
48  if (mIdx >= mDigits->size()) {
49  return false;
50  }
51  mLastDigit = &((*mDigits)[mIdx++]);
52  }
53  chipData.clear();
54  chipData.setStartID(mIdx - 1);
55  chipData.setChipID(mLastDigit->getChipIndex());
56  chipData.setROFrame(mLastDigit->getROFrame());
57  chipData.getData().emplace_back(mLastDigit);
58  mLastDigit = nullptr;
59 
60  while (mIdx < mDigits->size()) {
61  mLastDigit = &((*mDigits)[mIdx++]);
62  if (chipData.getChipID() != mLastDigit->getChipIndex())
63  break;
64  if (chipData.getROFrame() != mLastDigit->getROFrame())
65  break;
66  chipData.getData().emplace_back(mLastDigit);
67  mLastDigit = nullptr;
68  }
69  return true;
70 }

73 void DigitPixelReader::openInput(const std::string inpName, o2::detectors::DetID det)
74 {
75  // open input file, load digits, MC labels
76  assert(det.getID() == o2::detectors::DetID::ITS || det.getID() == o2::detectors::DetID::MFT);
77 
78  clear();
79 
80  if (!(mInputTree = o2::utils::RootChain::load("o2sim", inpName))) {
81  LOG(FATAL) << "Failed to load Digits tree from " << inpName << FairLogger::endl;
82  }
83  std::string brName = det.getName();
84  mInputTree->SetBranchAddress((brName + "Digit").data(), &mDigitsSelf);

86  LOG(FATAL) << "Failed to find ITSDigit branch in the " << mInputTree->GetName()

88  }
89  setDigits(mDigitsSelf);
90
