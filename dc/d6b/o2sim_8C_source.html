53 
54  auto pid = getpid();

64  run->SetName(confref.getMCEngine().c_str()); // Transport engine
65  run->SetIsMT(confref.getIsMT()); // MT mode
66 
67  // construct geometry / including magnetic field

69

84  std::stringstream geomss;

91 
92  std::time_t runStart = std::time(nullptr);

96  auto rtdb = run->GetRuntimeDb();

98 
99  std::stringstream s2;
100  s2 << confref.getOutPrefix();
101  if (asservice) {
102  s2 << "_" << pid;
103  }
104  s2 << "_par.root";
105  std::string parfilename = s2.str();
106  parOut->open(parfilename.c_str());
107  rtdb->setOutput(parOut);
108  rtdb->saveOutput();
109  rtdb->print();
110  o2::PDG::addParticlesToPdgDataBase(0);
111 
112  {
113  // store GRPobject
114  o2::parameters::GRPObject grp;
115  grp.setRun(run->GetRunId());
116  grp.setTimeStart(runStart);
117  grp.setTimeEnd(std::time(nullptr));
118  TObjArray* modArr = run->GetListOfModules();
119  TIter next(modArr);
120  FairModule* module = nullptr;

123  if (!det) {

126  if (det->GetDetId() < o2::detectors::DetID::First) {

128  }
129  if (det->GetDetId() > o2::detectors::DetID::Last) {
130  continue; // passive
131  }

133  }
134  grp.print();
135  printf("VMC: %p\n", TVirtualMC::GetMC());
136  auto field = dynamic_cast<o2::field::MagneticField*>(run->GetField());

139  o2::units::Current_t currL3 = field->getCurrentSolenoid();
140  grp.setL3Current(currL3);
141  grp.setDipoleCurrent(currDip);
142  }
143  // save
144  std::string grpfilename = confref.getOutPrefix() + "_grp.root";
145  TFile grpF(grpfilename.c_str(), "recreate");
146  grpF.WriteObjectAny(&grp, grp.Class(), "GRP");
147  }
148  // todo: save beam information in the grp
149 
150  timer.Stop();
151  Double_t rtime = timer.RealTime();
152  Double_t ctime = timer.CpuTime();
153 
154  // extract max memory usage for init
155  FairSystemInfo sysinfo;
156  LOG(INFO) << "Init: Real time " << rtime << " s, CPU time " << ctime << "s";
157  LOG(INFO) << "Init: Memory used " << sysinfo.GetMaxMemory() << " MB";
158 
159  return run;
160 }
