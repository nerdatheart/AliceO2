41 bool MCAnalysisFileWrapper::read(const std::string& filepath)
42 {
43  ROOTIOUtilities rootutil(filepath);
44  // look for and read meta info of step logger and analysis run
45  if (!rootutil.hasObject(defaults::mcAnalysisMetaInfoName)) {
46  rootutil.close();
47  return false;
48  }
49  rootutil.readObject(mAnalysisMetaInfo, defaults::mcAnalysisMetaInfoName);
50 
51  // try to recover histograms
52  TH1* histoRecover = nullptr;
53 
54  if (!rootutil.changeToTDirectory(defaults::mcAnalysisObjectsDirName)) {
55  rootutil.close();
56  return false;
57  }
58  while (true) {
59  rootutil.readObject(histoRecover);
60  // assuming that all objects have been read
61  if (!histoRecover) {
62  break;
63  }
64  TH1* histo = dynamic_cast<TH1*>(histoRecover->Clone());
65  histo->SetDirectory(0);
66  mHistograms.push_back(std::shared_ptr<TH1>(histo));
67  }
68  rootutil.close();
69  isSane();
70  return true;
71 }

73 void MCAnalysisFileWrapper::write(const std::string& filedir) const
74 {
75  if (!isSane()) {
76  LOG(ERROR) << "Analysis file cannot be written, see above.";
77  return;
78  }
79  const std::string outputDir = filedir + "/" + mAnalysisMetaInfo.analysisName;
80  if (!createDirectory(outputDir)) {
81  LOG(ERROR) << "Directory " << outputDir << " could not be created for analysis " << mAnalysisMetaInfo.analysisName << ". Skip...\n";
82  return;
83  }
84  ROOTIOUtilities rootutil(outputDir + "/Analysis.root", ETFileMode::kRECREATE);
85 
86  LOG(INFO) << "Save histograms of analysis " << mAnalysisMetaInfo.analysisName << "\n\tat " << filedir << "\n";
87  rootutil.writeObject(&mAnalysisMetaInfo, defaults::mcAnalysisMetaInfoName);
88  rootutil.changeToTDirectory(defaults::mcAnalysisObjectsDirName);
89  for (const auto& h : mHistograms) {
90  rootutil.writeObject(h.get());
91  }
92  rootutil.close();
93 }
