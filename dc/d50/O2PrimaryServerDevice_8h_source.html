105  bool HandleRequest(FairMQMessagePtr& request, int /*index*/)
106  {
107  LOG(INFO) << "GOT A REQUEST WITH SIZE " << request->GetSize();
108  std::string requeststring(static_cast<char*>(request->GetData()), request->GetSize());
109 
110  if (requeststring.compare("configrequest") == 0) {
111  return HandleConfigRequest(request);
112  }
113 
114  else if (requeststring.compare("primrequest") != 0) {
115  LOG(INFO) << "unknown request\n";
116  return true;
117  }
118 
119  // we only need the initialized generator at this moment
120  if (mGeneratorInitThread.joinable()) {
121  mGeneratorInitThread.join();
122  }
123 
124  static int counter = 0;
125  if (counter >= mMaxEvents && mNeedNewEvent) {
126  return false;
127  }
128 
129  LOG(INFO) << "Received request for work ";
130  if (mNeedNewEvent) {
131  mStack.Reset();
132  mPrimGen.GenerateEvent(&mStack);
133  mNeedNewEvent = false;
134  mPartCounter = 0;
135  counter++;
136  }
137 
138  auto& prims = mStack.getPrimaries();
139  auto numberofparts = (int)std::ceil(prims.size() / (1. * mChunkGranularity));
140 
141  o2::Data::PrimaryChunk m;
142  o2::Data::SubEventInfo i;
143  i.eventID = counter;
144  i.maxEvents = mMaxEvents;
145  i.part = mPartCounter + 1;
146  i.nparts = numberofparts;
147  i.seed = counter + 10;
148  i.index = m.mParticles.size();
149  m.mEventIDs.emplace_back(i);
150 
151  auto startindex = mPartCounter * mChunkGranularity;
152  auto endindex = startindex + mChunkGranularity;
153  auto startiter = prims.begin() + mPartCounter * mChunkGranularity;
154  auto enditer = endindex < prims.size() ? startiter + mChunkGranularity : prims.end();
155  std::copy(startiter, enditer, std::back_inserter(m.mParticles));
156 
157  LOG(WARNING) << "Sending " << m.mParticles.size() << " particles\n";
158  LOG(WARNING) << "treating ev " << counter << " part " << i.part << " out of " << i.nparts << "\n";
159 
160  mPartCounter++;
161  if (mPartCounter == numberofparts) {
162  mNeedNewEvent = true;
163  }
164 
165  TMessage* tmsg = new TMessage(kMESS_OBJECT);
166  tmsg->WriteObjectAny((void*)&m, TClass::GetClass("o2::Data::PrimaryChunk"));
167 
168  auto free_tmessage = [](void* data, void* hint) { delete static_cast<TMessage*>(hint); };
169 
170  std::unique_ptr<FairMQMessage> message(
171  fTransportFactory->CreateMessage(tmsg->Buffer(), tmsg->BufferSize(), free_tmessage, tmsg));
172 
173  // send answer
174  if (Send(message, "primary-get") > 0) {
175  LOG(INFO) << "reply send ";
176  return true;
177  }
178  return true;
179  }
