145 void CDBInterface::loadNoiseAndPedestalFromFile()
146 {
147  auto file = TFile::Open(mPedestalNoiseFileName.data());
148  CalPad* pedestals{ nullptr };
149  CalPad* noise{ nullptr };
150  file->GetObject("Pedestals", pedestals);
151  file->GetObject("Noise", noise);
152  delete file;
153 
154  if (!pedestals) {
155  LOG(FATAL) << "No valid pedestal object was loaded" << FairLogger::endl;
156  }
157 
158  if (!noise) {
159  LOG(FATAL) << "No valid noise object was loaded" << FairLogger::endl;
160  }
161 
162  mPedestals.reset(pedestals);
163  mNoise.reset(noise);
164 
165  LOG(INFO) << "Loaded Noise and pedestal from file '" << mPedestalNoiseFileName << "'" << FairLogger::endl;
166 }

185 void CDBInterface::createDefaultPedestals()
186 {
187  // ===| create random pedestals |=============================================
188  mPedestals = std::make_unique<CalPad>("Pedestals");
189 
190  // distribution based on test beam data
191  const float meanPedestal = 72.5;
192  const float sigmaPedestal = 9.0;
193 
194  // set a minimum and maximum for the value
195  const float minPedestal = meanPedestal - 4 * sigmaPedestal;
196  const float maxPedestal = meanPedestal + 4 * sigmaPedestal;
197 
198  for (auto& calArray : mPedestals->getData()) {
199  for (auto& val : calArray.getData()) {
200  float random = gRandom->Gaus(meanPedestal, sigmaPedestal);
201  if (random < minPedestal) {
202  random = minPedestal;
203  }
204  if (random > maxPedestal) {
205  random = maxPedestal;
206  }
207  val = random;
208  }
209  }
210 }
