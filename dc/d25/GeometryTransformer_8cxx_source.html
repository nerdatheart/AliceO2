39 void GeometryTransformer::init()
40 {
42  const double degToRad = TMath::DegToRad();
43 
44  ROOT::Math::Rotation3D planeRot(ROOT::Math::RotationX(Constants::sBeamAngle * degToRad));
45 
46  for (int iside = 0; iside < 2; ++iside) {
47  bool isRight = (iside == 0);
48  double angle = isRight ? 0. : 180.;
49  ROOT::Math::Rotation3D rot(ROOT::Math::RotationY(angle * degToRad));
50  double xSign = (iside == 0) ? 1. : -1.;
51  for (int ichamber = 0; ichamber < 4; ++ichamber) {
52  ROOT::Math::Translation3D planeTrans(0., 0., Constants::sDefaultChamberZ[ichamber]);
53  for (int irpc = 0; irpc < 9; ++irpc) {
54  double xPos = xSign * Constants::getRPCCenterPosX(ichamber, irpc);
55  double sign = (irpc % 2 == 0) ? 1. : -1;
56  if (iside == 1) {
57  sign *= -1.;
58  }
59  double zPos = sign * Constants::sRPCZShift;
60  double newZ = Constants::sDefaultChamberZ[0] + zPos;
61  double oldZ = Constants::sDefaultChamberZ[0] - zPos;
62  double yPos = Constants::getRPCHalfHeight(ichamber) * (irpc - 4) * (1. + newZ / oldZ);
63  int deId = Constants::getDEId(isRight, ichamber, irpc);
64  ROOT::Math::Translation3D trans(xPos, yPos, zPos);
65  mTransformations[deId] = planeTrans * planeRot * trans * rot;
66  LOG(DEBUG) << "DeID " << deId << "\n" << mTransformations[deId];
67  }
68  }
69  }
