33 BOOST_AUTO_TEST_CASE(TestNoWait) {
34  Monitoring metrics;
35  InputSpec spec{ "clusters", "TPC", "CLUSTERS" };

39  };

41  std::vector<ForwardRoute> forwards;
42  TimesliceIndex index;
43 
44  auto policy = CompletionPolicyHelpers::consumeWhenAny();
45  DataRelayer relayer(policy, inputs, forwards, metrics, index);
46  relayer.setPipelineLength(4);
47 
48  // Let's create a dummy O2 Message with two headers in the stack:
49  // - DataHeader matching the one provided in the input
50  DataHeader dh;
51  dh.dataDescription = "CLUSTERS";
52  dh.dataOrigin = "TPC";
53  dh.subSpecification = 0;
54 
55  DataProcessingHeader dph{0,1};
56  Stack stack{dh, dph};
57  auto transport = FairMQTransportFactory::CreateTransportFactory("zeromq");
58  FairMQMessagePtr header = transport->CreateMessage(stack.size());
59  FairMQMessagePtr payload = transport->CreateMessage(1000);
60  memcpy(header->GetData(), stack.data(), stack.size());
61  relayer.relay(std::move(header),std::move(payload));
62  auto ready = relayer.getReadyToProcess();

64  BOOST_CHECK_EQUAL(ready[0].slot.index, 0);
65  BOOST_CHECK_EQUAL(ready[0].op, CompletionPolicy::CompletionOp::Consume);
66  auto result = relayer.getInputsForTimeslice(ready[0].slot);
67  // One for the header, one for the payload
68  BOOST_REQUIRE_EQUAL(result.size(),2);

169  memcpy(header->GetData(), stack.data(), stack.size());
