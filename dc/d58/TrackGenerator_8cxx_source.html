42 std::vector<Track> TrackGenerator::generate(int nTracks)
43 {
46  std::vector<Track> tracks;
47  Track track;
48  std::array<float, 3> pos;
49  std::array<float, 2> dir;
50  for (int itrack = 0; itrack < nTracks; ++itrack) {
51  for (int ipos = 0; ipos < 3; ++ipos) {
52  pos[ipos] =
53  std::uniform_real_distribution<float>{ mPositionLimits[2 * ipos], mPositionLimits[2 * ipos + 1] }(mGenerator);
54  }
55  track.setPosition(pos[0], pos[1], pos[2]);
56 
57  std::array<float, 4> limits = getLimitsForAcceptance(pos);
58 
59  for (int idir = 0; idir < 2; ++idir) {
60  dir[idir] = std::uniform_real_distribution<float>{ limits[2 * idir], limits[2 * idir + 1] }(mGenerator);
61  }
62  track.setDirection(dir[0], dir[1], 1.);
63  tracks.push_back(track);
64  }
65  return tracks;
66 }

69 std::array<float, 4> TrackGenerator::getLimitsForAcceptance(std::array<float, 3> pos)
70 {
73  std::array<float, 4> limits = mSlopeLimits;
74  float dZ = -1600. - pos[2];
75  if (dZ == 0.) {
76  dZ = 0.0001;
77  }
78 
79  // These are the maximum x and y position of MT11 with the standard alignment
80  std::array<float, 2> maxValues{ { 257., 306.7 } };
81  for (int icoor = 0; icoor < 2; ++icoor) {
82  int imin = 2 * icoor;
83  int imax = 2 * icoor + 1;
84  float minSlope = (pos[icoor] - maxValues[icoor]) / dZ;
85  float maxSlope = (maxValues[icoor] - pos[icoor]) / dZ;
86  if (minSlope > limits[imin]) {
87  limits[imin] = minSlope;
88  }
89  if (maxSlope < limits[imax]) {
90  limits[imax] = maxSlope;
91  }
92  }
93  return limits;
94 }
