220 // This in general (as in STL) is a bad idea, but here it is safe to inherit from an allocator since we
221 // have no additional data and only override some methods so we don't get into slicing and other problems.
222 template <typename T>
223 class SpectatorAllocator : public boost::container::pmr::polymorphic_allocator<T>
224 {
225  public:
226  using boost::container::pmr::polymorphic_allocator<T>::polymorphic_allocator;
227 
228  // skip default construction of empty elements
229  // this is important for two reasons: one: it allows us to adopt an existing buffer (e.g. incoming message) and
230  // quickly construct large vectors while skipping the element initialization.
231  template <class U>
232  void construct(U*)
233  {
234  }
235 
236  // dont try to call destructors, makes no sense since resource is managed externally AND allowed
237  // types cannot have side effects
238  template <typename U>
239  void destroy(U*)
240  {
241  }
242 
243  T* allocate(size_t size) { return reinterpret_cast<T*>(this->resource()->allocate(size * sizeof(T), 0)); }
244  void deallocate(T* ptr, size_t size)
245  {
246  this->resource()->deallocate(const_cast<typename std::remove_cv<T>::type*>(ptr), size);
247  }
248 };

315 // return the message associated with the container or nullptr if it does not make sense (e.g. when we are just
316 // watching an existing message or when the container is not using FairMQMemoryResource as backend).
317 template <typename ContainerT>
318 // typename std::enable_if<std::is_base_of<boost::container::pmr::polymorphic_allocator<typename
319 // ContainerT::value_type>,
320 // typename ContainerT::allocator_type>::value == true,
321 // FairMQMessagePtr>::type
322 FairMQMessagePtr getMessage(ContainerT&& container_, FairMQMemoryResource* targetResource = nullptr)
323 {
324  auto container = std::move(container_);
325  auto alloc = container.get_allocator();
326 
327  auto resource = dynamic_cast<FairMQMemoryResource*>(alloc.resource());
328  if (!resource && !targetResource) {
329  throw std::runtime_error("Neither the container or target resource specified");
330  }
331  size_t containerSizeBytes = container.size() * sizeof(typename ContainerT::value_type);
332  if ((!targetResource && resource) || (resource && targetResource && resource->is_equal(*targetResource))) {
333  auto message = resource->getMessage(static_cast<void*>(
334  const_cast<typename std::remove_const<typename ContainerT::value_type>::type*>(container.data())));
335  if (message)
336  message->SetUsedSize(containerSizeBytes);
337  return std::move(message);
338  } else {
339  auto message = targetResource->getTransportFactory()->CreateMessage(containerSizeBytes);
340  std::memcpy(static_cast<o2::byte*>(message->GetData()), container.data(), containerSizeBytes);
341  return std::move(message);
342  }
343 };

365 // This returns a unique_ptr of const vector, does not allow modifications at the cost of pointer
366 // semantics for access.
367 // use auto or decltype to catch the return value (or use span)
368 template <typename ElemT>
369 auto adoptVector(size_t nelem, FairMQMessage* message)
370 {
371  using DataType = std::vector<ElemT, ByteSpectatorAllocator>;
372 
373  struct doubleDeleter {
374  // kids: don't do this at home! (but here it's OK)
375  // this stateful deleter allows a single unique_ptr to manage 2 resources at the same time.
376  std::unique_ptr<SpectatorMessageResource> extra;
377  void operator()(const DataType* ptr) { delete ptr; }
378  };
379 
380  using OutputType = std::unique_ptr<const DataType, doubleDeleter>;
381 
382  auto resource = std::make_unique<SpectatorMessageResource>(message);
383  auto output = new DataType(nelem, ByteSpectatorAllocator{ resource.get() });
384  return OutputType(output, doubleDeleter{ std::move(resource) });
385 }
