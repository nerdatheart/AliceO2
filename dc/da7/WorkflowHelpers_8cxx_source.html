34 std::vector<TopoIndexInfo>
35  WorkflowHelpers::topologicalSort(size_t nodeCount,
36  int const* edgeIn,
37  int const* edgeOut,
38  size_t byteStride,
39  size_t edgesCount)
40 {
41  size_t stride = byteStride / sizeof(int);
42  using EdgeIndex = int;
43  // Create the index which will be returned.
44  std::vector<TopoIndexInfo> index(nodeCount);
45  for (int wi = 0; wi < nodeCount; ++wi) {
46  index[wi] = { wi, 0 };
47  }
48  std::vector<EdgeIndex> remainingEdgesIndex(edgesCount);
49  for (EdgeIndex ei = 0; ei < edgesCount; ++ei) {
50  remainingEdgesIndex[ei] = ei;
51  }
52 
53  // Create a vector where at each position we have true
54  // if the vector has dependencies, false otherwise
55  std::vector<bool> nodeDeps(nodeCount, false);
56  for (EdgeIndex ei = 0; ei < edgesCount; ++ei) {
57  nodeDeps[*(edgeOut + ei * stride)] = true;
58  }
59 
60  // We start with all those which do not have any dependencies
61  // They are layer 0.
62  std::list<TopoIndexInfo> L;
63  for (int ii = 0; ii < index.size(); ++ii) {
64  if (nodeDeps[ii] == false) {
65  L.push_back({ ii, 0 });
66  }
67  }
68 
69  // The final result.
70  std::vector<TopoIndexInfo> S;
71  // The set of vertices which can be reached by the current node
72  std::set<TopoIndexInfo> nextVertex;
73  // The set of edges which are not related to the current node.
74  std::vector<EdgeIndex> nextEdges;
75  while (!L.empty()) {
76  auto node = L.front();
77  S.push_back(node);
78  L.pop_front();
79  nextVertex.clear();
80  nextEdges.clear();
81 
82  // After this, nextVertex will contain all the vertices
83  // which have the current node as incoming.
84  // nextEdges will contain all the edges which are not related
85  // to the current node.
86  for (auto& ei : remainingEdgesIndex) {
87  if (*(edgeIn + ei * stride) == node.index) {
88  nextVertex.insert({ *(edgeOut + ei * stride), node.layer + 1 });
89  } else {
90  nextEdges.push_back(ei);
91  }
92  }
93  remainingEdgesIndex.swap(nextEdges);
94 
95  // Of all the vertices which have node as incoming,
96  // check if there is any other incoming node.
97  std::set<TopoIndexInfo> hasPredecessors;
98  for (auto& ei : remainingEdgesIndex) {
99  for (auto& m : nextVertex) {
100  if (m.index == *(edgeOut + ei * stride)) {
101  hasPredecessors.insert({ m.index, m.layer });
102  }
103  }
104  }
105  std::vector<TopoIndexInfo> withPredecessor;
106  std::set_difference(nextVertex.begin(), nextVertex.end(),
107  hasPredecessors.begin(), hasPredecessors.end(),
108  std::back_inserter(withPredecessor));
109  std::copy(withPredecessor.begin(), withPredecessor.end(), std::back_inserter(L));
110  }
111  return S;
112 }

189 void
190 WorkflowHelpers::constructGraph(const WorkflowSpec &workflow,
191  std::vector<DeviceConnectionEdge> &logicalEdges,
192  std::vector<OutputSpec> &outputs,
193  std::vector<LogicalForwardInfo> &forwardedInputsInfo) {
194  assert(!workflow.empty());
195  // This is the state. Oif is the iterator I use for the searches.
196  std::list<LogicalOutputInfo> availableOutputsInfo;
197  auto const &constOutputs = outputs; // const version of the outputs
198  decltype(availableOutputsInfo.begin()) oif;
199  // Forwards is a local cache to avoid adding forwards before time.
200  std::vector<LogicalOutputInfo> forwards;
201 
202  // Notice that availableOutputsInfo MUST be updated first, since it relies on
203  // the size of outputs to be the one before the update.
204  auto enumerateAvailableOutputs = [&workflow,&outputs,&availableOutputsInfo]() {
205  for (size_t wi = 0; wi < workflow.size(); ++wi) {
206  auto &producer = workflow[wi];
207 
208  for (size_t oi = 0; oi < producer.outputs.size(); ++oi) {
209  auto &out = producer.outputs[oi];
210  auto uniqueOutputId = outputs.size();
211  availableOutputsInfo.emplace_back(LogicalOutputInfo{wi, uniqueOutputId, false});
212  outputs.push_back(out);
213  }
214  }
215  };
216 
217  // Notice that if the output is actually a forward, we need to store that
218  // information so that when we add it at device level we know which output
219  // channel we need to connect it too.
220  auto hasMatchingOutputFor = [&workflow, &constOutputs,
221  &availableOutputsInfo, &oif,
222  &forwardedInputsInfo](size_t ci, size_t ii) {
223  assert(ci < workflow.size());
224  assert(ii < workflow[ci].inputs.size());
225  auto &input = workflow[ci].inputs[ii];
226  auto matcher = [&input, &constOutputs](const LogicalOutputInfo &outputInfo) -> bool {
227  auto &output = constOutputs[outputInfo.outputGlobalIndex];
228  return intersect(LogicalChannelDomain(input), LogicalChannelRange(output));
229  };
230  oif = std::find_if(availableOutputsInfo.begin(),
231  availableOutputsInfo.end(),
232  matcher);
233  if (oif != availableOutputsInfo.end() && oif->forward) {
234  LogicalForwardInfo forward;
235  forward.consumer = ci;
236  forward.inputLocalIndex = ii;
237  forward.outputGlobalIndex = oif->outputGlobalIndex;
238  forwardedInputsInfo.emplace_back(LogicalForwardInfo{ci, ii, oif->outputGlobalIndex});
239  }
240  return oif != availableOutputsInfo.end();
241  };
242 
243  // We have consumed the input, therefore we remove it from the list.
244  // We will insert the forwarded inputs only at the end of the iteration.
245  auto findNextOutputFor = [&availableOutputsInfo,&constOutputs,&oif,&workflow](
246  size_t ci, size_t &ii) {
247  auto &input = workflow[ci].inputs[ii];
248  auto matcher = [&input, &constOutputs](const LogicalOutputInfo &outputInfo) -> bool {
249  auto &output = constOutputs[outputInfo.outputGlobalIndex];
250  return intersect(LogicalChannelDomain(input), LogicalChannelRange(output));
251  };
252  oif = availableOutputsInfo.erase(oif);
253  oif = std::find_if(oif, availableOutputsInfo.end(), matcher);
254  return oif;
255  };
256 
257  auto numberOfInputsFor = [&workflow](size_t ci) {
258  auto &consumer = workflow[ci];
259  return consumer.inputs.size();
260  };
261 
262  auto maxInputTimeslicesFor = [&workflow](size_t pi) {
263  auto &processor = workflow[pi];
264  return processor.maxInputTimeslices;
265  };
266 
267  // Trivial, but they make reading easier..
268  auto getOutputAssociatedProducer = [&oif]() {
269  return oif->specIndex;
270  };
271 
272  // Trivial, but they make reading easier..
273  auto getAssociateOutput = [&oif]() {
274  return oif->outputGlobalIndex;
275  };
276 
277  auto isForward = [&oif]() {

281  // Trivial but makes reasing easier in the outer loop.
282  auto createEdge = [&logicalEdges](size_t producer,
283  size_t consumer,
284  size_t tpi,
285  size_t ptpi,

298 
299  auto errorDueToMissingOutputFor = [&workflow](size_t ci, size_t ii) {
300  auto input = workflow[ci].inputs[ii];
301  std::ostringstream str;
302  str << "No matching output found for " << input.origin.as<std::string>() << " "
303  << input.description.as<std::string>() << " "
304  << input.subSpec << "\n";
305  throw std::runtime_error(str.str());
306  };
307 
308 
309  // Whenever we have a set of forwards, we need to append it
310  // the the global list of outputs, so that they can be matched
311  // and we need to add a ForwardRoute for the current consumer
312  // because it is the one who will actually do the forwarding.
313  auto appendForwardsToPossibleOutputs = [&availableOutputsInfo, &forwards]() {
314  for(auto &forward : forwards) {
315  availableOutputsInfo.push_back(forward);
316  }
317  };
318 
319  // Given we create a forward every time we match and input and an
320  // output, having no forwards means we did not find any matching.
321  auto noMatchingOutputFound = [&forwards]() {
322  return forwards.empty();
323  };
324 
325  // Forwards is basically a cache to record
326  auto newEdgeBetweenDevices = [&forwards]() {
327  forwards.clear();
328  };
329 
330  auto forwardOutputFrom = [&forwards](size_t consumer, size_t uniqueOutputId) {
331  forwards.push_back(LogicalOutputInfo{consumer, uniqueOutputId, true});
332  };
333 
334  // This is the outer loop
335  //
336  // Here we iterate over dataprocessor items in workflow and we consider them
337  // as consumer, since we are interested in their inputs.
338  // Notice also we need to search for all the matching inputs, since
339  // we could have more than one source that matches (e.g. in the
340  // case of a time merger).
341  // Once consumed, an output is not actually used anymore, however
342  // we append it as a forward.
343  // Finally, If a device has n-way pipelining, we need to create one node per
344  // parallel pipeline and add an edge for each.
345  enumerateAvailableOutputs();
346 
347  for (size_t consumer = 0; consumer < workflow.size(); ++consumer) {
348  for (size_t input = 0; input < numberOfInputsFor(consumer); ++input) {
349  newEdgeBetweenDevices();
350 
351  while (hasMatchingOutputFor(consumer, input)) {
352  auto producer = getOutputAssociatedProducer();
353  auto uniqueOutputId = getAssociateOutput();
354  for (size_t tpi = 0; tpi < maxInputTimeslicesFor(consumer); ++tpi) {
355  for (size_t ptpi = 0; ptpi < maxInputTimeslicesFor(producer); ++ptpi) {
356  createEdge(producer, consumer, tpi, ptpi, uniqueOutputId, input, isForward());
357  }
358  forwardOutputFrom(consumer, uniqueOutputId);
359  }
360  findNextOutputFor(consumer, input);
361  }
362  if (noMatchingOutputFound()) {
363  errorDueToMissingOutputFor(consumer, input);
364  }
365  appendForwardsToPossibleOutputs();
366  }
367  }
368 }

370 std::vector<EdgeAction>
371 WorkflowHelpers::computeOutEdgeActions(
372  const std::vector<DeviceConnectionEdge> &edges,
373  const std::vector<size_t> &index
374  ) {
375  DeviceConnectionEdge last{ULONG_MAX,ULONG_MAX,ULONG_MAX,ULONG_MAX,ULONG_MAX,ULONG_MAX};
376 
377  assert(edges.size() == index.size());
378  std::vector<EdgeAction> actions(edges.size(), EdgeAction{false, false});
379  for (size_t i : index) {
380  auto &edge = edges[i];
381  auto &action = actions[i];
382  action.requiresNewDevice = last.producer != edge.producer
383  || last.producerTimeIndex != edge.producerTimeIndex;
384  action.requiresNewChannel = last.consumer != edge.consumer
385  || last.producer != edge.producer
386  || last.timeIndex != edge.timeIndex
387  || last.producerTimeIndex != edge.producerTimeIndex;
388  last = edge;
389  }
390  return actions;
391 }

419 void
420 WorkflowHelpers::sortEdges(std::vector<size_t> &inEdgeIndex,
421  std::vector<size_t> &outEdgeIndex,
422  const std::vector<DeviceConnectionEdge> &edges) {
423  inEdgeIndex.resize(edges.size());
424  outEdgeIndex.resize(edges.size());
425  std::iota(inEdgeIndex.begin(), inEdgeIndex.end(), 0);
426  std::iota(outEdgeIndex.begin(), outEdgeIndex.end(), 0);
427 
428  // Two indexes, one to bind the outputs, the other
429  // one to connect the inputs. The
430  auto outSorter = [&edges](size_t i, size_t j) {
431  auto &a = edges[i];
432  auto &b = edges[j];
433  return std::tie(a.producer, a.producerTimeIndex, a.timeIndex, a.consumer)
434  < std::tie(b.producer, b.producerTimeIndex, b.timeIndex, b.consumer);
435  };
436  auto inSorter = [&edges](size_t i, size_t j) {
437  auto &a = edges[i];
438  auto &b = edges[j];
439  return std::tie(a.consumer, a.timeIndex, a.producer, a.producerTimeIndex)
440  < std::tie(b.consumer, b.timeIndex, b.producer, b.producerTimeIndex);
441  };
442 
443  std::sort(inEdgeIndex.begin(), inEdgeIndex.end(), inSorter);
444  std::sort(outEdgeIndex.begin(), outEdgeIndex.end(), outSorter);
445 }

449  if (workflow.empty()){

451  }

456  // An index many to one index to go from a given input to the

463 
464  for (auto &spec : workflow)

502 std::vector<InputSpec> WorkflowHelpers::computeDanglingOutputs(WorkflowSpec const& workflow)
503 {
504  std::vector<UnifiedDataSpecType> tmp;
505  std::vector<InputSpec> results;
506 
507  for (auto& spec : workflow) {
508  for (auto& input : spec.inputs) {
509  tmp.push_back({ input.origin, input.description, input.subSpec, 0 });
510  }
511  for (auto& output : spec.outputs) {
512  tmp.push_back({ output.origin, output.description, output.subSpec, 1 });
513  }
514  }
515 
516  auto cmp = [](UnifiedDataSpecType const& lhs, UnifiedDataSpecType const& rhs) -> bool {
517  return std::tie(lhs.origin, lhs.description, lhs.subSpec, lhs.isOutput) <
518  std::tie(rhs.origin, rhs.description, rhs.subSpec, rhs.isOutput);
519  };
520  std::sort(tmp.begin(), tmp.end(), cmp);
521 
522  // Once sorted, all the transitions which begin with an output
523  // mean there was no input.
524  bool isFirst = true;
525  UnifiedDataSpecType last;
526  int i = 0;
527  for (auto& unified : tmp) {
528  if (last.origin != unified.origin ||
529  last.description != unified.description ||
530  last.subSpec != unified.subSpec) {
531  isFirst = true;
532  last = unified;
533  }
534  if (isFirst && unified.isOutput == 0) {
535  isFirst = false;
536  continue;
537  }
538  if (isFirst && unified.isOutput == 1) {
539  isFirst = false;
540  char buf[64];
541  results.push_back(InputSpec{ (snprintf(buf, 64, "dangling%d", i), buf), unified.origin,
542  unified.description, unified.subSpec });
543  ++i;
544  continue;
545  }
546  assert(isFirst == false);
547  }
548 
549  return results;
550 }
