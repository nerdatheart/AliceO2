74  // should go into a helper
75  // this function queries the sim config data and initializes the SimConfig singleton
76  // returns true if successful / false if not
77  static bool querySimConfig(FairMQChannel& channel)
78  {
79  auto text = new std::string("configrequest");
80  std::unique_ptr<FairMQMessage> request(channel.NewMessage(const_cast<char*>(text->c_str()),
81  text->length(), CustomCleanup, text));
82  std::unique_ptr<FairMQMessage> reply(channel.NewMessage());
83 
84  int timeoutinMS = 100000; // wait for 100s max
85  if (channel.Send(request, timeoutinMS) >= 0) {
86  LOG(INFO) << "Waiting for configuration answer ";
87  if (channel.Receive(reply, timeoutinMS) > 0) {
88  LOG(INFO) << "Configuration answer received, containing " << reply->GetSize() << " bytes ";
89 
90  // the answer is a TMessage containing the simulation Configuration
91  auto message = std::make_unique<o2::devices::TMessageWrapper>(reply->GetData(), reply->GetSize());
92  auto config = static_cast<o2::conf::SimConfigData*>(message.get()->ReadObjectAny(message.get()->GetClass()));
93  if (!config) {
94  return false;
95  }
96 
97  LOG(INFO) << "COMMUNICATED ENGINE " << config->mMCEngine;
98  auto& conf = o2::conf::SimConfig::Instance();
99  conf.resetFromConfigData(*config);
100  } else {
101  LOG(ERROR) << "No configuration received within " << timeoutinMS << "ms\n";
102  return false;
103  }
104  } else {
105  LOG(ERROR) << "Could not send configuration request within " << timeoutinMS << "ms\n";
106  return false;
107  }
108  return true;
109  }

111  // initializes the simulation classes; queries the configuration on a given channel
112  static bool initSim(FairMQChannel& channel, std::unique_ptr<FairRunSim>& simptr)
113  {
114  if (!querySimConfig(channel)) {
115  return false;
116  }
117 
118  LOG(INFO) << "Setting up the simulation ...";
119  simptr = std::move(std::unique_ptr<FairRunSim>(o2sim_init(true)));

122  // to finish initialization (trigger further cross section table building etc) -- which especially

127  LOG(INFO) << "MEM-STAMP END OF SIM INIT" << sysinfo.GetCurrentMemory() / (1024. * 1024) << " "

129 
130  return true;

133  bool Kernel(FairMQChannel& requestchannel, FairMQChannel& dataoutchannel)
134  {
135  auto text = new std::string("primrequest");
136 
137  // create message object with a pointer to the data buffer,
138  // its size,
139  // custom deletion function (called when transfer is done),
140  // and pointer to the object managing the data buffer
141  FairMQMessagePtr request(requestchannel.NewMessage(const_cast<char*>(text->c_str()), // data
142  text->length(), // size
143  CustomCleanup,
144  text));
145  FairMQMessagePtr reply(dataoutchannel.NewMessage());
146 
147  mVMCApp->setSimDataChannel(&dataoutchannel);
148 
149  LOG(INFO) << "Requesting work ";
150  int timeoutinMS = 10000; // wait for 10s max -- we should have a more robust solution
151  if (requestchannel.Send(request, timeoutinMS) >= 0) {
152  LOG(INFO) << "Waiting for answer ";
153  // asking for primary generation
154 
155  if (requestchannel.Receive(reply, timeoutinMS) > 0) {
156  LOG(INFO) << "Answer received, containing " << reply->GetSize() << " bytes " << FairLogger::endl;
157 
158  // wrap incoming bytes as a TMessageWrapper which offers "adoption" of a buffer
159  auto message = new TMessageWrapper(reply->GetData(), reply->GetSize());
160  auto chunk = static_cast<o2::Data::PrimaryChunk*>(message->ReadObjectAny(message->GetClass()));
161 
162  mVMCApp->setPrimaries(chunk->mParticles);
163 
164  auto info = chunk->mEventIDs[0];
165  mVMCApp->setSubEventInfo(info);
166 
167  LOG(INFO) << "Processing " << chunk->mParticles.size() << FairLogger::endl;
168  gRandom->SetSeed(chunk->mEventIDs[0].seed);
169 
170  mVMC->ProcessRun(1);
171  FairSystemInfo sysinfo;
172  LOG(INFO) << "TIME-STAMP " << mTimer.RealTime() << "\t";
173  mTimer.Continue();
174  LOG(INFO) << "MEM-STAMP " << sysinfo.GetCurrentMemory() / (1024. * 1024) << " "
175  << sysinfo.GetMaxMemory() << " MB\n";
176  delete message;
177  } else {
178  return false;
179  }
180  } else {
181  return false;
182  }
183  return true;
184  }
