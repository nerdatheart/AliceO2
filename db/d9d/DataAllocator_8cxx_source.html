41 std::string
42 DataAllocator::matchDataHeader(const Output& spec, size_t timeslice) {
43  // FIXME: we should take timeframeId into account as well.
44  for (auto &output : mAllowedOutputRoutes) {
45  if (DataSpecUtils::match(output.matcher, spec.origin, spec.description, spec.subSpec)
46  && ((timeslice % output.maxTimeslices) == output.timeslice)) {
47  return output.channel;
48  }
49  }
50  std::ostringstream str;
51  str << "Worker is not authorised to create message with "
52  << "origin(" << spec.origin.as<std::string>() << ")"
53  << "description(" << spec.description.as<std::string>() << ")"
54  << "subSpec(" << spec.subSpec << ")";
55  throw std::runtime_error(str.str());
56 }

68  auto dataPtr = payloadMessage->GetData();

81 DataAllocator::adoptChunk(const Output& spec, char *buffer, size_t size, fairmq_free_fn *freefn, void *hint = nullptr) {
82  // Find a matching channel, create a new message for it and put it in the
83  // queue to be sent at the end of the processing

85 
86  FairMQMessagePtr headerMessage = headerMessageFromOutput(spec, channel, //
87  o2::header::gSerializationMethodNone, //
88  size //
89  );
90 
91  FairMQParts parts;
92 
93  // FIXME: how do we want to use subchannels? time based parallelism?
94  auto context = mContextRegistry->get<MessageContext>();
95  FairMQMessagePtr payloadMessage = context->proxy().getDevice()->NewMessageFor(channel, 0, buffer, size, freefn, hint);
96  auto dataPtr = payloadMessage->GetData();
97  LOG(DEBUG) << "New payload at " << payloadMessage->GetData();
98  auto dataSize = payloadMessage->GetSize();
99  parts.AddPart(std::move(headerMessage));

101  context->addPart(std::move(parts), channel);
102  return DataChunk{reinterpret_cast<char *>(dataPtr), dataSize};

177 Output DataAllocator::getOutputByBind(OutputRef&& ref)
178 {
179  if (ref.label.empty()) {
180  throw std::runtime_error("Invalid (empty) OutputRef provided.");
181  }
182  for (size_t ri = 0, re = mAllowedOutputRoutes.size(); ri != re; ++ri) {
183  if (mAllowedOutputRoutes[ri].matcher.binding.value == ref.label) {
184  auto spec = mAllowedOutputRoutes[ri].matcher;
185  return Output{ spec.origin, spec.description, ref.subSpec, spec.lifetime, std::move(ref.headerStack) };
186  }
187  }
188  throw std::runtime_error("Unable to find OutputSpec with label " + ref.label);
189  assert(false);
190 }
