101  template <typename T, typename V>
102  V insertAdd(std::map<T, V>& m, T const& key, V value)
103  {
104  const auto iter = m.find(key);
105  V accum{ 0 };
106  if (iter != m.end()) {
107  iter->second += value;
108  accum = iter->second;
109  } else {
110  m.insert(std::make_pair(key, value));
111  accum = value;
112  }
113  return accum;
114  }

122  void consumeHits(FairMQParts& data, int& index)
123  {
124  auto detIDmessage = std::move(data.At(index++));
125  // this should be a detector ID
126  LOG(INFO) << detIDmessage->GetSize();
127  if (detIDmessage->GetSize() == 4) {
128  auto ptr = (int*)detIDmessage->GetData();
129  o2::detectors::DetID id(ptr[0]);
130  LOG(INFO) << "I1 " << ptr[0] << " NAME " << id.getName();
131 
132  // get the detector than can interpret it
133  auto detector = mDetectorInstances[id].get();
134  detector->fillHitBranch(*mOutTree, data, index);
135  }
136  }

149  bool handleSimData(FairMQParts& data, int /*index*/)
150  {
151  LOG(INFO) << "SIMDATA channel got " << data.Size() << " parts\n";
152 
153  // extract the event info
154  auto infomessage = std::move(data.At(0));
155  o2::Data::SubEventInfo info;
156  // could actually avoid the copy
157  memcpy((void*)&info, infomessage->GetData(), infomessage->GetSize());
158 
159  auto accum = insertAdd<uint32_t, uint32_t>(mPartsCheckSum, info.eventID, (uint32_t)info.part);
160  // auto totalsize = insertAdd<uint32_t, size_t>(mITSTotalSize, info.eventID, itshits.size());
161 
162  int index = 1;
163  consumeData<std::vector<o2::MCTrack>>("MCTrack", data, index);
164  consumeData<std::vector<o2::TrackReference>>("TrackRefs", data, index);
165  consumeData<o2::dataformats::MCTruthContainer<o2::TrackReference>>("IndexedTrackRefs", data, index);
166  while (index < data.Size()) {
167  consumeHits(data, index);
168  }
169  mEntries++;
170 
171  if (isDataComplete<uint32_t>(accum, info.nparts)) {
172  LOG(INFO) << "EVERYTHING IS HERE FOR EVENT " << info.eventID << "\n";
173  mEventChecksum += info.eventID;
174  // we also need to check if we have all events
175  if (isDataComplete<uint32_t>(mEventChecksum, info.maxEvents)) {
176  return false;
177  }
178  }
179  return true;
180  }

197 // init detector instances used to write hit data to a TTree
198 void O2HitMerger::initDetInstances()
199 {
200  using o2::detectors::DetID;
201 
202  mDetectorInstances.resize(DetID::Last);
203  // like a factory of detector objects
204 
205  int counter = 0;
206  for (int i = DetID::First; i <= DetID::Last; ++i) {
207  if (i == DetID::TPC) {
208  mDetectorInstances[i] = std::move(std::make_unique<o2::TPC::Detector>(true));
209  counter++;
210  }
211  if (i == DetID::ITS) {
212  mDetectorInstances[i] = std::move(std::make_unique<o2::ITS::Detector>(true));
213  counter++;
214  }
215  if (i == DetID::MFT) {
216  mDetectorInstances[i] = std::move(std::make_unique<o2::MFT::Detector>());
217  counter++;
218  }
219  if (i == DetID::TRD) {
220  mDetectorInstances[i] = std::move(std::make_unique<o2::trd::Detector>(true));
221  counter++;
222  }
223  if (i == DetID::PHS) {
224  mDetectorInstances[i] = std::move(std::make_unique<o2::phos::Detector>(true));
225  counter++;
226  }
227  if (i == DetID::EMC) {
228  mDetectorInstances[i] = std::move(std::make_unique<o2::EMCAL::Detector>(true));
229  counter++;
230  }
231  if (i == DetID::HMP) {
232  mDetectorInstances[i] = std::move(std::make_unique<o2::hmpid::Detector>(true));
233  counter++;
234  }
235  if (i == DetID::TOF) {
236  mDetectorInstances[i] = std::move(std::make_unique<o2::tof::Detector>(true));
237  counter++;
238  }
239  if (i == DetID::FIT) {
240  mDetectorInstances[i] = std::move(std::make_unique<o2::fit::Detector>(true));
241  counter++;
242  }
243  if (i == DetID::MCH) {
244  mDetectorInstances[i] = std::move(std::make_unique<o2::mch::Detector>(true));
245  counter++;
246  }
247  }
248  if (counter != DetID::Last) {
249  LOG(WARNING) << " O2HitMerger: Some Detectors are potentially missing in this initialization ";
250  }
251 }
