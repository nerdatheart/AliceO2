83  static bool finished = false;

88  if (firstChannel == channel) {
89  auto roMode = digitizertask->isContinuousReadout() ? o2::parameters::GRPObject::CONTINUOUS : o2::parameters::GRPObject::PRESENT;

96  // extract which sector to treat
97  using TPCSectorHeader = o2::TPC::TPCSectorHeader;

99  if (sectorHeader == nullptr) {
100  LOG(ERROR) << "TPC sector header missing, skipping processing";

103  auto sector = sectorHeader->sector;
104  LOG(INFO) << "GOT ASSIGNED SECTOR " << sector;
105  // the active sectors need to be propagated
106  uint64_t activeSectors = 0;
107  activeSectors = sectorHeader->activeSectors;

115  // lambda that snapshots digits to be sent out; prepares and attaches header with sector information
116  auto snapshotDigits = [sector, &pc, channel, activeSectors](std::vector<o2::TPC::Digit> const& digits) {
117  o2::TPC::TPCSectorHeader header{ sector };
118  header.activeSectors = activeSectors;

120  pc.outputs().snapshot(Output{ "TPC", "DIGITS", static_cast<SubSpecificationType>(channel), Lifetime::Timeframe,
121  header },

124  // lambda that snapshots labels to be sent out; prepares and attaches header with sector information
125  auto snapshotLabels = [&sector, &pc, &channel, activeSectors](o2::dataformats::MCTruthContainer<o2::MCCompLabel> const& labels) {
126  o2::TPC::TPCSectorHeader header{ sector };
127  header.activeSectors = activeSectors;
128  pc.outputs().snapshot(Output{ "TPC", "DIGITSMCTR", static_cast<SubSpecificationType>(channel),
129  Lifetime::Timeframe, header },
130  const_cast<o2::dataformats::MCTruthContainer<o2::MCCompLabel>&>(labels));
131  };
132  // lambda that snapshots digits grouping (triggers) to be sent out; prepares and attaches header with sector information
133  auto snapshotEvents = [sector, &pc, channel, activeSectors](const std::vector<DigiGroupRef>& events) {
134  o2::TPC::TPCSectorHeader header{ sector };
135  header.activeSectors = activeSectors;
136  LOG(INFO) << "Send TRIGGERS for sector " << sector << " channel " << channel << " | size " << events.size();
137  pc.outputs().snapshot(Output{ "TPC", "DIGTRIGGERS", static_cast<SubSpecificationType>(channel), Lifetime::Timeframe,
138  header },
139  const_cast<std::vector<DigiGroupRef>&>(events));
140  };
141 
142  // no more tasks can be marked with a negative sector
143  if (sector < 0) {
144  // we are ready to quit or we received a NOP
145 
146  // notify channels further down the road that we are done (why do I have to send digitArray here????)
147  // (we are essentially sending empty messages with a header containing the stop signal)
148  digitArrayRaw->clear();
149  mcTruthArrayRaw->clear();
150  std::vector<DigiGroupRef> evAccDummy;
151 
152  snapshotEvents(evAccDummy);
153  snapshotDigits(*digitArrayRaw);
154  snapshotLabels(*mcTruthArrayRaw);
155  if (sector == -1) {
156  pc.services().get<ControlService>().readyToQuit(false);
157  finished = true;
158  }
159  return;

162  // obtain collision contexts
163  auto context = pc.inputs().get<o2::steer::RunContext*>("collisioncontext");
164  auto& timesview = context->getEventRecords();
165  LOG(DEBUG) << "GOT " << timesview.size() << " COLLISSION TIMES";
166 
167  // if there is nothing ... return
168  if (timesview.size() == 0) {
169  return;
170  }
171 
172  TStopwatch timer;
173  timer.Start();
174 
175  // detect number of possible drift times (remember that a drift
176  // time is convenient unit for digit pileup), any multiple of this
177  // unit should also be ok
178  const auto TPCDRIFT = 100000;
179  double maxtime = 0;
180  for (auto e : timesview) {
181  maxtime = std::max(maxtime, e.timeNS);
182  }
183 
184  std::vector<std::vector<o2::TPC::HitGroup>*> hitvectorsleft; // "TPCHitVector"
185  std::vector<o2::TPC::TPCHitGroupID> hitidsleft; // "TPCHitIDs"
186  std::vector<std::vector<o2::TPC::HitGroup>*> hitvectorsright; // "TPCHitVector"
187  std::vector<o2::TPC::TPCHitGroupID> hitidsright; // "TPCHitIDs"
188 
189  // try to access event parts information
190  const auto& parts = context->getEventParts();
191  // query max size of entries
192  const int maxsourcespercollision = context->getMaxNumberParts();
193  const int numberofcollisions = context->getNCollisions();
194  const int maxnumberofentries = maxsourcespercollision * numberofcollisions;
195  LOG(INFO) << " MAX ENTRIES " << maxnumberofentries;
196  hitvectorsleft.resize(maxnumberofentries, nullptr);
197  hitvectorsright.resize(maxnumberofentries, nullptr);
198 
199  // this is to accum digits from different drift times
200  std::vector<o2::TPC::Digit> digitAccum;
201  o2::dataformats::MCTruthContainer<o2::MCCompLabel> labelAccum;
202  std::vector<DigiGroupRef> eventAccum;
203 
204  if (digitizertask->isContinuousReadout()) {
205  // minimum 2 drifts is a safe bet; an electron might
206  // need 1 full drift and might hence land in the second drift time
207  auto ndrifts = 2 + (int)(maxtime / TPCDRIFT);
208  for (int drift = 1; drift <= ndrifts; ++drift) {
209  auto starttime = (drift - 1) * TPCDRIFT;
210  auto endtime = drift * TPCDRIFT;
211  LOG(DEBUG) << "STARTTIME " << starttime << " ENDTIME " << endtime;
212  digitizertask->setStartTime(starttime);
213  digitizertask->setEndTime(endtime);
214 
215  hitidsleft.clear();
216  hitidsright.clear();
217 
218  // obtain candidate hit(ids) for this time range --> left
219  o2::TPC::getHits(*simChains.get(), *context.get(), hitvectorsleft, hitidsleft, getBranchNameLeft(sector).c_str(),
220  starttime, endtime, o2::TPC::calcDriftTime);
221  // --> right
222  o2::TPC::getHits(*simChains.get(), *context.get(), hitvectorsright, hitidsright,
223  getBranchNameRight(sector).c_str(), starttime, endtime, o2::TPC::calcDriftTime);
224 
225  LOG(DEBUG) << "DRIFTTIME " << drift << " SECTOR " << sector << " : SELECTED LEFT " << hitidsleft.size() << " IDs"
226  << " SELECTED RIGHT " << hitidsright.size();
227 
228  // invoke digitizer if anything to digitize within this drift interval
229  if (hitidsleft.size() > 0 || hitidsright.size() > 0) {
230  digitizertask->setData(&hitvectorsleft, &hitvectorsright, &hitidsleft, &hitidsright, context.get());
231  digitizertask->setupSector(sector);
232  digitizertask->Exec2("");
233 
234  std::copy(digitArrayRaw->begin(), digitArrayRaw->end(), std::back_inserter(digitAccum));
235  labelAccum.mergeAtBack(*mcTruthArrayRaw);
236 
237  // NOTE: we would like to send it here in order to avoid copying/accumulating !!
238  }
239  }
240  eventAccum.emplace_back(0, digitAccum.size()); // all digits are grouped to 1 super-event
241  } else { // pseudo-triggered mode
242  for (int coll = 0; coll < numberofcollisions; coll++) {
243  int startSize = digitAccum.size();
244  digitizertask->setStartTime(0);
245  digitizertask->setEndTime(2 * TPCDRIFT);
246  hitidsleft.clear();
247  hitidsright.clear();
248  // obtain candidate hit(ids) for this collision --> left
249  o2::TPC::getHits(*simChains.get(), *context.get(), hitvectorsleft, hitidsleft,
250  getBranchNameLeft(sector).c_str(), coll);
251  // obtain candidate hit(ids) for this collision --> right
252  o2::TPC::getHits(*simChains.get(), *context.get(), hitvectorsright, hitidsright,
253  getBranchNameRight(sector).c_str(), coll);
254  int tmpND = 0;
255  if (hitidsleft.size() > 0 || hitidsright.size() > 0) {
256  digitizertask->setData(&hitvectorsleft, &hitvectorsright, &hitidsleft, &hitidsright, context.get());
257  digitizertask->setupSector(sector);
258  digitizertask->Exec2("");
259 
260  std::copy(digitArrayRaw->begin(), digitArrayRaw->end(), std::back_inserter(digitAccum));
261  labelAccum.mergeAtBack(*mcTruthArrayRaw);
262  // NOTE: we would like to send it here in order to avoid copying/accumulating !!
263  tmpND = digitArrayRaw->size();
264  }
265  // register digits from this collision as a pseudo-trigger
266  eventAccum.emplace_back(startSize, digitAccum.size() - startSize);
267  }
268  }
269  // snapshot / "send" digits + MC truth
270  snapshotEvents(eventAccum);
271  snapshotDigits(digitAccum);
272  snapshotLabels(labelAccum);
273  timer.Stop();
274  LOG(INFO) << "Digitization took " << timer.CpuTime() << "s";
275  };

277  // init function return a lambda taking a ProcessingContext
278  auto initIt = [digitizertask, digitArray, mcTruthArray, simChain, simChains, doit](InitContext& ctx) {
279  // Switch on distortions and get initial space-charge density histogram if provided in environment variables
280  auto useDistortions = ctx.options().get<int>("distortionType");
281  auto gridSizeString = ctx.options().get<std::string>("gridSize");
282  auto triggeredMode = ctx.options().get<bool>("TPCtriggered");
283 
284  std::vector<int> gridSize;
285  std::stringstream ss(gridSizeString);
286  while (ss.good()) {
287  std::string substr;
288  getline(ss, substr, ',');
289  gridSize.push_back(std::stoi(substr));
290  }
291  auto inputHistoString = ctx.options().get<std::string>("initialSpaceChargeDensity");
292  std::vector<std::string> inputHisto;
293  std::stringstream ssHisto(inputHistoString);
294  while (ssHisto.good()) {
295  std::string substr;
296  getline(ssHisto, substr, ',');
297  inputHisto.push_back(substr);
298  }
299  if (useDistortions > 0) {
300  o2::TPC::SpaceCharge::SCDistortionType distortionType = useDistortions == 1 ? o2::TPC::SpaceCharge::SCDistortionType::SCDistortionsRealistic : o2::TPC::SpaceCharge::SCDistortionType::SCDistortionsConstant;
301  std::unique_ptr<TH3> hisSCDensity;
302  if (TString(inputHisto[0].data()).EndsWith(".root") && inputHisto[1].size() != 0) {
303  auto fileSCInput = std::unique_ptr<TFile>(TFile::Open(inputHisto[0].data()));
304  hisSCDensity.reset((TH3*)fileSCInput->Get(inputHisto[1].data()));
305  hisSCDensity->SetDirectory(nullptr);
306  }
307  digitizertask->enableSCDistortions(distortionType, hisSCDensity.get(), gridSize[0], gridSize[1], gridSize[2]);
308  }
309  digitizertask->setContinuousReadout(!triggeredMode);
310 
311  digitizertask->Init2();
312  // the task takes the ownership of digit array + mc truth array
313  // TODO: make this clear in the API
314  digitizertask->setOutputData(digitArray.get(), mcTruthArray.get());
315 
316  // setup the input chain
317  simChains->emplace_back(new TChain("o2sim"));
318  // add the background chain
319  simChains->back()->AddFile(ctx.options().get<std::string>("simFile").c_str());
320  auto signalfilename = ctx.options().get<std::string>("simFileS");
321  if (signalfilename.size() > 0) {
322  simChains->emplace_back(new TChain("o2sim"));
323  simChains->back()->AddFile(signalfilename.c_str());
324  }
325 
326  LOG(INFO) << "HAVE " << simChains->size() << " chains \n";
327 
328  simChain->AddFile(ctx.options().get<std::string>("simFile").c_str());
329  return doit;
330  };
