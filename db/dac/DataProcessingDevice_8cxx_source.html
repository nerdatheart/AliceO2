100 bool DataProcessingDevice::ConditionalRun()
101 {
102  mServiceRegistry.get<CallbackService>()(CallbackService::Id::ClockTick);
103  bool active = false;
104  for (auto& channel : mSpec.inputChannels) {
105  FairMQParts parts;
106  auto result = this->ReceiveAsync(parts, channel.name);
107  if (result > 0) {
108  this->handleData(parts);
109  active |= this->tryDispatchComputation();
110  }
111  }
112  if (active == false) {
113  mServiceRegistry.get<CallbackService>()(CallbackService::Id::Idle);
114  }
115  mRelayer.processDanglingInputs(mExpirationHandlers, mServiceRegistry);
116  this->tryDispatchComputation();
117  return true;
118 }

124 bool DataProcessingDevice::handleData(FairMQParts& parts)
125 {
126  assert(mSpec.inputChannels.empty() == false);
127  assert(parts.Size() > 0);
128 
129  static const std::string handleDataMetricName = "dpl/in_handle_data";
130  // Initial part. Let's hide all the unnecessary and have
131  // simple lambdas for each of the steps I am planning to have.
132  assert(!mSpec.inputs.empty());
133 
134  // These duplicate references are created so that each function
135  // does not need to know about the whole class state, but I can
136  // fine grain control what is exposed at each state.
137  auto& monitoringService = mServiceRegistry.get<Monitoring>();
138  monitoringService.send({ 1, "dpl/in_handle_data" });
139  ScopedExit metricFlusher([&monitoringService] {
140  monitoringService.send({ 1, "dpl/in_handle_data"});
141  monitoringService.send({ 0, "dpl/in_handle_data"});
142  monitoringService.flushBuffer(); });
143 
144  auto& device = *this;
145  auto& errorCount = mErrorCount;
146  auto& relayer = mRelayer;
147  auto& serviceRegistry = mServiceRegistry;
148 
149  // This is how we validate inputs. I.e. we try to enforce the O2 Data model
150  // and we do a few stats. We bind parts as a lambda captured variable, rather
151  // than an input, because we do not want the outer loop actually be exposed
152  // to the implementation details of the messaging layer.
153  auto isValidInput = [&monitoringService, &parts]() -> bool {
154  // monitoringService.send({ (int)parts.Size(), "inputs/parts/total" });
155  monitoringService.send({ (int)parts.Size(), "inputs/parts/total" });
156 
157  for (size_t i = 0; i < parts.Size() ; ++i) {
158  LOG(DEBUG) << " part " << i << " is " << parts.At(i)->GetSize() << " bytes";
159  }
160  if (parts.Size() % 2) {
161  return false;
162  }
163  for (size_t hi = 0; hi < parts.Size()/2; ++hi) {
164  auto pi = hi*2;
165  auto dh = o2::header::get<DataHeader*>(parts.At(pi)->GetData());
166  if (!dh) {
167  LOG(ERROR) << "Header is not a DataHeader?";
168  return false;
169  }
170  if (dh->payloadSize != parts.At(pi+1)->GetSize()) {
171  LOG(ERROR) << "DataHeader payloadSize mismatch";
172  return false;
173  }
174  auto dph = o2::header::get<DataProcessingHeader*>(parts.At(pi)->GetData());
175  if (!dph) {
176  LOG(ERROR) << "Header stack does not contain DataProcessingHeader";
177  return false;
178  }
179  LOG(DEBUG) << "Timeslice is " << dph->startTime;
180  LOG(DEBUG) << " DataOrigin is " << dh->dataOrigin.str;
181  LOG(DEBUG) << " DataDescription is " << dh->dataDescription.str;
182  }
183  return true;
184  };

186  auto reportError = [&device](const char* message) {
187  device.error(message);
188  };
189 
190  auto putIncomingMessageIntoCache = [&parts,&relayer,&reportError]() {
191  // We relay execution to make sure we have a complete set of parts
192  // available.
193  for (size_t pi = 0; pi < (parts.Size()/2); ++pi) {
194  auto headerIndex = 2*pi;
195  auto payloadIndex = 2*pi+1;
196  assert(payloadIndex < parts.Size());
197  auto relayed = relayer.relay(std::move(parts.At(headerIndex)),
198  std::move(parts.At(payloadIndex)));
199  if (relayed == DataRelayer::WillNotRelay) {
200  reportError("Unable to relay part.");
201  return;
202  }
203  LOG(DEBUG) << "Relaying part idx: " << headerIndex;
204  }
205  };
206 
207  // Second part. This is the actual outer loop we want to obtain, with
208  // implementation details which can be read. Notice how most of the state
209  // is actually hidden. For example we do not expose what "input" is. This
210  // will allow us to keep the same toplevel logic even if the actual meaning
211  // of input is changed (for example we might move away from multipart
212  // messages). Notice also that we need to act diffently depending on the
213  // actual CompletionOp we want to perform. In particular forwarding inputs
214  // also gets rid of them from the cache.
215  if (isValidInput() == false) {
216  reportError("Parts should come in couples. Dropping it.");
217  return true;
218  }
219  putIncomingMessageIntoCache();
220  return true;
221 }

223 bool DataProcessingDevice::tryDispatchComputation()
224 {
225  // This is the actual hidden state for the outer loop. In case we decide we
226  // want to support multithreaded dispatching of operations, I can simply
227  // move these to some thread local store and the rest of the lambdas
228  // should work just fine.
229  std::vector<DataRelayer::RecordAction> completed;
230  std::vector<std::unique_ptr<FairMQMessage>> currentSetOfInputs;
231 
232  auto& allocator = mAllocator;
233  auto& context = mFairMQContext;
234  auto& device = *this;
235  auto& errorCallback = mError;
236  auto& errorCount = mErrorCount;
237  auto& forwards = mSpec.forwards;
238  auto& inputsSchema = mSpec.inputs;
239  auto& processingCount = mProcessingCount;
240  auto& rdfContext = mDataFrameContext;
241  auto& relayer = mRelayer;
242  auto& rootContext = mRootContext;
243  auto& serviceRegistry = mServiceRegistry;
244  auto& statefulProcess = mStatefulProcess;
245  auto& statelessProcess = mStatelessProcess;
246  auto& stringContext = mStringContext;
247  auto& timingInfo = mTimingInfo;
248  auto& timesliceIndex = mServiceRegistry.get<TimesliceIndex>();
249 
250  // These duplicate references are created so that each function
251  // does not need to know about the whole class state, but I can
252  // fine grain control what is exposed at each state.
253  // FIXME: I should use a different id for this state.
254  auto& monitoringService = mServiceRegistry.get<Monitoring>();
255  monitoringService.send({ 1, "dpl/in_handle_data" });
256  ScopedExit metricFlusher([&monitoringService] {
257  monitoringService.send({ 1, "dpl/in_handle_data"});
258  monitoringService.send({ 0, "dpl/in_handle_data"});
259  monitoringService.flushBuffer(); });
260 
261  auto reportError = [&device](const char* message) {
262  device.error(message);
263  };
264 
265  // For the moment we have a simple "immediately dispatch" policy for stuff
266  // in the cache. This could be controlled from the outside e.g. by waiting
267  // for a few sets of inputs to arrive before we actually dispatch the
268  // computation, however this can be defined at a later stage.
269  auto canDispatchSomeComputation = [&completed, &relayer]() -> bool {
270  completed = relayer.getReadyToProcess();
271  return completed.empty() == false;
272  };
273 
274  // We use this to get a list with the actual indexes in the cache which
275  // indicate a complete set of inputs. Notice how I fill the completed
276  // vector and return it, so that I can have a nice for loop iteration later
277  // on.
278  auto getReadyActions = [&relayer, &completed, &monitoringService]() -> std::vector<DataRelayer::RecordAction> {
279  LOG(DEBUG) << "Getting parts to process";
280  int pendingInputs = (int)relayer.getParallelTimeslices() - completed.size();
281  monitoringService.send({ pendingInputs, "inputs/relayed/pending" });
282  if (completed.empty()) {
283  monitoringService.send({ 1, "inputs/relayed/incomplete" });
284  }
285  return completed;
286  };
287 
288  // This is needed to convert from a pair of pointers to an actual DataRef
289  // and to make sure the ownership is moved from the cache in the relayer to
290  // the execution.
291  auto fillInputs = [&relayer, &inputsSchema, &currentSetOfInputs](TimesliceSlot slot) -> InputRecord {
292  currentSetOfInputs = std::move(relayer.getInputsForTimeslice(slot));
293  InputSpan span{ [&currentSetOfInputs](size_t i) -> char const* {
294  return currentSetOfInputs.at(i) ? static_cast<char const*>(currentSetOfInputs.at(i)->GetData()) : nullptr;
295  },
296  currentSetOfInputs.size() };
297  return InputRecord{ inputsSchema, std::move(span) };
298  };
299 
300  // This is the thing which does the actual computation. No particular reason
301  // why we do the stateful processing before the stateless one.
302  // PROCESSING:{START,END} is done so that we can trigger on begin / end of processing
303  // in the GUI.
304  auto dispatchProcessing = [&processingCount, &allocator, &statefulProcess, &statelessProcess, &monitoringService,
305  &context, &rootContext, &stringContext, &rdfContext, &serviceRegistry, &device](TimesliceSlot slot, InputRecord& record) {
306  if (statefulProcess) {
307  LOG(DEBUG) << "PROCESSING:START:" << slot.index;
308  monitoringService.send({ processingCount++, "dpl/stateful_process_count" });
309  ProcessingContext processContext{record, serviceRegistry, allocator};
310  monitoringService.send({ 2, "dpl/in_handle_data" });
311  statefulProcess(processContext);
312  monitoringService.send({ 1, "dpl/in_handle_data" });
313  LOG(DEBUG) << "PROCESSING:END:" << slot.index;
314  }
315  if (statelessProcess) {
316  LOG(DEBUG) << "PROCESSING:START:" << slot.index;
317  monitoringService.send({ processingCount++, "dpl/stateless_process_count" });
318  ProcessingContext processContext{record, serviceRegistry, allocator};
319  monitoringService.send({ 2, "dpl/in_handle_data" });
320  statelessProcess(processContext);
321  monitoringService.send({ 1, "dpl/in_handle_data" });
322  LOG(DEBUG) << "PROCESSING:END:" << slot.index;
323  }
324 
325  DataProcessor::doSend(device, context);
326  DataProcessor::doSend(device, rootContext);
327  DataProcessor::doSend(device, stringContext);
328  DataProcessor::doSend(device, rdfContext);
329  };
330 
331  // Error handling means printing the error and updating the metric
332  auto errorHandling = [&errorCallback, &monitoringService, &serviceRegistry](std::exception& e, InputRecord& record) {
333  monitoringService.send({ 3, "dpl/in_handle_data" });
334  LOG(ERROR) << "Exception caught: " << e.what() << std::endl;
335  if (errorCallback) {
336  monitoringService.send({ 1, "error" });
337  ErrorContext errorContext{record, serviceRegistry, e};
338  errorCallback(errorContext);
339  }
340  monitoringService.send({ 1, "dpl/in_handle_data" });
341  };
342 
343  // I need a preparation step which gets the current timeslice id and
344  // propagates it to the various contextes (i.e. the actual entities which
345  // create messages) because the messages need to have the timeslice id into
346  // it.
347  auto prepareAllocatorForCurrentTimeSlice = [&timingInfo, &rootContext, &stringContext, &context, &relayer, &timesliceIndex](TimesliceSlot i) {
348  auto timeslice = timesliceIndex.getTimesliceForSlot(i);
349  LOG(DEBUG) << "Timeslice for cacheline is " << timeslice.value;
350  timingInfo.timeslice = timeslice.value;
351  rootContext.clear();
352  context.clear();
353  stringContext.clear();
354  };

356  // This is how we do the forwarding, i.e. we push
357  // the inputs which are shared between this device and others
358  // to the next one in the daisy chain.
359  // FIXME: do it in a smarter way than O(N^2)
360  auto forwardInputs = [&reportError, &forwards, &device, &currentSetOfInputs](TimesliceSlot slot, InputRecord& record) {
361  assert(record.size()*2 == currentSetOfInputs.size());
362  LOG(DEBUG) << "FORWARDING:START:" << slot.index;
363  for (size_t ii = 0, ie = record.size(); ii < ie; ++ii) {
364  DataRef input = record.getByPos(ii);
365 
366  // If is now possible that the record is not complete when
367  // we forward it, because of a custom completion policy.
368  // this means that we need to skip the empty entries in the
369  // record for being forwarded.
370  if (input.header == nullptr || input.payload == nullptr) {
371  continue;
372  }
373  auto dh = o2::header::get<DataHeader*>(input.header);
374  if (!dh) {
375  reportError("Header is not a DataHeader?");
376  continue;
377  }
378  auto dph = o2::header::get<DataProcessingHeader*>(input.header);
379  if (!dph) {
380  reportError("Header stack does not contain DataProcessingHeader");
381  continue;
382  }
383 
384  auto &header = currentSetOfInputs[ii*2];
385  auto &payload = currentSetOfInputs[ii*2+1];
386 
387  for (auto forward : forwards) {
388  LOG(DEBUG) << "Input part content";
389  LOG(DEBUG) << dh->dataOrigin.str;
390  LOG(DEBUG) << dh->dataDescription.str;
391  LOG(DEBUG) << dh->subSpecification;
392  if (DataSpecUtils::match(forward.matcher, dh->dataOrigin, dh->dataDescription, dh->subSpecification)
393  && (dph->startTime % forward.maxTimeslices) == forward.timeslice) {
394 
395  if (header.get() == nullptr) {
396  LOG(ERROR) << "Missing header!";
397  continue;
398  }
399  auto fdph = o2::header::get<DataProcessingHeader*>(header.get()->GetData());
400  if (fdph == nullptr) {
401  LOG(ERROR) << "Forwarded data does not have a DataProcessingHeader";
402  continue;
403  }
404  auto fdh = o2::header::get<DataHeader*>(header.get()->GetData());
405  if (fdh == nullptr) {
406  LOG(ERROR) << "Forwarded data does not have a DataHeader";
407  continue;
408  }
409  LOG(DEBUG) << "Forwarding data to " << forward.channel;
410  LOG(DEBUG) << "Forwarded timeslice is " << fdph->startTime;
411  LOG(DEBUG) << "Forwarded channel is " << forward.channel;
412  FairMQParts forwardedParts;
413  forwardedParts.AddPart(std::move(header));
414  forwardedParts.AddPart(std::move(payload));
415  assert(forwardedParts.Size() == 2);
416  assert(o2::header::get<DataProcessingHeader*>(forwardedParts.At(0)->GetData()));
417  LOG(DEBUG) << o2::header::get<DataProcessingHeader*>(forwardedParts.At(0)->GetData())->startTime;
418  LOG(DEBUG) << forwardedParts.At(0)->GetSize();
419  // FIXME: this should use a correct subchannel
420  device.Send(forwardedParts, forward.channel, 0);
421  }
422  }
423  }
424  LOG(DEBUG) << "FORWARDING:END";
425  };
426 
427  if (canDispatchSomeComputation() == false) {
428  return false;
429  }
430 
431  for (auto action: getReadyActions()) {
432  if (action.op == CompletionPolicy::CompletionOp::Wait) {
433  continue;
434  }
435 
436  prepareAllocatorForCurrentTimeSlice(TimesliceSlot{ action.slot });
437  InputRecord record = fillInputs(action.slot);
438  if (action.op == CompletionPolicy::CompletionOp::Discard) {
439  if (forwards.empty() == false) {
440  forwardInputs(action.slot, record);
441  continue;
442  }
443  }
444  try {
445  for (size_t ai = 0; ai != record.size(); ai++) {
446  auto cacheId = action.slot.index * record.size() + ai;
447  auto state = record.isValid(ai) ? 2 : 0;
448  monitoringService.send({ state, "data_relayer/" + std::to_string(cacheId) });
449  }
450  dispatchProcessing(action.slot, record);
451  for (size_t ai = 0; ai != record.size(); ai++) {
452  auto cacheId = action.slot.index * record.size() + ai;
453  auto state = record.isValid(ai) ? 3 : 0;
454  monitoringService.send({ state, "data_relayer/" + std::to_string(cacheId) });
455  }
456  } catch(std::exception &e) {
457  errorHandling(e, record);
458  }
459  // We forward inputs only when we consume them. If we simply Process them,
460  // we keep them for next message arriving.
461  if (action.op == CompletionPolicy::CompletionOp::Consume) {
462  if (forwards.empty() == false) {
463  forwardInputs(action.slot, record);
464  }
465  }
466  }
467 
468  return true;
469 }
