48  struct inputSpecEqual {

50  {
51  // -1 means 'match all subSpec'
52  if (a.subSpec == -1 || b.subSpec == -1) {
53  return a.description == b.description && a.origin == b.origin;
54  } else {
55  return a.description == b.description && a.origin == b.origin && a.subSpec == b.subSpec;
