102 // extract num TPC lanes, a lane is a streaming line of processors (digitizer-clusterizer-etc)
103 // by default this will be std::max(the number of physical cores, numberofsectors)
104 // as a temporary means to fully use a machine and as a way to play with different topologies
105 int getNumTPCLanes(std::vector<int> const& sectors, ConfigContext const& configcontext)
106 {
107  auto lanes = configcontext.options().get<int>("tpc-lanes");
108  if (lanes < 0) {
109  LOG(FATAL) << "tpc-lanes needs to be positive\n";
110  return 0;
111  }
112  // crosscheck with sectors
113  return std::min(lanes, (int)sectors.size());
114 }

116 void initTPC()
117 {
118  // We only want to do this for the DPL master
119  // I am not aware of an easy way to query if "I am DPL master" so
120  // using for the moment a mechanism defining/setting an environment variable
121  // with the parent ID and query inside forks if this environment variable exists
122  // (it assumes fundamentally that the master executes this function first)
123  std::stringstream streamthis;
124  std::stringstream streamparent;
125 
126  streamthis << "TPCGEMINIT_PID" << getpid();
127  streamparent << "TPCGEMINIT_PID" << getppid();
128  if (getenv(streamparent.str().c_str())) {
129  LOG(DEBUG) << "GEM ALREADY INITIALIZED ... SKIPPING HERE";
130  return;
131  } else {
132  LOG(DEBUG) << "INITIALIZING TPC GEMAmplification";
133  }
134  setenv(streamthis.str().c_str(), "ON", 1);
135 
136  auto& cdb = o2::TPC::CDBInterface::instance();
137  cdb.setUseDefaults();
138  // by invoking this constructor we make sure that a common file will be created
139  // in future we should take this from OCDB and just forward per message
140  const static auto& ampl = o2::TPC::GEMAmplification::instance();
141 }

143 void extractTPCSectors(std::vector<int>& sectors, ConfigContext const& configcontext)
144 {
145  auto sectorsstring = configcontext.options().get<std::string>("tpc-sectors");
146  if (sectorsstring.compare("all") != 0) {
147  // we expect them to be , separated
148  std::stringstream ss(sectorsstring);
149  std::vector<std::string> stringtokens;
150  while (ss.good()) {
151  std::string substr;
152  getline(ss, substr, ',');
153  stringtokens.push_back(substr);
154  }
155  // now try to convert each token to int
156  for (auto& token : stringtokens) {
157  try {
158  auto s = std::stoi(token);
159  sectors.emplace_back(s);
160  } catch (std::invalid_argument e) {
161  }
162  }
163  return;
164  }
165 
166  // all sectors otherwise by default
167  for (int s = 0; s < o2::TPC::Sector::MAXSECTOR; ++s) {
168  sectors.emplace_back(s);
169  }
170 }

194 WorkflowSpec defineDataProcessing(ConfigContext const& configcontext)
195 {
196  // reserve one entry which fill be filled with the SimReaderSpec at the end
197  // this places the processor at the beginning of the workflow in the upper left corner of the GUI
198  WorkflowSpec specs(1);
199 
200  // we will first of all read the GRP to detect which components need
201  // instantiations
202  // (for the moment this assumes the file o2sim_grp.root to be in the current directory)
203  const auto grp = readGRP();
204  if (!grp) {
205  return specs;
206  }
207 
208  // lambda to extract detectors which are enabled in the workflow
209  // will complain if user gave wrong input in construction of DetID
210  auto isEnabled = [&configcontext, grp](o2::detectors::DetID id) {
211  auto extractIgnored = [&configcontext]() {
212  auto skipString = configcontext.options().get<std::string>("skipDet");
213  std::vector<o2::detectors::DetID> skipped;
214  if (skipString.compare("none") != 0) {
215  // we expect them to be , separated
216  std::stringstream ss(skipString);
217  std::vector<std::string> stringtokens;
218  while (ss.good()) {
219  std::string substr;
220  getline(ss, substr, ',');
221  stringtokens.push_back(substr);
222  }
223 
224  // now try to convert each token to o2::detectors::DetID
225  for (auto& token : stringtokens) {
226  skipped.emplace_back(token.c_str());
227  }
228  }
229  return skipped;
230  };
231 
232  static auto skipped = extractIgnored();
233  bool is_skipped = std::find(skipped.begin(), skipped.end(), id) != skipped.end();
234  LOG(INFO) << id.getName() << " is skipped " << is_skipped;
235  bool is_ingrp = grp->isDetReadOut(id);
236  LOG(INFO) << id.getName() << " is in grp " << is_ingrp;
237  return !is_skipped && is_ingrp;
238  };
239 
240  std::vector<o2::detectors::DetID> detList; // list of participating detectors
241  int fanoutsize = 0;
242  if (wantCollisionTimePrinter()) {
243  specs.emplace_back(o2::steer::getCollisionTimePrinter(fanoutsize++));
244  }
245 
246  // the TPC part
247  // we need to init this anyway since TPC is treated a bit special (for the moment)
248  initTPC();
249  // keeps track of which subchannels correspond to tpc channels
250  auto tpclanes = std::make_shared<std::vector<int>>();
251  // keeps track of which tpc sectors to process
252  auto tpcsectors = std::make_shared<std::vector<int>>();
253  if (isEnabled(o2::detectors::DetID::TPC)) {
254 
255  extractTPCSectors(*tpcsectors.get(), configcontext);
256  auto lanes = getNumTPCLanes(*tpcsectors.get(), configcontext);
257  detList.emplace_back(o2::detectors::DetID::TPC);
258 
259  for (int l = 0; l < lanes; ++l) {
260  specs.emplace_back(o2::steer::getTPCDriftTimeDigitizer(fanoutsize));
261  tpclanes->emplace_back(fanoutsize); // this records that TPC is "listening under this subchannel"
262  fanoutsize++;
263  }
264 
265  auto tpcRecoOutputType = configcontext.options().get<std::string>("tpc-reco-type");
266  if (tpcRecoOutputType.empty()) {
267  // for writing digits to disc
268  specs.emplace_back(o2::TPC::getTPCDigitRootWriterSpec(lanes));
269  } else {
270  // attach the TPC reco workflow
271  auto tpcRecoWorkflow = o2::TPC::RecoWorkflow::getWorkflow(true, lanes, "digitizer", tpcRecoOutputType.c_str());
272  specs.insert(specs.end(), tpcRecoWorkflow.begin(), tpcRecoWorkflow.end());
273  }
274  }
275 
276  // the ITS part
277  if (isEnabled(o2::detectors::DetID::ITS)) {
278  detList.emplace_back(o2::detectors::DetID::ITS);
279  // connect the ITS digitization
280  specs.emplace_back(o2::ITSMFT::getITSDigitizerSpec(fanoutsize++));
281  // connect ITS digit writer
282  specs.emplace_back(o2::ITSMFT::getITSDigitWriterSpec());
283  }
284 
285  // the MFT part
286  if (isEnabled(o2::detectors::DetID::MFT)) {
287  detList.emplace_back(o2::detectors::DetID::MFT);
288  // connect the MFT digitization
289  specs.emplace_back(o2::ITSMFT::getMFTDigitizerSpec(fanoutsize++));
290  // connect MFT digit writer
291  specs.emplace_back(o2::ITSMFT::getMFTDigitWriterSpec());
292  }
293 
294  // the TOF part
295  if (isEnabled(o2::detectors::DetID::TOF)) {
296  detList.emplace_back(o2::detectors::DetID::TOF);
297  // connect the TOF digitization
298  specs.emplace_back(o2::tof::getTOFDigitizerSpec(fanoutsize++));
299  // add TOF digit writer
300  specs.emplace_back(o2::tof::getTOFDigitWriterSpec());
301  // add TOF clusterer
302  specs.emplace_back(o2::tof::getTOFClusterizerSpec());
303  // add TOF cluster writer
304  specs.emplace_back(o2::tof::getTOFClusterWriterSpec());
305  }
306 
307  // the FIT part
308  if (isEnabled(o2::detectors::DetID::FIT)) {
309  detList.emplace_back(o2::detectors::DetID::FIT);
310  // connect the FIT digitization
311  specs.emplace_back(o2::fit::getFITDigitizerSpec(fanoutsize++));
312  // connect the FIT digit writer
313  specs.emplace_back(o2::fit::getFITDigitWriterSpec());
314  }
315 
316  // GRP updater: must come after all detectors since requires their list
317  specs.emplace_back(o2::parameters::getGRPUpdaterSpec(detList));
318 
319  // The SIM Reader. NEEDS TO BE LAST
320  specs[0] = o2::steer::getSimReaderSpec(fanoutsize, tpcsectors, tpclanes);
321 
322  return specs;
323 }
