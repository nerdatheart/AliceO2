112  template <typename ContextType>
113  void operator()(ContextType&& context)
114  {
115  if (!mTree || !mFile || mFile->IsZombie()) {
116  throw std::runtime_error("Writer is invalid state, probably closed previously");
117  }
118  mTreeStructure->exec(std::forward<ContextType>(context), mBranchSpecs);
119  mTree->Fill();
120  for (auto& spec : mBranchSpecs) {
121  if (!spec.releaseFct) {
122  continue;
123  }
124  spec.releaseFct(spec.data);
125  spec.data = nullptr;
126  spec.releaseFct = nullptr;
127  }
128  }

202  void setupInstance(std::vector<BranchSpec>& specs, TTree* tree)
203  {
204  static_cast<PrevT>(*this).setupInstance(specs, tree);
205  constexpr size_t Index = STAGE - 1;
206  specs[Index].branch = tree->Branch(specs[Index].name.c_str(), &mStore);
207  specs[Index].classinfo = TClass::GetClass(typeid(type));
208  LOG(INFO) << Index << ": branch " << specs[Index].name << " set up";
209  }
210 
211  // process previous stage and this stage
212  void process(InputContext& context, std::vector<BranchSpec>& specs)
213  {
214  static_cast<PrevT>(*this).process(context, specs);
215  constexpr size_t Index = STAGE - 1;
216  auto data = context.get<type*>(specs[Index].key.c_str());
217  // could either copy to the corresponding store variable or use the object
218  // directly. TBranch::SetAddress supports only non-const pointers, so this is
219  // a hack
220  // FIXME: get rid of the const_cast
221  // FIXME: using object directly results in a segfault in the Streamer when
222  // using std::vector<o2::test::Polymorphic> in test_RootTreeWriter.cxx
223  //specs[Index].branch->SetAddress(const_cast<type*>(data.get()));
224  // handling copy-or-move is also more complicated because the returned smart
225  // pointer wraps a const buffer which might reside in the input queue and
226  // thus can not be moved.
227  //copyOrMove(mStore, (type&)*data);
228  mStore = *data;
229  // the data object is a smart pointer and we have to keep the data alieve
230  // until the tree is actually filled after the recursive processing of inputs
231  // release the base pointer from the smart pointer instance and keep the
232  // daleter to be called after tree fill.
233  // Note: the deleter might not be existent if the content was moved before
234  specs[Index].data = const_cast<type*>(data.release());
235  auto deleter = data.get_deleter();
236  specs[Index].releaseFct = [deleter](void* buffer) { deleter(reinterpret_cast<type*>(buffer)); };
237  }
