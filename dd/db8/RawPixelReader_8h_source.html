98  void convertDigits2Raw(std::string outName, std::string inpName, std::string digTreeName, std::string digBranchName,
99  UShort_t moduleMin = 0, UShort_t moduleMax = 0xffff,
100  Long64_t evFirst = 0, Long64_t evLast = -1)
101  {
102  TStopwatch swTot;
103 
104  mModuleMin = moduleMin;
105  mModuleMax = moduleMax < mMapping.getNModules() ? moduleMax : mMapping.getNModules() - 1;
106 
107  LOG(INFO) << "Converting ALPIDE digits to raw data for modules " << mModuleMin << " : "
108  << mModuleMax << FairLogger::endl;
109 
110  std::unique_ptr<TTree> inpTree = o2::utils::RootChain::load(digTreeName, inpName);
111  if (!inpTree) {
112  LOG(FATAL) << "Failed to find the tree " << digTreeName << " in the input " << inpName << FairLogger::endl;
113  }
114  std::vector<o2::ITSMFT::Digit>* digiVec = nullptr;
115 
116  inpTree->SetBranchAddress(digBranchName.data(), &digiVec);
117  if (!digiVec) {
118  LOG(FATAL) << "Failed to load digits branch " << digBranchName << " from the tree " << digTreeName
119  << " of the input " << inpName << FairLogger::endl;
120  }
121  if (outName.empty()) {
122  outName = "raw" + digBranchName + ".raw";
123  LOG(INFO) << "Output file name is not provided, set to " << outName << FairLogger::endl;
124  }
125 
126  if (evLast < 1 || inpTree->GetEntries() < evLast) {
127  evLast = inpTree->GetEntries() - 1;
128  }
129 
130  mCoder.openOutput(outName);
131  o2::ITSMFT::ChipPixelData chipData;
132  chipData.setROFrame(0);
133  chipData.setChipID(0);
134  int counterROF = 0, counterTot = 0, shiftROF = 0;
135  PrevChipDone prev; // previously processed data info
136 
137  for (auto ev = evFirst; ev <= evLast; ev++) {
138  inpTree->GetEntry(ev);
139  for (const auto& dig : *digiVec) {
140  int modID = mMapping.chipID2Module(dig.getChipIndex());
141  if (modID < mModuleMin || modID > mModuleMax) { // skip unwanted modules
142  continue;
143  }
144  UInt_t rof = dig.getROFrame() + shiftROF;
145  if (rof != chipData.getROFrame() || dig.getChipIndex() != chipData.getChipID()) {
146  if (rof != chipData.getROFrame()) { // new frame
147  counterTot += counterROF;
148  LOG(INFO) << "Converted " << counterROF << " digits in ROFrame " << chipData.getROFrame()
149  << " (" << counterTot << " in total)" << FairLogger::endl;
150  if (rof < chipData.getROFrame()) { // ROFs should be in the increasing order
151  // this situation is possible if we work with the chian of the digits, each element of the
152  // chain starting from 0. We will introduce a shift in the ROFs to avoid ups and downs
153  shiftROF = chipData.getROFrame() + 1;
154  LOG(WARNING) << "New ROF=" << dig.getROFrame() << " is inferior to last ROF=" << chipData.getROFrame() << " following ROFs will be shifter by " << shiftROF << FairLogger::endl;
155  }
156  counterROF = 0;
157  }
158  convertChipDigits(&chipData, prev);
159  chipData.clear();
160  chipData.setROFrame(dig.getROFrame() + shiftROF);
161  chipData.setChipID(dig.getChipIndex());
162  }
163  o2::ITSMFT::PixelData pix(&dig);
164  chipData.getData().push_back(pix);
165  counterROF++;
166  }
167  //
168  mCoder.flushBuffer(); // flushing is expensive, don't do this too often
169  }
170 
171  if (chipData.getData().size()) {
172  convertChipDigits(&chipData, prev);
173  counterTot += counterROF;
174  LOG(INFO) << "Converted " << counterROF << " digits in ROFrame " << chipData.getROFrame()
175  << " (last ROF, total hits: " << counterTot << ")" << FairLogger::endl;
176  }
177  convertChipDigits(nullptr, prev); // to close the streem
178  mCoder.closeIO();
179  swTot.Stop();
180  swTot.Print();
181  delete digiVec;
182  }

255  void convertChipDigits(o2::ITSMFT::ChipPixelData* pixData, PrevChipDone& prev)
256  {
260 
261  int chipInMod = 0, rof = 0, chipID = 0, modID = 0, nChipsInMod = 0;
262  if (pixData) {
263  rof = pixData->getROFrame();
264  chipID = pixData->getChipID();
265  modID = mMapping.chipID2Module(chipID, chipInMod);
266  nChipsInMod = mMapping.getNChipsInModule(modID);
267  } else { // finalize last ROF
268  rof = prev.rof + 1; // fake ROF to close the previous
269  }
270 
271  if (prev.chipID != -1 && (rof != prev.rof || modID != prev.modID)) {
272  // new readout frame or chip started, finish the current one
273 
274  // if new module is started, finish the current one
275  if (modID != prev.modID || rof != prev.rof) {
276  int goToChip = mMapping.module2ChipID(prev.modID, prev.nChipsInMod); // maxChip_of_module + 1
277  for (++prev.chipID; prev.chipID < goToChip; prev.chipID++) {
278  addEmptyChip(prev.chipID, prev.rof);
279  }
280  addModuleTrailer(prev.modID, prev.rof);
281  prev.moduleClosed = true;
282  prev.chipInMod = -1;
283  }
284  if (rof > prev.rof) { // close current ROF filling empty modules
285  for (++prev.modID; prev.modID < mMapping.getNModules(); prev.modID++) {
286  addEmptyModule(prev.modID, prev.rof);
287  }
288  // fill empty ROFs till new ROF
289  for (++prev.rof; prev.rof < rof; prev.rof++) {
290  addEmptyROF(prev.rof);
291  }
292  prev.modID = -1;
293  }
294 
295  // close empty modules of current ROF up to the new module
296  for (++prev.modID; prev.modID < modID; prev.modID++) {
297  addEmptyModule(prev.modID, rof);
298  }
299 
300  // start new chip/module only if there is some data
301  if (pixData) {
302  if (prev.moduleClosed) { // open new module if there is an input
303  addModuleHeader(modID, rof);
304  prev.moduleClosed = false;
305  }
306  // fill empty chips of the module till current one
307  for (++prev.chipInMod; prev.chipInMod < chipInMod; prev.chipInMod++) {
308  prev.chipID = mMapping.module2ChipID(prev.modID, prev.chipInMod);
309  addEmptyChip(prev.chipID, rof);
310  }
311  }
312  }
313  //
314  if (pixData) {
315  if (prev.moduleClosed) {
316  addModuleHeader(modID, rof);
317  prev.moduleClosed = false;
318  }
319  addDigits(pixData);
320  }
321  // sync:
322  prev.chipID = chipID;
323  prev.modID = modID;
324  prev.rof = rof;
325  prev.chipInMod = chipInMod;
326  prev.nChipsInMod = nChipsInMod;
327  }

330  void addDigits(o2::ITSMFT::ChipPixelData* chipData)
331  {
332  // sort data in row/col
333  auto& pixels = chipData->getData();
334  std::sort(pixels.begin(), pixels.end(),
335  [](auto lhs, auto rhs) {
336  if (lhs.getRow() < rhs.getRow())
337  return true;
338  if (lhs.getRow() > rhs.getRow())
339  return false;
340  return lhs.getCol() < rhs.getCol();
341  });
342  int chipInMod;
343  int modID = mMapping.chipID2Module(chipData->getChipID(), chipInMod);
344  for (const auto& pix : pixels) {
345  mCoder.addPixel(pix.getRow(), pix.getCol());
346  }
347  static int cnt = 0;
348  mCoder.finishChipEncoding(chipInMod, chipData->getROFrame());
349  }

352  bool decodeAlpideHitsFormat(const std::vector<Coder::HitsRecord>& records, ChipPixelData& chipData)
353  {
354  chipData.setROFrame(mCurrROF);
355  chipData.setChipID(mMapping.module2ChipID(mCurrModule, mCurrChipInModule));
356  //
357  int nhitsChip = 0; // total hits counter
358  int nRightCHits = 0; // counter for the hits in the right column of the current double column
359  std::uint16_t rightColHits[AlpideCoder::NRows]; // buffer for the accumulation of hits in the right column
360  std::uint16_t colDPrev = 0xffff; // previously processed double column (to dected change of the double column)
361  int nrec = records.size();
362  for (int ih = 0; ih < nrec; ih++) {
363  const auto& hitr = records[ih];
364 
365  std::uint16_t row = hitr.address >> 1;
366  // abs id of left column in double column
367  std::uint16_t colD = (std::uint16_t(hitr.region) * Coder::NDColInReg + std::uint16_t(hitr.dcolumn)) << 1;
368 
369  // if we start new double column, transfer the hits accumulated in the right column buffer of prev. double column
370  if (colD != colDPrev) {
371  colDPrev++;
372  for (int ihr = 0; ihr < nRightCHits; ihr++) {
373  chipData.getData().emplace_back(rightColHits[ihr], colDPrev);
374  }
375  colDPrev = colD;
376  nRightCHits = 0; // reset the buffer
377  }
378 
379  bool rightC = ((row & 0x1) ? 1 - (hitr.address & 0x1) : (hitr.address & 0x1)); // true for right column / lalse for left
380  int col = colD + rightC;
381 #ifdef _RAW_READER_DEBUG_
382  printf("%04d/%03d ", col, row);
383 #endif
384  // we want to have hits sorted in column/row, so the hits in right column of given double column
385  // are first collected in the temporary buffer
386  if (rightC) {
387  rightColHits[nRightCHits++] = row;
388  } else {
389  chipData.getData().emplace_back(row, col); // left column hits are added directly to the container
390  }
391  nhitsChip++;
392  if (hitr.hitmap) { // extra hits
393 #ifdef _RAW_READER_DEBUG_
394  printf(" [ ");
395 #endif
396  for (int ip = 0; ip < Coder::HitMapSize; ip++) {
397  if (hitr.hitmap & (0x1 << ip)) {
398  int addr = hitr.address + ip + 1;
399  int rowE = addr >> 1;
400  rightC = ((rowE & 0x1) ? 1 - (addr & 0x1) : (addr & 0x1)); // true for right column / lalse for left
401  int colE = colD + rightC;
402  if (rightC) { // same as above
403  rightColHits[nRightCHits++] = rowE;
404  } else {
405  chipData.getData().emplace_back(rowE, colE); // left column hits are added directly to the container
406  }
407 #ifdef _RAW_READER_DEBUG_
408  printf("%04d/%03d ", colE, rowE);
409 #endif
410  nhitsChip++;
411  }
412  }
413 #ifdef _RAW_READER_DEBUG_
414  printf("]");
415 #endif
416  }
417 #ifdef _RAW_READER_DEBUG_
418  printf("\n");
419 #endif
420  }
421 
422  // transfer last right-column buffer
423  colDPrev++;
424  for (int ihr = 0; ihr < nRightCHits; ihr++) {
425  chipData.getData().emplace_back(rightColHits[ihr], colDPrev);
426  }
427 
428 #ifdef _RAW_READER_DEBUG_
429  printf("decoded %d hits in %d records of chip %d(%d/%d) at ROF %d\n", nhitsChip, nrec,
430  chipData.getChipID(), mCurrModule, mCurrChipInModule, mCurrROF);
431 #endif
432 
433  return true;
434  }
