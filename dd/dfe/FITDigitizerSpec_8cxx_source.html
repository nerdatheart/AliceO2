79  void run(framework::ProcessingContext& pc)
80  {
81  static bool finished = false;
82  if (finished) {
83  return;
84  }
85 
86  // read collision context from input
87  auto context = pc.inputs().get<o2::steer::RunContext*>("collisioncontext");
88  auto& timesview = context->getEventRecords();
89 
90  // if there is nothing to do ... return
91  if (timesview.size() == 0) {
92  return;
93  }
94 
95  TStopwatch timer;
96  timer.Start();
97 
98  LOG(INFO) << "CALLING FIT DIGITIZATION";
99 
100  static std::vector<o2::fit::HitType> hits;
101  o2::dataformats::MCTruthContainer<o2::MCCompLabel> labelAccum;
102  o2::dataformats::MCTruthContainer<o2::MCCompLabel> labels;
103  o2::fit::Digit digit;
104  std::vector<o2::fit::Digit> digitAccum; // digit accumulator
105 
106  auto& eventParts = context->getEventParts();
107  // loop over all composite collisions given from context
108  // (aka loop over all the interaction records)
109  for (int collID = 0; collID < timesview.size(); ++collID) {
110  mDigitizer.setEventTime(timesview[collID].timeNS);
111 
112  // for each collision, loop over the constituents event and source IDs
113  // (background signal merging is basically taking place here)
114  for (auto& part : eventParts[collID]) {
115  mDigitizer.setEventID(part.entryID);
116  // mDigitizer.setSrcID(part.sourceID);
117 
118  // get the hits for this event and this source
119  hits.clear();
120  retrieveHits(mSimChains, "FITHit", part.sourceID, part.entryID, &hits);
121  LOG(INFO) << "For collision " << collID << " eventID " << part.entryID << " found " << hits.size() << " hits ";
122 
123  // call actual digitization procedure
124  labels.clear();
125  // digits.clear();
126  mDigitizer.process(&hits, &digit);
127  digit.printStream(std::cout);
128  auto data = digit.getChDgData();
129  LOG(INFO) << "Have " << data.size() << " fired channels ";
130  // copy digits into accumulator
131  digitAccum.push_back(digit); // we should move it there actually
132  LOG(INFO) << "Have " << digitAccum.back().getChDgData().size() << " fired channels ";
133  // labelAccum.mergeAtBack(*labels);
134  // LOG(INFO) << "Have " << digits->size() << " digits ";
135  }
136  }
137  // if (mDigitizer.isContinuous()) {
138  // digits->clear();
139  // labels->clear();
140  // digitizer->flushOutputContainer(*digits.get());
141  // LOG(INFO) << "FLUSHING LEFTOVER STUFF " << digits->size();
142  // // copy digits into accumulator
143  // std::copy(digits->begin(), digits->end(), std::back_inserter(*digitsAccum.get()));
144  // labelAccum.mergeAtBack(*labels);
145  // }
146 
147  // LOG(INFO) << "Have " << labelAccum.getNElements() << " TOF labels ";
148  // here we have all digits and we can send them to consumer (aka snapshot it onto output)
149  pc.outputs().snapshot(Output{ "FIT", "DIGITS", 0, Lifetime::Timeframe }, digitAccum);
150  // pc.outputs().snapshot(Output{ "FIT", "DIGITSMCTR", 0, Lifetime::Timeframe }, labelAccum);
151  LOG(INFO) << "FIT: Sending ROMode= " << mROMode << " to GRPUpdater";
152  pc.outputs().snapshot(Output{ "FIT", "ROMode", 0, Lifetime::Timeframe }, mROMode);
153  timer.Stop();
154  LOG(INFO) << "Digitization took " << timer.CpuTime() << "s";
155 
156  // we should be only called once; tell DPL that this process is ready to exit
157  pc.services().get<ControlService>().readyToQuit(false);
158  finished = true;
159  }
