100 inline void HitProcessingManager::sampleCollisionConstituents()
101 {
102  auto getBackgroundRoundRobin = [this]() {
103  static int bgcounter = 0;
104  int numbg = mSimChains[0]->GetEntries();
105  if (bgcounter == numbg) {
106  bgcounter = 0;
107  }
108  return EventPart(0, bgcounter++);
109  };
110 
111  const int nsignalids = mSimChains.size() - 1;
112  auto getSignalRoundRobin = [this, nsignalids]() {
113  static int bgcounter = 0;
114  static int signalid = 0;
115  static std::vector<int> counter(nsignalids, 0);
116  if (signalid == nsignalids) {
117  signalid = 0;
118  }
119  const auto realsourceid = signalid + 1;
120  int numentries = mSimChains[realsourceid]->GetEntries();
121  if (counter[signalid] == numentries) {
122  counter[signalid] = 0;
123  }
124  EventPart e(realsourceid, counter[signalid]);
125  counter[signalid]++;
126  signalid++;
127  return e;
128  };
129 
130  // we fill mRunContext.mEventParts
131  auto& eventparts = mRunContext.getEventParts();
132  eventparts.clear();
133  eventparts.resize(mRunContext.getEventRecords().size());
134  for (int i = 0; i < mRunContext.getEventRecords().size(); ++i) {
135  eventparts[i].clear();
136  // push any number of constituents?
137  // for the moment just 2 : one background and one signal
138  eventparts[i].emplace_back(getBackgroundRoundRobin());
139  if (mSimChains.size() > 1) {
140  eventparts[i].emplace_back(getSignalRoundRobin());
141  }
142  }
143 
144  // push any number of constituents?
145  // for the moment just max 2 : one background and one signal
146  mRunContext.setMaxNumberParts(1);
147  if (mSimChains.size() > 1) {
148  mRunContext.setMaxNumberParts(2);
149  }
150 
151  mRunContext.printCollisionSummary();
152 }
