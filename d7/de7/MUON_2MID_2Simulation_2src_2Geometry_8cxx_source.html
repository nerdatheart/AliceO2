215 TGeoVolume* createChamber(int iChamber)
216 {
218 
219  auto chamber = new TGeoVolumeAssembly(Form("SC1%d", iChamber + 1));
220 
221  // create the 3 types of RPC
222  auto shortRPC = createRPC("short", iChamber);
223  auto longRPC = createRPC("long", iChamber);
224  auto cutRPC = createRPC("cut", iChamber);
225 
226  // positions
227  double x = 0., y = 0., z = Constants::sRPCZShift;
228 
229  // rotations
230  auto rotX = new TGeoRotation("rotX", 90., 0., 90., -90., 180., 0.);
231  auto rotY = new TGeoRotation("rotY", 90., 180., 90., 90., 180., 0.);
232  auto rotZ = new TGeoRotation("rotZ", 90., 180., 90., 270., 0., 0.);
233 
234  // place them on the chamber
235  for (int iRPC = 0; iRPC < Constants::sNRPCLines; iRPC++) {
236  x = (iRPC == 0) ? Constants::sRPCShortCenterPos : Constants::sRPCCenterPos;
237  x *= Constants::sScaleFactors[iChamber];
238  y = kRPCypos[iRPC] * Constants::sScaleFactors[iChamber];
239 
240  switch (iRPC) {
241  case 0: // short
242  chamber->AddNode(shortRPC, 0, new TGeoTranslation(x, y, z));
243  chamber->AddNode(shortRPC, 9, new TGeoCombiTrans(-x, -y, -z, rotY));
244  break;
245  case 1: // cut
246  chamber->AddNode(cutRPC, 1, new TGeoTranslation(x, y, z));
247  chamber->AddNode(cutRPC, 17, new TGeoCombiTrans(x, -y, z, rotX));
248  break;
249  case 8: // cut
250  chamber->AddNode(cutRPC, 8, new TGeoCombiTrans(-x, y, z, rotY));
251  chamber->AddNode(cutRPC, 10, new TGeoCombiTrans(-x, -y, z, rotZ));
252  break;
253  default: // long
254  if (iRPC >= 5)
255  x *= -1;
256  chamber->AddNode(longRPC, iRPC, new TGeoTranslation(x, y, z));
257  chamber->AddNode(longRPC, 2 * Constants::sNRPCLines - iRPC, new TGeoCombiTrans(x, -y, z, rotX));
258  break;
259  }
260 
261  z *= -1; // switch the z side for the next RPC placement
262  }
263 
264  return chamber;
265 }

284
