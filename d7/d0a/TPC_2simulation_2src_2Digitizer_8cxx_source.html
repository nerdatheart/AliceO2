88 void Digitizer::ProcessHitGroup(const HitGroup& inputgroup, const Sector& sector, const float eventTime,
89  const int eventID, const int sourceID)
90 {
91  const static Mapper& mapper = Mapper::instance();
92  const static ParameterDetector& detParam = ParameterDetector::defaultInstance();
93  const static ParameterElectronics& eleParam = ParameterElectronics::defaultInstance();
94 
95  static GEMAmplification& gemAmplification = GEMAmplification::instance();
96  gemAmplification.updateParameters();
97  static ElectronTransport& electronTransport = ElectronTransport::instance();
98  electronTransport.updateParameters();
99  static SAMPAProcessing& sampaProcessing = SAMPAProcessing::instance();
100  sampaProcessing.updateParameters();
101 
102  const int nShapedPoints = eleParam.getNShapedPoints();
103  static std::vector<float> signalArray;
104  signalArray.resize(nShapedPoints);
105 
106  const int MCTrackID = inputgroup.GetTrackID();
107  for (size_t hitindex = 0; hitindex < inputgroup.getSize(); ++hitindex) {
108  const auto& eh = inputgroup.getHit(hitindex);
109 
110  GlobalPosition3D posEle(eh.GetX(), eh.GetY(), eh.GetZ());
111 
112  // Distort the electron position in case space-charge distortions are used
113  if (mUseSCDistortions) {
114  mSpaceChargeHandler->distortElectron(posEle);
115  }
116 
119  if (electronTransport.isCompletelyOutOfSectorCoarseElectronDrift(posEle, sector)) {
120  continue;
121  }
122 
124  const int nPrimaryElectrons = static_cast<int>(eh.GetEnergyLoss());
125 
127 
129  for (int iEle = 0; iEle < nPrimaryElectrons; ++iEle) {
130 
132  float driftTime = 0.f;
133  const GlobalPosition3D posEleDiff = electronTransport.getElectronDrift(posEle, driftTime);
134  const float absoluteTime = driftTime + eventTime + eh.GetTime() * 0.001;
135 
137  if (electronTransport.isElectronAttachment(driftTime)) {
138  continue;
139  }
140 
142  if (std::abs(posEleDiff.Z()) > detParam.getTPClength()) {
143  continue;
144  }
145 
147  const DigitPos digiPadPos = mapper.findDigitPosFromGlobalPosition(posEleDiff);
148  if (!digiPadPos.isValid()) {
149  continue;
150  }
151 
153  if (digiPadPos.getCRU().sector() != sector) {
154  continue;
155  }
156 
158  const int nElectronsGEM = gemAmplification.getStackAmplification(digiPadPos.getCRU(), digiPadPos.getPadPos());
159  if (nElectronsGEM == 0) {
160  continue;
161  }
162 
163  const GlobalPadNumber globalPad = mapper.globalPadNumber(digiPadPos.getGlobalPadPos());
164  const float ADCsignal = sampaProcessing.getADCvalue(static_cast<float>(nElectronsGEM));
165  sampaProcessing.getShapedSignal(ADCsignal, absoluteTime, signalArray);
166  for (float i = 0; i < nShapedPoints; ++i) {
167  const float time = absoluteTime + i * eleParam.getZBinWidth();
168  const MCCompLabel label(MCTrackID, eventID, sourceID);
169  mDigitContainer->addDigit(label, digiPadPos.getCRU(), sampaProcessing.getTimeBinFromTime(time), globalPad,
170  signalArray[i]);
171  }
172 
174  }
176  }
177 }
