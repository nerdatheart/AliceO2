33 TGeoVolume* createAirVacuumCave(const char* name)
34 {
35  // create the air medium (only used for the geometry test)
36  auto& mgr = o2::Base::MaterialManager::Instance();
37 
38  const int nAir = 4;
39  Float_t aAir[nAir] = { 12.0107, 14.0067, 15.9994, 39.948 };
40  Float_t zAir[nAir] = { 6., 7., 8., 18. };
41  Float_t wAir[nAir] = { 0.000124, 0.755267, 0.231781, 0.012827 };
42  Float_t dAirVacuum = 1.20479E-10;
43  const int kID = 90; // to avoid conflicts with definitions of other MCH materials
44 
45  mgr.Mixture("MCH", kID, "Air", aAir, zAir, dAirVacuum, nAir, wAir);
46  mgr.Medium("MCH", kID, "Air", kID,
47  false, /* isvol */
48  0, /* ifield */
49  -1.0, /* fieldm */
50  -1.0, /* tmaxfd */
51  -1.0, /* stemax */
52  -1.0, /* deemax */
53  -1.0, /* epsil */
54  -1.0 /* stmin */);
55  return gGeoManager->MakeBox(name, gGeoManager->GetMedium("MCH_Air"), 2000.0, 2000.0, 3000.0);
56 }

58 void dump(std::ostream& out, const TGeoNode& n, int level, int maxdepth, std::string prefix)
59 {
60  if (level >= maxdepth) {
61  return;
62  }
63 
64  if (level == 0) {
65  out << n.GetName() << "\n";
66  }
67 
68  if (level < maxdepth) {
69  for (int i = 0; i < n.GetNdaughters(); i++) {
70  TGeoNode* d = n.GetDaughter(i);
71  if (i == n.GetNdaughters() - 1) {
72  out << prefix + "└──" << d->GetName()
73  << "\n";
74  dump(out, *d, level + 1, maxdepth, prefix + " ");
75  } else {
76  out << prefix + "├──" << d->GetName()
77  << "\n";
78  dump(out, *d, level + 1, maxdepth, prefix + "│ ");
79  }
80  }
81  }

205 Vector3D<double> getNormalVector(const o2::Transform3D& t)
206 {
207  Point3D<double> px, py, po;
208  t.LocalToMaster(Point3D<double>{ 0, 1, 0 }, py);
209  t.LocalToMaster(Point3D<double>{ 1, 0, 0 }, px);
210  t.LocalToMaster(Point3D<double>{ 0, 0, 0 }, po);
211  Vector3D<double> a{ px - po };
212  Vector3D<double> b{ py - po };
213  return a.Cross(b).Unit();
214 }

216 TH2* getRadio(int detElemId, float xmin, float ymin, float xmax, float ymax, float xstep, float ystep, float thickness)
217 {
218  if (xmin >= xmax || ymin >= ymax) {
219  std::cerr << "incorrect limits\n";
220  return nullptr;
221  }
222  TH2* hmatb = new TH2F("hmatb", "hmatb", (int)((xmax - xmin) / xstep), xmin, xmax, (int)((ymax - ymin) / ystep), ymin, ymax);
223 
224  auto t = o2::mch::getTransformation(detElemId, *gGeoManager);
225 
226  auto normal = getNormalVector(t);
227 
228  for (auto x = xmin; x < xmax; x += xstep) {
229  for (auto y = ymin; y < ymax; y += ystep) {
230  auto matb = getMatBudget(t, normal, x, y, thickness);
231  if (std::isfinite(matb.meanX2X0)) {
232  hmatb->Fill(x, y, matb.meanX2X0);
233  }
234  }
235  }
236  return hmatb;
237 }
