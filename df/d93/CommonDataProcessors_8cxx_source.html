36 DataProcessorSpec CommonDataProcessors::getGlobalFileSink(std::vector<InputSpec> const& danglingOutputInputs,
37  std::vector<InputSpec>& unmatched)
38 {
39  auto writerFunction = [danglingOutputInputs](InitContext& ic) -> std::function<void(ProcessingContext&)> {
40  auto filename = ic.options().get<std::string>("outfile");
41  auto keepString = ic.options().get<std::string>("keep");
42 
43  if (filename.empty()) {
44  throw std::runtime_error("output file missing");
45  }
46 
47  bool hasOutputsToWrite = false;
48  auto [variables, outputMatcher] = DataDescriptorQueryBuilder::buildFromKeepConfig(keepString);
49  std::vector<ContextElement> context(variables.size());
50  for (auto& spec : danglingOutputInputs) {
51  if (outputMatcher->match(spec, context)) {
52  hasOutputsToWrite = true;
53  }
54  }
55  if (hasOutputsToWrite == false) {
56  return std::move([](ProcessingContext& pc) mutable -> void {
57  static bool once = false;
60  if (!once) {
61  LOG(INFO) << "No dangling output to be dumped.";
62  once = true;
63  }
64  sleep(1);
65  });
66  }
67  auto output = std::make_shared<std::ofstream>(filename.c_str(), std::ios_base::binary);
68  return std::move([ output, matcher = outputMatcher, contextSize = variables.size() ](ProcessingContext & pc) mutable->void {
69  std::vector<ContextElement> matchingContext(contextSize);
70  LOG(INFO) << "processing data set with " << pc.inputs().size() << " entries";
71  for (const auto& entry : pc.inputs()) {
72  LOG(INFO) << " " << *(entry.spec);
73  auto header = DataRefUtils::getHeader<header::DataHeader*>(entry);
74  auto dataProcessingHeader = DataRefUtils::getHeader<DataProcessingHeader*>(entry);
75  if (matcher->match(*header, matchingContext) == false) {
76  continue;
77  }
78  output->write(reinterpret_cast<char const*>(header), sizeof(header::DataHeader));
79  output->write(reinterpret_cast<char const*>(dataProcessingHeader), sizeof(DataProcessingHeader));
80  output->write(entry.payload, o2::framework::DataRefUtils::getPayloadSize(entry));
81  LOG(INFO) << "wrote data, size " << o2::framework::DataRefUtils::getPayloadSize(entry);
82  }
83  });
84  };
85 
86  std::vector<InputSpec> validBinaryInputs;
87  auto onlyTimeframe = [](InputSpec const& input) {
88  return input.lifetime == Lifetime::Timeframe;
89  };
90 
91  auto noTimeframe = [](InputSpec const& input) {
92  return input.lifetime != Lifetime::Timeframe;
93  };
94 
95  std::copy_if(danglingOutputInputs.begin(), danglingOutputInputs.end(),
96  std::back_inserter(validBinaryInputs), onlyTimeframe);
97  std::copy_if(danglingOutputInputs.begin(), danglingOutputInputs.end(),
98  std::back_inserter(unmatched), noTimeframe);
99 
100  DataProcessorSpec spec{
101  "dpl-global-binary-file-sink",
102  validBinaryInputs,
103  Outputs{},
104  AlgorithmSpec(writerFunction),
105  { { "outfile", VariantType::String, "dpl-out.bin", { "Name of the output file" } },
106  { "keep", VariantType::String, "", { "Comma separated list of ORIGIN/DESCRIPTION/SUBSPECIFICATION to save in outfile" } } }
107  };
108 
109  return spec;
110 }
