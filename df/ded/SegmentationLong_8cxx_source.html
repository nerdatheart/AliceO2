85 std::vector<Point> checkGaps(const Segmentation& seg, double xstep = 1.0, double ystep = 1.0)
86 {
87  std::vector<Point> gaps;
88  auto bbox = o2::mch::mapping::getBBox(seg);
89  auto env = o2::mch::mapping::getEnvelop(seg);
90 
91  if (env.size() != 1) {
92  throw std::runtime_error("assumption env contour = one polygon is not verified");
93  }
94 
95  for (double x = bbox.xmin() - xstep; x <= bbox.xmax() + xstep; x += xstep) {
96  for (double y = bbox.ymin() - ystep; y <= bbox.ymax() + ystep; y += ystep) {
97  double distanceToEnveloppe =
98  std::sqrt(o2::mch::contour::squaredDistancePointToPolygon(o2::mch::contour::Vertex<double>{ x, y }, env[0]));
99  bool withinEnveloppe = env.contains(x, y) && (distanceToEnveloppe > 1E-5);
100  if (withinEnveloppe && !seg.isValid(seg.findPadByPosition(x, y))) {
101  gaps.push_back(std::make_pair(x, y));
102  }
103  }
104  }
105  return gaps;
106 }

129 int testOnePosition(const o2::mch::mapping::Segmentation& seg, rapidjson::Value& tp)
130 {
131  double x = tp["x"].GetDouble();
132  double y = tp["y"].GetDouble();
133  int paduid = seg.findPadByPosition(x, y);
134 
135  bool isOutside{ false };
136 
137  if (tp.HasMember("isoutside")) {
138  isOutside = (tp["isoutside"].GetString() == std::string("true"));
139  }
140 
141  if (seg.isValid(paduid) && isOutside) {
142  std::cerr << "found a pad where I was not expecting one" << std::endl;
143  return 1;
144  }
145  if (!seg.isValid(paduid) && !isOutside) {
146  std::cerr << "did not find a pad where I was expecting one" << std::endl;
147  return 1;
148  }
149 
150  if (seg.isValid(paduid)) {
151  double px = seg.padPositionX(paduid);
152  double py = seg.padPositionY(paduid);
153 
154  double ex = tp["px"].GetDouble();
155  double ey = tp["py"].GetDouble();
156  if (!areEqual(px, ex) || !areEqual(py, ey)) {
157  std::cout << "got different positions here : got px,py=" << px << "," << py << " vs expected x,y=" << ex << ","
158  << ey << std::endl;
159  }
160  }
161  return 0;
162 }
