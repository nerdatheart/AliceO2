52 bool ROOTIOUtilities::openTFile()
53 {
54  auto it = mTFileModesNames.find(mTFileMode);
55 
56  if (mTFileOpened) {
57  // check whether changing the mode is successful, if not, close the TFile
58  if (mTFile->ReOpen(it->second) < 0) {
59  closeTFile();
60  }
61  }
62  // if no TFile was opened or if it was closed due to failed mode change, try to open it again from scratch
63  if (!mTFileOpened && !mFilepath.empty()) {
64  mTFile = new TFile(mFilepath.c_str(), it->second);
65  }
66  if (mTFile) {
67  mTFile->cd();
68  mTDirectory = mTFile->GetDirectory(0);
69  mTFileOpened = true;
70  }
71  return mTFileOpened;
72 }

98 bool ROOTIOUtilities::changeToTDirectory(const std::string& dirname)
99 {
100  // if file not yet opened
101  if (!mTFileOpened) {
102  openTFile();
103  }
104  // if it was already opened, check if we are already in the desired directory
105  else if (dirname.compare(mTDirectoryName) == 0) {
106  return true;
107  }
108  // now the file might be open
109  if (mTFileOpened) {
110  // update pointer to directory
111  mTDirectory = mTFile->GetDirectory(dirname.c_str());
112 
113  if (mTDirectory) {
114  mTFile->cd(dirname.c_str());
115  mTDirectoryName = dirname;
116  } else if (mTFileMode == ETFileMode::kRECREATE || mTFileMode == ETFileMode::kUPDATE) {
117  mTDirectory = mTFile->mkdir(dirname.c_str());
118  // this could also fail e.g. if a directory above the root directory was desired
119  if (mTDirectory) {
120  mTFile->cd(dirname.c_str());
121  mTDirectoryName = dirname;
122  }
123  }
124  }
125  resetKeyList();
126  return (mTDirectory != nullptr);
127 }

154 bool ROOTIOUtilities::changeToTTree(const std::string& treename)
155 {
156  // if file not yet opened
157  if (!mTFileOpened) {
158  // if it's not possible to open the desired file, just return
159  if (!openTFile()) {
160  return false;
161  }
162  }
163 
164  // If desired treename is the same as the opened tree, keep it open but reset counters etc.
165  else if (mTTreeOpened) {
166  if (treename.compare(mTTree->GetName()) == 0) {
167  resetTTreeConnection();
168  return true;
169  }
170  // if it's a different tree, close it entirely
171  closeTTree();
172  }
173 
174  // first, try to get existing TTree by name
175  mTTree = dynamic_cast<TTree*>(mTFile->Get(treename.c_str()));
176  // If it does not exist, try to create one depending on the file mode
177  if (!mTTree && (mTFileMode == ETFileMode::kRECREATE || mTFileMode == ETFileMode::kUPDATE)) {
178  mTTree = new TTree(treename.c_str(), treename.c_str());
179  }
180  // check, whether on of the former attempts was successful
181  if (mTTree) {
182  mTTreeOpened = true;
183  resetTTreeCounter();
184  }
185  return mTTreeOpened;
186 }

218 bool ROOTIOUtilities::fetchData(int event)
219 {
220  // is the tree even open?
221  if (!mTTreeOpened) {
222  return false;
223  }
224  // out of bounds?
225  if (event >= mTTreeEntries) {
226  return false;
227  }
228  int gotten = 0;
229  // specific entry required
230  if (event > -1) {
231  gotten = mTTree->GetEntry(event);
232  }
233  // just get the next one
234  else {
235  gotten = mTTree->GetEntry(mTTreeCounter++);
236  }
237  return (gotten > 0);
238 }
