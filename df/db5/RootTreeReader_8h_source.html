223  void addBranchSpec(KeyType key, const char* branchName)
224  {
225  // right now we allow the same key to appear for multiple branches
226  mBranchSpecs.emplace_back(key, std::make_unique<BranchSpec>(BranchSpec{ branchName }));
227  auto branch = mInput.GetBranch(mBranchSpecs.back().second->name.c_str());
228  if (branch) {
229  branch->SetAddress(&(mBranchSpecs.back().second->data));
230  mBranchSpecs.back().second->branch = branch;
231  mBranchSpecs.back().second->classinfo = TClass::GetClass(branch->GetClassName());
232  LOG(INFO) << "branch set up: " << branchName;
233  } else {
234  std::string msg("can not find branch ");
235  msg += branchName;
236  throw std::runtime_error(msg);
