77 inline void InteractionSampler::generateCollisionTimes(std::vector<o2::InteractionRecord>& dest)
78 {
79  // fill vector with interaction records

87 inline void InteractionSampler::nextCollidingBC()
88 {
89  // increment bunch ID till next colliding bunch
90  do {
91  if (++mBCCurrent > mBCMax) { // did we exhaust full orbit?
92  mBCCurrent = mBCMin;
93  if (++mOrbit >= o2::constants::lhc::MaxNOrbits) { // wrap orbit (should not happen in run3)
94  warnOrbitWrapped();
95  mOrbit = 0;
96  }
97  }
98  } while (!mBCFilling.testBC(mBCCurrent));
99 }

102 inline int InteractionSampler::genPoissonZT()
103 {
104  // generate 0-truncated poisson number
105  // https://en.wikipedia.org/wiki/Zero-truncated_Poisson_distribution
106  int k = 1;
107  double t = mMuBCZTRed, u = gRandom->Rndm(), s = t;
108  while (s < u) {
109  s += t *= mMuBC / (++k);
110  }
111  return k;
112 }
113 }
