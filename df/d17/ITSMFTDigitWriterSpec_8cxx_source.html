62  void run(framework::ProcessingContext& pc)
63  {
64  if (mFinished) {
65  return;
66  }
67  std::string detStr = mID.getName();
68  std::string detStrL = mID.getName();
69  std::transform(detStrL.begin(), detStrL.end(), detStrL.begin(), ::tolower);
70 
71  // retrieve the digits from the input
72  auto inDigits = pc.inputs().get<std::vector<o2::ITSMFT::Digit>>((detStr + "digits").c_str());
73  auto inROFs = pc.inputs().get<std::vector<o2::ITSMFT::ROFRecord>>((detStr + "digitsROF").c_str());
74  auto inMC2ROFs = pc.inputs().get<std::vector<o2::ITSMFT::MC2ROFRecord>>((detStr + "digitsMC2ROF").c_str());
75  auto inLabels = pc.inputs().get<MCCont*>((detStr + "digitsMCTR").c_str());
76  LOG(INFO) << "RECEIVED DIGITS SIZE " << inDigits.size();
77 
78  auto digitsP = &inDigits;
79  auto labelsRaw = inLabels.get();
80  // connect this to a particular branch
81 
82  auto brDig = getOrMakeBranch(*mOutTree.get(), (detStr + "Digit").c_str(), &digitsP);
83  auto brLbl = getOrMakeBranch(*mOutTree.get(), (detStr + "DigitMCTruth").c_str(), &labelsRaw);
84  mOutTree->Fill();
85 
86  mOutFile->cd();
87  mOutFile->WriteObjectAny(&inROFs, "std::vector<o2::ITSMFT::ROFRecord>", (detStr + "DigitROF").c_str());
88  mOutFile->WriteObjectAny(&inMC2ROFs, "std::vector<o2::ITSMFT::MC2ROFRecord>", (detStr + "DigitMC2ROF").c_str());
89  mOutTree->Write();
90  mOutTree.reset(); // delete the tree before closing the file
91  mOutFile->Close();
92  mFinished = true;
93  pc.services().get<ControlService>().readyToQuit(false);
94  }
95 
96  protected:
97  ITSMFTDPLDigitWriter() {}
98  template <typename T>
99  TBranch* getOrMakeBranch(TTree& tree, std::string brname, T* ptr)
100  {
101  if (auto br = tree.GetBranch(brname.c_str())) {
102  br->SetAddress(static_cast<void*>(ptr));
103  return br;
104  }
105  // otherwise make it
106  return tree.Branch(brname.c_str(), ptr);
107  }

153 DataProcessorSpec getITSDigitWriterSpec()
154 {
155  std::string detStr = o2::detectors::DetID::getName(ITSDPLDigitWriter::DETID);
156  std::string detStrL = detStr;
157  std::transform(detStrL.begin(), detStrL.end(), detStrL.begin(), ::tolower);
158  auto detOrig = ITSDPLDigitWriter::DETOR;
159 
160  std::vector<InputSpec> inputs;
161  inputs.emplace_back(InputSpec{ (detStr + "digits").c_str(), detOrig, "DIGITS", 0, Lifetime::Timeframe });
162  inputs.emplace_back(InputSpec{ (detStr + "digitsROF").c_str(), detOrig, "DIGITSROF", 0, Lifetime::Timeframe });
163  inputs.emplace_back(InputSpec{ (detStr + "digitsMC2ROF").c_str(), detOrig, "DIGITSMC2ROF", 0, Lifetime::Timeframe });
164  inputs.emplace_back(InputSpec{ (detStr + "digitsMCTR").c_str(), detOrig, "DIGITSMCTR", 0, Lifetime::Timeframe });
165 
166  return DataProcessorSpec{
167  (detStr + "DigitWriter").c_str(),
168  inputs,
169  {}, // no output
170  AlgorithmSpec(adaptFromTask<ITSDPLDigitWriter>()),
171  Options{
172  { (detStrL + "-digit-outfile").c_str(), VariantType::String, (detStrL + "digits.root").c_str(), { "Name of the input file" } },
