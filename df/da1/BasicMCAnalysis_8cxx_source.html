84 void BasicMCAnalysis::analyze(const std::vector<StepInfo>* const steps, const std::vector<MagCallInfo>* const magCalls)
85 {
86  // first of all, count events
87  histNEvents->Fill(0.5);
88  // to store the volume name
89  std::string volName = "";
90  // loop over magnetic field calls
91  for (const auto& call : *magCalls) {
92  if (call.stepid < 0) {
93  continue;
94  }
95  auto step = steps->operator[](call.stepid);
96  mAnalysisManager->getLookupVolName(step.volId, volName);
97  if (call.B < 0.01) {
98  histSmallMagFieldCallsPerVolPerEvent->Fill(volName.c_str(), 1.);
99  }
100  histMagFieldCallsPerVolPerEvent->Fill(volName.c_str(), 1.);
101  }
102 
103  // total number of steps in this event
104  int nSteps = 0;
105  // to store particle ID
106  int pdgId = 0;
107  // get the mean step size per event
108  float meanStepSizePerEvent = 0.;
109  // monitor number of different tracks
110  std::vector<int> tracks;
111  // step sizes per volume id in one event
112  std::vector<float> stepSizesPerVol;
113  // number of steps per volume ID in this event
114  std::vector<int> nStepsPerVol;
115  // steps sizes per PDG ID in one event, use a map here since PDG IDs can be large (~10^10)
116  // don't attempt to handle such a vector
117  std::unordered_map<int, float> stepSizesPerPDGMap;
118  // number of steps per PDG ID in this event
119  std::unordered_map<int, int> nStepsPerPDGMap;
120 
121  // loop over all steps in an event
122  for (const auto& step : *steps) {
123  // prepare for PDG ids and volume names
124  mAnalysisManager->getLookupPDG(step.trackID, pdgId);
125  mAnalysisManager->getLookupVolName(step.volId, volName);
126  // first increment the total number of steps over all events
127  nSteps++;
128 
129  // avoid double counting of tracks in an event, so check if track ID is already registered
130  if (std::find(tracks.begin(), tracks.end(), step.trackID) == tracks.end() && step.trackID > -1) {
131  // if not, there is a new track
132  tracks.push_back(step.trackID);
133  // increment track per PDG ID and treat PDGs as alphanumeric labels, so we can easily deflate later
134  histNTracksPerPDGPerEvent->Fill(std::to_string(pdgId).c_str(), 1.);
135  }
136  // summarize all volume Ids over all events to see, how many volumes were traversed in total
137  if (std::find(volIds.begin(), volIds.end(), step.volId) == volIds.end() && step.volId > -1) {
138  volIds.push_back(step.volId);
139  }
140 
141  // get the spatial coordinates of this step
142  histStepsXPerEvent->Fill(step.x);
143  histStepsYPerEvent->Fill(step.y);
144  histStepsZPerEvent->Fill(step.z);
145  histRZ->Fill(step.z, std::sqrt(step.x * step.x + step.y * step.y));
146  // extract the step length
147  float stepLength = 0.;
148  // be save and check whether there are e.g. NaNs (happened!)
149  if (std::isfinite(step.step) && step.step > 0.) {
150  stepLength = step.step;
151  }
152  // summarise all steps sizes
153  histStepSizesPerEvent->Fill(stepLength);
154  // summarise steps sizes per volume in this event
155  if (stepSizesPerVol.size() <= step.volId) {
156  stepSizesPerVol.resize(step.volId + 1, 0.);
157  nStepsPerVol.resize(step.volId + 1, 0);
158  }
159  stepSizesPerVol[step.volId] += stepLength;
160  nStepsPerVol[step.volId]++;
161  // summarise step sizes per PDG ID in this event
162  if (stepSizesPerPDGMap.find(pdgId) == stepSizesPerPDGMap.end()) {
163  stepSizesPerPDGMap[pdgId] = stepLength;
164  nStepsPerPDGMap[pdgId] = 1;
165  } else {
166  stepSizesPerPDGMap[pdgId] += stepLength;
167  nStepsPerPDGMap[pdgId]++;
168  }
169 
170  // secondaries
171  histNSecondariesPerEvent->Fill(0.5, step.nsecondaries);
172  histNSecondariesPerVolPerEvent->Fill(volName.c_str(), step.nsecondaries);
173  }
174  // add number of steps
175  histNSteps->Fill(0.5, nSteps);
176  histNSteps->SetEntries(nSteps);
177  histNStepsPerEvent->Fill(0.5, nSteps);
178  histNStepsPerEvent->SetEntries(nSteps);
179 
180  // add number of tracks
181  histNTracks->Fill(0.5, tracks.size());
182  histNTracks->SetEntries(tracks.size());
183  histNTracksPerEvent->Fill(0.5, tracks.size());
184  histNTracksPerEvent->SetEntries(tracks.size());
185  // update number of steps, number of steps per volume, mean step length and mean step length per volume
186  float meanStepSizes = 0.;
187  for (int i = 0; i < nStepsPerVol.size(); i++) {
188  // if no step was made with that volume id, continue
189  if (nStepsPerVol[i] < 1) {
190  continue;
191  }
192  mAnalysisManager->getLookupVolName(i, volName);
193  // number of steps per volume
194  histNStepsPerVolPerEvent->Fill(volName.c_str(), nStepsPerVol[i]);
195  meanStepSizes += stepSizesPerVol[i];
196  histMeanStepSizePerVolPerEvent->Fill(volName.c_str(), stepSizesPerVol[i] / float(nStepsPerVol[i]));
197  // check in how many events a certain volume was present
198  if (volPresent.find(volName) == volPresent.end()) {
199  volPresent.insert(std::pair<std::string, float>(volName, 1.));
200  } else {
201  volPresent[volName]++;
202  }
203  }
204  histNStepsPerVolPerEvent->SetEntries(nSteps);
205  histMeanStepSizePerEvent->Fill(0.5, meanStepSizes / float(nSteps));
206  // since the step size is the difference between 2 points in 3D space,
207  // the exact number of entries is nSteps-nTracks, however nTracks << nSteps is expected
208  histMeanStepSizePerEvent->SetEntries(nSteps);
209  histMeanStepSizePerVolPerEvent->SetEntries(nSteps);
210  // number of steps per PDG ID and mean step length per PDG ID
211  for (auto& sp : nStepsPerPDGMap) {
212  std::string pdgString(std::to_string(sp.first));
213  // number of steps per volume
214  histNStepsPerPDGPerEvent->Fill(pdgString.c_str(), sp.second);
215  histMeanStepSizePerPDGPerEvent->Fill(pdgString.c_str(), stepSizesPerPDGMap[sp.first] / float(sp.second));
216  // check in how many events a certain volume was present
217  if (pdgPresent.find(pdgString) == pdgPresent.end()) {
218  pdgPresent.insert(std::pair<std::string, float>(pdgString, 1.));
219  } else {
220  pdgPresent[pdgString]++;
221  }
222  }
223  histNStepsPerPDGPerEvent->SetEntries(nSteps);
224  histMeanStepSizePerPDGPerEvent->SetEntries(nSteps);
225 }

227 void BasicMCAnalysis::finalize()
228 {
229  // fill and update (e.g. scaling) some histograms
230  histNVols->Fill(0.5, volIds.size());
231  histNVols->SetEntries(volIds.size());
232 
233  // just normalise over all events
234  histNStepsPerEvent->Scale(1. / histNEvents->GetEntries());
235  histNTracksPerEvent->Scale(1. / histNEvents->GetEntries());
236  histNSecondariesPerEvent->Scale(1. / histNEvents->GetEntries());
237  histStepsXPerEvent->Scale(1. / histNEvents->GetEntries());
238  histStepsYPerEvent->Scale(1. / histNEvents->GetEntries());
239  histStepsZPerEvent->Scale(1. / histNEvents->GetEntries());
240  histStepSizesPerEvent->Scale(1. / histNEvents->GetEntries());
241  histMeanStepSizePerEvent->Scale(1. / histNEvents->GetEntries());
242  // normalize per volume to number of events where a certain volume was present
243  for (auto& vp : volPresent) {
244  vp.second = 1. / vp.second;
245  }
246  for (auto& pp : pdgPresent) {
247  pp.second = 1. / pp.second;
248  }
249  // scale histograms bin-wise
250  utilities::scalePerBin(histNStepsPerVolPerEvent, volPresent);
251  utilities::scalePerBin(histMagFieldCallsPerVolPerEvent, volPresent);
252  utilities::scalePerBin(histSmallMagFieldCallsPerVolPerEvent, volPresent);
253  utilities::scalePerBin(histNSecondariesPerVolPerEvent, volPresent);
254  utilities::scalePerBin(histMeanStepSizePerVolPerEvent, volPresent);
255  utilities::scalePerBin(histMeanStepSizePerPDGPerEvent, pdgPresent);
256  utilities::scalePerBin(histNStepsPerPDGPerEvent, pdgPresent);
257  utilities::scalePerBin(histNTracksPerPDGPerEvent, pdgPresent);
258 
259  int nVolBins = histNStepsPerVolPerEvent->GetNbinsX();
260  for (int i = 1; i < nVolBins + 1; i++) {
261  histRelNStepsPerVolPerEvent->Fill(histNStepsPerVolPerEvent->GetXaxis()->GetBinLabel(i), histNStepsPerVolPerEvent->GetBinContent(i));
262  }
263  histRelNStepsPerVolPerEvent->Scale(1. / double(histNStepsPerEvent->GetMaximum()));
264  histRelNStepsPerVolPerEvent->SetEntries(histNSteps->GetEntries());
265 
266  int nPDGBins = histNTracksPerPDGPerEvent->GetNbinsX();
267  for (int i = 1; i < nPDGBins + 1; i++) {
268  histRelNTracksPerPDGPerEvent->Fill(histNTracksPerPDGPerEvent->GetXaxis()->GetBinLabel(i), histNTracksPerPDGPerEvent->GetBinContent(i));
269  }
270  histRelNTracksPerPDGPerEvent->Scale(1. / double(histNTracksPerEvent->GetMaximum()));
271  histRelNTracksPerPDGPerEvent->SetEntries(histNTracks->GetEntries());
272 
273  nPDGBins = histNStepsPerPDGPerEvent->GetNbinsX();
274  for (int i = 1; i < nPDGBins + 1; i++) {
275  histRelNStepsPerPDGPerEvent->Fill(histNStepsPerPDGPerEvent->GetXaxis()->GetBinLabel(i), histNStepsPerPDGPerEvent->GetBinContent(i));
276  }
277  histRelNStepsPerPDGPerEvent->Scale(1. / double(histNStepsPerEvent->GetMaximum()));
278  histRelNStepsPerPDGPerEvent->SetEntries(histNSteps->GetEntries());
279 }
