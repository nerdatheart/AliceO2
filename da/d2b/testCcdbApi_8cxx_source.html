132 void countItems(const string& s, int& countObjects, int& countSubfolders)
133 {
134  countObjects = 0;
135  countSubfolders = 0;
136  std::stringstream ss(s);
137  std::string line;
138  bool subfolderMode = false;
139  while (std::getline(ss, line, '\n')) {
140  ltrim(line);
141  rtrim(line);
142  if (line.length() == 0) {
143  continue;
144  }
145 
146  if (line.find("subfolders") != std::string::npos) {
147  subfolderMode = true;
148  continue;
149  }
150 
151  if (subfolderMode) {
152  countSubfolders++;
153  if (line.find(']') != std::string::npos) {
154  break;
155  }
156  }
157 
158  if (line.find(R"("path")") == 0) {
159  countObjects++;
160  }
161  }
162 }
