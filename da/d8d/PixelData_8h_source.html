121  void maskFiredInSample(const ChipPixelData& sample)
122  {
124  const auto& pixelsS = sample.getData();
125  UInt_t nC = mPixels.size();
126  if (!nC) {
127  return;
128  }
129  UInt_t nS = pixelsS.size();
130  if (!nS) {
131  return;
132  }
133  UInt_t itC = 0, itS = 0;
134  while (itC < nC && itS < nS) {
135  auto& pix0 = mPixels[itC];
136  const auto& pixC = pixelsS[itS];
137  if (pix0 == pixC) { // same
138  pix0.setMask();
139  if (mFirstUnmasked == itC++) { // mFirstUnmasked should flag 1st unmasked pixel entry
140  mFirstUnmasked = itC;
141  }
142  itS++;
143  } else if (pix0 < pixC) {
144  itC++;
145  } else {
146  itS++;
147  }
148  }
149  }
