31 // use the tasking system of DPL
32 // just need to implement 2 special methods init + run (there is no need to inherit from anything)
33 class TOFDPLClustererTask
34 {
35  using MCLabelContainer = o2::dataformats::MCTruthContainer<o2::MCCompLabel>;
36 
37  public:
38  void init(framework::InitContext& ic)
39  {
40  // nothing special to be set up
41  }
42 
43  void run(framework::ProcessingContext& pc)
44  {
45  static bool finished = false;
46  if (finished) {
47  return;
48  }
49  // get digit data
50  auto digits = pc.inputs().get<std::vector<o2::tof::Digit>*>("tofdigits");
51  auto digitlabels = pc.inputs().get<o2::dataformats::MCTruthContainer<o2::MCCompLabel>*>("tofdigitlabels");
52  mReader.setDigitArray(digits.get());
53  mClusterer.setMCTruthContainer(&mClsLabels);
54 
55  // call actual clustering routine
56  mClustersArray.clear();
57  mClsLabels.clear();
58  mClusterer.process(mReader, mClustersArray, digitlabels.get());
59 
60  LOG(INFO) << "TOF CLUSTERER : TRANSFORMED " << digits->size()
61  << " DIGITS TO " << mClustersArray.size() << " CLUSTERS";
62 
63  // send clusters
64  pc.outputs().snapshot(Output{ "TOF", "CLUSTERS", 0, Lifetime::Timeframe }, mClustersArray);
65  // send labels
66  pc.outputs().snapshot(Output{ "TOF", "CLUSTERSMCTR", 0, Lifetime::Timeframe }, mClsLabels);
67 
68  // declare done
69  finished = true;
70  pc.services().get<ControlService>().readyToQuit(false);
71  }
72 
73  private:
74  DigitDataReader mReader;
75  Clusterer mClusterer;
76 
77  std::vector<Cluster> mClustersArray;
78  MCLabelContainer mClsLabels;
79 };
