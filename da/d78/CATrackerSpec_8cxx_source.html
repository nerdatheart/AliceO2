45  constexpr static size_t NSectors = o2::TPC::Sector::MAXSECTOR;
46  using ClusterGroupParser = o2::algorithm::ForwardParser<o2::TPC::ClusterGroupHeader>;
47  struct ProcessAttributes {
48  // the input comes in individual calls and we need to buffer until
49  // data set is complete, have to think about a DPL feature to take
50  // ownership of an input
51  std::array<std::vector<unsigned char>, NSectors> inputs;
52  std::array<std::vector<MCLabelContainer>, NSectors> mcInputs;
53  std::bitset<NSectors> validInputs = 0;
54  std::bitset<NSectors> validMcInputs = 0;
55  std::unique_ptr<ClusterGroupParser> parser;
56  std::unique_ptr<o2::TPC::TPCCATracking> tracker;
57  int verbosity = 1;

61  auto initFunction = [processMC, fanIn](InitContext& ic) {
62  auto options = ic.options().get<std::string>("tracker-options");
63 
64  auto processAttributes = std::make_shared<ProcessAttributes>();
65  {
66  processAttributes->nParallelInputs = fanIn;
67  auto& parser = processAttributes->parser;
68  auto& tracker = processAttributes->tracker;
69  parser = std::make_unique<ClusterGroupParser>();
70  tracker = std::make_unique<o2::TPC::TPCCATracking>();
71  if (tracker->initialize(options.c_str()) != 0) {
72  throw std::invalid_argument("TPCCATracking initialization failed");
73  }
74  processAttributes->validInputs.reset();
75  processAttributes->validMcInputs.reset();
76  }

78  auto processingFct = [processAttributes, processMC](ProcessingContext& pc) {
79  auto& parser = processAttributes->parser;
80  auto& tracker = processAttributes->tracker;
81  uint64_t activeSectors = 0;
82  auto& verbosity = processAttributes->verbosity;
83 
84  // FIXME cleanup almost duplicated code
85  auto& validMcInputs = processAttributes->validMcInputs;
86  auto& mcInputs = processAttributes->mcInputs;
87  if (processMC) {
88  // we can later extend this to multiple inputs
89  std::vector<std::string> inputLabels(processAttributes->nParallelInputs);
90  std::generate(inputLabels.begin(), inputLabels.end(), [counter = std::make_shared<int>(0)]() { return "mclblin" + std::to_string((*counter)++); });
91  for (auto& inputLabel : inputLabels) {
92  auto ref = pc.inputs().get(inputLabel);
93  auto const* sectorHeader = DataRefUtils::getHeader<o2::TPC::TPCSectorHeader*>(ref);
94  if (sectorHeader == nullptr) {
95  // FIXME: think about error policy
96  LOG(ERROR) << "sector header missing on header stack";
97  return;
98  }
99  const int& sector = sectorHeader->sector;
100  if (sector < 0) {
101  // FIXME: this we have to sort out once the steering is implemented
102  continue;
103  }
104  if (validMcInputs.test(sector)) {
105  // have already data for this sector, this should not happen in the current
106  // sequential implementation, for parallel path merged at the tracker stage
107  // multiple buffers need to be handled
108  throw std::runtime_error("can only have one data set per sector");
109  }
110  mcInputs[sector] = std::move(pc.inputs().get<std::vector<MCLabelContainer>>(inputLabel.c_str()));
111  validMcInputs.set(sector);
112  activeSectors |= sectorHeader->activeSectors;
113  if (verbosity > 1) {
114  LOG(INFO) << "received " << *(ref.spec) << " MC label containers"
115  << " for sector " << sector //
116  << std::endl //
117  << " mc input status: " << validMcInputs //
118  << std::endl //
119  << " active sectors: " << std::bitset<NSectors>(activeSectors); //
120  }
121  }
122  }
123 
124  std::vector<std::string> inputLabels(processAttributes->nParallelInputs);
125  std::generate(inputLabels.begin(), inputLabels.end(), [counter = std::make_shared<int>(0)]() { return "input" + std::to_string((*counter)++); });
126  auto& validInputs = processAttributes->validInputs;
127  auto& inputs = processAttributes->inputs;
128  for (auto& inputLabel : inputLabels) {
129  auto ref = pc.inputs().get(inputLabel);
130  auto payploadSize = DataRefUtils::getPayloadSize(ref);
131  auto const* sectorHeader = DataRefUtils::getHeader<o2::TPC::TPCSectorHeader*>(ref);
132  if (sectorHeader == nullptr) {
133  // FIXME: think about error policy
134  LOG(ERROR) << "sector header missing on header stack";
135  return;
136  }
137  const int& sector = sectorHeader->sector;
138  if (sector < 0) {
139  // FIXME: this we have to sort out once the steering is implemented
140  continue;
141  }
142  if (validInputs.test(sector)) {
143  // have already data for this sector, this should not happen in the current
144  // sequential implementation, for parallel path merged at the tracker stage
145  // multiple buffers need to be handled
146  throw std::runtime_error("can only have one data set per sector");
147  }
148  inputs[sector].resize(payploadSize);
149  std::copy(ref.payload, ref.payload + payploadSize, inputs[sector].begin());
150  validInputs.set(sector);
151  activeSectors |= sectorHeader->activeSectors;
152  if (verbosity > 1) {
153  LOG(INFO) << "received " << *(ref.spec) << ", size " << inputs[sector].size() //
154  << " for sector " << sector //
155  << std::endl //
156  << " input status: " << validInputs //
157  << std::endl //
158  << " active sectors: " << std::bitset<NSectors>(activeSectors); //
159  }
160  }
161 
162  if (activeSectors == 0 || (activeSectors & validInputs.to_ulong()) != activeSectors ||
163  (processMC && (activeSectors & validMcInputs.to_ulong()) != activeSectors)) {
164  // not all sectors available
165  // FIXME: do we need to send something
166  return;
167  }
168  assert(processMC == false || validMcInputs == validInputs);
169  if (verbosity > 0) {
170  LOG(INFO) << "running tracking for sectors " << validInputs;
171  }
172  ClusterNativeAccessFullTPC clusterIndex;
173  memset(&clusterIndex, 0, sizeof(clusterIndex));
174  for (size_t index = 0; index < NSectors; index++) {
175  if (!validInputs.test(index)) {
176  continue;
177  }
178  const auto& input = inputs[index];
179  auto mcIterator = mcInputs[index].begin();
180  parser->parse(&input.front(), input.size(),
181  [](const typename ClusterGroupParser::HeaderType& h) {
182  // check the header, but in this case there is no validity check
183  return true;
184  },
185  [](const typename ClusterGroupParser::HeaderType& h) {
186  // get the size of the frame including payload
187  // and header and trailer size, e.g. payload size
188  // from a header member
189  return h.nClusters * sizeof(ClusterNative) + ClusterGroupParser::totalOffset;
190  },
191  [&](typename ClusterGroupParser::FrameInfo& frame) {
192  int sector = frame.header->sector;
193  int padrow = frame.header->globalPadRow;
194  int nClusters = frame.header->nClusters;
195  LOG(DEBUG) << " sector " << std::setw(2) << std::setfill('0') << sector << " padrow "
196  << std::setw(3) << std::setfill('0') << padrow << " clusters " << std::setw(3)
197  << std::setfill(' ') << nClusters;
198  clusterIndex.clusters[sector][padrow] = reinterpret_cast<const ClusterNative*>(frame.payload);
199  clusterIndex.nClusters[sector][padrow] = nClusters;
200  assert(processMC == false || mcIterator != mcInputs[index].end());
201  if (processMC && mcIterator != mcInputs[index].end()) {
202  clusterIndex.clustersMCTruth[sector][padrow] = &(*mcIterator);
203  ++mcIterator;
204  }
205 
206  return true;
207  });
208  }
209 
210  std::vector<TrackTPC> tracks;
211  MCLabelContainer tracksMCTruth;
212  int retVal = tracker->runTracking(clusterIndex, &tracks, (processMC ? &tracksMCTruth : nullptr));
213  if (retVal != 0) {
214  // FIXME: error policy
215  LOG(ERROR) << "tracker returned error code " << retVal;
216  }
217  LOG(INFO) << "found " << tracks.size() << " track(s)";
218  pc.outputs().snapshot(OutputRef{ "output" }, tracks);
219  if (processMC) {
220  LOG(INFO) << "sending " << tracksMCTruth.getIndexedSize() << " track label(s)";
221  pc.outputs().snapshot(OutputRef{ "mclblout" }, tracksMCTruth);
222  }
223 
224  validInputs.reset();
225  if (processMC) {
226  validMcInputs.reset();
227  for (auto& mcInput : mcInputs) {
228  mcInput.clear();
229  }
230  }
231  };
232 
233  return processingFct;
234  };

236  auto createInputSpecs = [fanIn](bool makeMcInput) {
237  std::vector<InputSpec> inputSpecs;
238  for (size_t n = 0; n < fanIn; ++n) {
239  std::string label = "input" + std::to_string(n);
240  inputSpecs.emplace_back(InputSpec{ label, gDataOriginTPC, "CLUSTERNATIVE", n, Lifetime::Timeframe });
241 
242  if (makeMcInput) {
243  label = "mclblin" + std::to_string(n);
244  constexpr o2::header::DataDescription datadesc("CLNATIVEMCLBL");
245  inputSpecs.emplace_back(InputSpec{ label, gDataOriginTPC, datadesc, n, Lifetime::Timeframe });
246  }
247  }
248  return std::move(inputSpecs);
249  };
250 
251  auto createOutputSpecs = [](bool makeMcOutput) {
252  std::vector<OutputSpec> outputSpecs{
253  OutputSpec{ { "output" }, gDataOriginTPC, "TRACKS", 0, Lifetime::Timeframe },
254  };
255  if (makeMcOutput) {
256  OutputLabel label{ "mclblout" };
257  constexpr o2::header::DataDescription datadesc("TRACKMCLBL");
258  outputSpecs.emplace_back(label, gDataOriginTPC, datadesc, 0, Lifetime::Timeframe);
259  }
260  return std::move(outputSpecs);
261  };

263  return DataProcessorSpec{ "tpc-tracker", // process id
264  { createInputSpecs(processMC) },
265  { createOutputSpecs(processMC) },
