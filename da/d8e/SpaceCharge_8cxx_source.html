103 void SpaceCharge::allocateMemory()
104 {
105  for (auto i = 0; i < mNZSlices; ++i) {
106  mSpaceChargeDensityA[i].resize(mNPhiBins * mNRBins);
107  mSpaceChargeDensityC[i].resize(mNPhiBins * mNRBins);
108  }
109 
110  for (int iz = 0; iz < mNZSlices; ++iz)
111  mCoordZ[iz] = (iz + 1) * mLengthZSlice;
112  for (int iphi = 0; iphi < mNPhiBins; ++iphi)
113  mCoordPhi[iphi] = (iphi + 1) * mWidthPhiBin;
114  for (int ir = 0; ir < mNRBins; ++ir)
115  mCoordR[ir] = (ir + 1) * mLengthRBin;

120  mMatrixLocalIonDriftDzC = new TMatrixD*[mNPhiBins];
121  mMatrixLocalIonDriftDrphiC = new TMatrixD*[mNPhiBins];
122  mMatrixLocalIonDriftDrC = new TMatrixD*[mNPhiBins];
123  for (int iphi = 0; iphi < mNPhiBins; ++iphi) {
124  mMatrixLocalIonDriftDzA[iphi] = new TMatrixD(mNRBins, mNZSlices);
125  mMatrixLocalIonDriftDrphiA[iphi] = new TMatrixD(mNRBins, mNZSlices);
126  mMatrixLocalIonDriftDrA[iphi] = new TMatrixD(mNRBins, mNZSlices);
127  mMatrixLocalIonDriftDzC[iphi] = new TMatrixD(mNRBins, mNZSlices);
128  mMatrixLocalIonDriftDrphiC[iphi] = new TMatrixD(mNRBins, mNZSlices);
129  mMatrixLocalIonDriftDrC[iphi] = new TMatrixD(mNRBins, mNZSlices);
130  }
131  mLookUpLocalIonDriftA = std::make_unique<AliTPCLookUpTable3DInterpolatorD>(mNRBins, mMatrixLocalIonDriftDrA, mCoordR.data(), mNPhiBins, mMatrixLocalIonDriftDrphiA, mCoordPhi.data(), mNZSlices, mMatrixLocalIonDriftDzA, mCoordZ.data(), mInterpolationOrder);

136 {

138  float bzField = o2field->solenoidField(); // magnetic field in kGauss

153 void SpaceCharge::calculateLookupTables()
154 {
155  // Potential, E field and electron distortion and correction lookup tables
156  mLookUpTableCalculator.ForceInitSpaceCharge3DPoissonIntegralDz(mNRBins, mNZSlices, mNPhiBins, 300, 1e-8);
157 
158  // Lookup tables for local ion drift along E field
159  if (mSCDistortionType == SCDistortionType::SCDistortionsRealistic) {
160  for (int iphi = 0; iphi < mNPhiBins; ++iphi) {
161  float phi = mCoordPhi[iphi];
162  TMatrixD& matrixDzA = *mMatrixLocalIonDriftDzA[iphi];
163  TMatrixD& matrixDzC = *mMatrixLocalIonDriftDzC[iphi];
164  TMatrixD& matrixDrphiA = *mMatrixLocalIonDriftDrphiA[iphi];
165  TMatrixD& matrixDrphiC = *mMatrixLocalIonDriftDrphiC[iphi];
166  TMatrixD& matrixDrA = *mMatrixLocalIonDriftDrA[iphi];
167  TMatrixD& matrixDrC = *mMatrixLocalIonDriftDrC[iphi];
168  int roc = o2::utils::Angle2Sector(phi);
169  for (int ir = 0; ir < mNRBins; ++ir) {
170  float radius = mCoordR[ir];
171  for (int iz = 0; iz < mNZSlices; ++iz) {
172  // A side
173  float z = mCoordZ[iz];
174  float x0[3] = { radius, phi, z };
175  float x1[3] = { radius, phi, z - mLengthZSlice };
176  double eVector0[3] = { 0.f, 0.f, 0.f };
177  double eVector1[3] = { 0.f, 0.f, 0.f };
178  mLookUpTableCalculator.GetElectricFieldCyl(x0, roc, eVector0);
179  mLookUpTableCalculator.GetElectricFieldCyl(x1, roc, eVector1);
180  matrixDzA(ir, iz) = DvDEoverv0 * (mLengthZSlice / 2.f) * (eVector0[2] + eVector1[2]);
181  matrixDrphiA(ir, iz) = (mLengthZSlice / 2.f) * (eVector0[1] + eVector1[1]) / sEzField;
182  matrixDrA(ir, iz) = (mLengthZSlice / 2.f) * (eVector0[0] + eVector1[0]) / sEzField;
183  // C side
184  x0[2] *= -1;
185  x1[2] *= -1;
186  mLookUpTableCalculator.GetElectricFieldCyl(x0, roc + 18, eVector0);
187  mLookUpTableCalculator.GetElectricFieldCyl(x1, roc + 18, eVector1);
188  matrixDzC(ir, iz) = DvDEoverv0 * (mLengthZSlice / 2.f) * (eVector0[2] + eVector1[2]);
189  matrixDrphiC(ir, iz) = (mLengthZSlice / 2.f) * (eVector0[1] + eVector1[1]) / sEzField;
190  matrixDrC(ir, iz) = (mLengthZSlice / 2.f) * (eVector0[0] + eVector1[0]) / sEzField;
191  }
192  }
193  }
194  mLookUpLocalIonDriftA->CopyFromMatricesToInterpolator();
195  mLookUpLocalIonDriftC->CopyFromMatricesToInterpolator();
196 
198  }
199 
200  mInitLookUpTables = true;
201 }

203 void SpaceCharge::updateLookupTables(float eventTime)
204 {
205  if (mTimeInit < 0.) {
206  mTimeInit = eventTime; // set the time of first initialization
207  }
208  if (TMath::Abs(eventTime - mTimeInit) < mLengthTimeSlice) {
209  return; // update only after one time bin has passed
210  }
211 
212  std::unique_ptr<std::unique_ptr<TMatrixD>[]> spaceChargeA = std::make_unique<std::unique_ptr<TMatrixD>[]>(mNPhiBins);
213  std::unique_ptr<std::unique_ptr<TMatrixD>[]> spaceChargeC = std::make_unique<std::unique_ptr<TMatrixD>[]>(mNPhiBins);
214  for (int iphi = 0; iphi < mNPhiBins; ++iphi) {
215  spaceChargeA[iphi] = std::make_unique<TMatrixD>(mNRBins, mNZSlices);
216  spaceChargeC[iphi] = std::make_unique<TMatrixD>(mNRBins, mNZSlices);
217  }
218  for (int iside = 0; iside < 2; ++iside) {
219  for (int iphi = 0; iphi < mNPhiBins; ++iphi) {
220  TMatrixD& chargeDensity = iside == 0 ? *spaceChargeA[iphi] : *spaceChargeC[iphi];
221  for (int ir = 0; ir < mNRBins; ++ir) {
222  for (int iz = 0; iz < mNZSlices; ++iz) {
223  if (iside == 0) {
224  chargeDensity(ir, iz) = mSpaceChargeDensityA[iz][ir + iphi * mNRBins];
225  } else {
226  chargeDensity(ir, iz) = mSpaceChargeDensityC[iz][ir + iphi * mNRBins];
227  }
228  }
229  }
230  }
231  }
232  mLookUpTableCalculator.SetInputSpaceChargeA((TMatrixD**)spaceChargeA.get());
233  mLookUpTableCalculator.SetInputSpaceChargeC((TMatrixD**)spaceChargeC.get());
234  calculateLookupTables();
235 }

295 void SpaceCharge::correctElectron(GlobalPosition3D& point)
296 {
297  if (!mInitLookUpTables) {
298  return;
299  }
300  const float x[3] = { point.X(), point.Y(), point.Z() };
301  float dx[3] = { 0.f, 0.f, 0.f };
302  float phi = TMath::ATan2(x[1], x[0]);
303  if (phi < 0) {
304  phi += TMath::TwoPi();
305  }
306  int roc = phi / TMath::Pi() * 9;
307  if (x[2] < 0) {
308  roc += 18;
309  }
310  mLookUpTableCalculator.GetCorrection(x, roc, dx);
311  point = GlobalPosition3D(x[0] + dx[0], x[1] + dx[1], x[2] + dx[2]);
312 }
