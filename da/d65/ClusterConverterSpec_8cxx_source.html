62  // FIXME check if there is functionality in the DPL to forward the stack

64  o2::header::Stack rawHeaderStack;
65  o2::header::Stack mcHeaderStack;
66  o2::TPC::TPCSectorHeader const* sectorHeaderMC = nullptr;

101  // clusters need to be sorted to write clusters of one CRU to raw pages
102  struct ClusterMapper {
103  std::vector<unsigned> clusterIds;
104  ClusterHardwareContainer metrics;
105  };
106  std::map<uint16_t, ClusterMapper> mapper;
107  for (unsigned clusterIndex = 0; clusterIndex < nClusters; clusterIndex++) {
108  // clusters are supposed to be sorted in CRU number, start a new entry
109  // for every new CRU, a map is needed if the clusters are not ordered
110  auto inputcluster = inClusters[clusterIndex];
111  uint16_t CRU = inputcluster.getCRU();
112  if (mapper.find(CRU) == mapper.end()) {
113  if (verbosity > 0) {
114  LOG(INFO) << "Inserting cluster set for CRU " << CRU;
115  }
116  mapper[CRU] = ClusterMapper{ {}, ClusterHardwareContainer{ 0, 0, 0, 0, 0, 0, 0, 0, 0xFFFFFFFF, 0, CRU } };
117  }
118  mapper[CRU].clusterIds.emplace_back(clusterIndex);
119  mapper[CRU].metrics.numberOfClusters++;
120  if (inputcluster.getTimeMean() < mapper[CRU].metrics.timeBinOffset) {
121  mapper[CRU].metrics.timeBinOffset = inputcluster.getTimeMean();
122  }
123  }
124 
125  ClusterHardwareContainer8kb clusterContainerMemory;
126  auto maxClustersPerContainer = clusterContainerMemory.getMaxNumberOfClusters();
127  unsigned nTotalPages = 0;
128  // at this point there is at least one cluster per record
129  for (const auto& m : mapper) {
130  auto nPages = (m.second.metrics.numberOfClusters - 1) / maxClustersPerContainer + 1;
131  if (verbosity > 0) {
132  LOG(INFO) << "CRU " << std::setw(3) << m.first << " " //
133  << std::setw(3) << m.second.metrics.numberOfClusters << " cluster(s)" //
134  << " in " << nPages << " page(s) of capacity " << maxClustersPerContainer; //
135  }
136  nTotalPages += nPages;
137  }
138  if (verbosity > 0) {
139  LOG(INFO) << "allocating " << nTotalPages << " output page(s), " << nTotalPages * sizeof(ClusterHardwareContainer8kb);
140  }
141  auto outputPages = pc.outputs().make<ClusterHardwareContainer8kb>(OutputRef{ "clusterout", fanSpec, std::move(rawHeaderStack) }, nTotalPages);
142 
143  auto outputPageIterator = outputPages.begin();
144  unsigned mcoutIndex = 0;
145  for (auto& cruClusters : mapper) {
146  LOG(DEBUG) << "processing CRU " << cruClusters.first;
147  auto clusterIndex = cruClusters.second.clusterIds.begin();
148  auto clusterIndexEnd = cruClusters.second.clusterIds.end();
149  while (clusterIndex != clusterIndexEnd && outputPageIterator != outputPages.end()) {
150  assert(cruClusters.second.metrics.numberOfClusters > 0);
151  ClusterHardwareContainer* clusterContainer = outputPageIterator->getContainer();
152  *clusterContainer = cruClusters.second.metrics;
153  if (clusterContainer->numberOfClusters > maxClustersPerContainer) {
154  clusterContainer->numberOfClusters = maxClustersPerContainer;
155  }
156 
157  // write clusters of the page
158  if (verbosity > 0) {
159  LOG(INFO) << "writing " << std::setw(3) << clusterContainer->numberOfClusters << " cluster(s) of CRU "
160  << std::setw(3) << clusterContainer->CRU;
161  }
162  for (unsigned int clusterInPage = 0;
163  clusterInPage < clusterContainer->numberOfClusters && clusterIndex != clusterIndexEnd;
164  clusterInPage++, ++clusterIndex) {
165  const auto& inputCluster = inClusters[*clusterIndex];
166  auto& outputCluster = clusterContainer->clusters[clusterInPage];
167  outputCluster.setCluster(inputCluster.getPadMean(),
168  inputCluster.getTimeMean() - clusterContainer->timeBinOffset,
169  inputCluster.getPadSigma() * inputCluster.getPadSigma(),
170  inputCluster.getTimeSigma() * inputCluster.getTimeSigma(), inputCluster.getQmax(),
171  inputCluster.getQ(), inputCluster.getRow(), 0);
172  if (mcin) {
173  // write the new sequence of MC labels
174  for (auto const& label : mcin->getLabels(*clusterIndex)) {
175  mcout.addElement(mcoutIndex, label);
176  }
177  ++mcoutIndex;
178  }
179  }
180  cruClusters.second.metrics.numberOfClusters -= clusterContainer->numberOfClusters;
181  ++outputPageIterator;
182  }
183  assert(cruClusters.second.metrics.numberOfClusters == 0);
184  }
185  if (mcin) {
186  if (verbosity > 0) {
187  LOG(INFO) << "writing MC labels for " << mcoutIndex << " cluster(s), index size " << mcout.getIndexedSize();
188  }
189  pc.outputs().snapshot(OutputRef{ "mclblout", fanSpec, std::move(mcHeaderStack) }, mcout);
190  }
191  };
192 
193  // return the actual processing function as a lambda function using variables
194  // of the init function
195  return processingFct;
196  };
