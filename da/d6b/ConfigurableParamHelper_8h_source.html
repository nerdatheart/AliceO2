63  // one of the key methods, using introspection to print itself
64  void printKeyValues(bool showprov) const final
65  {
66  // just a helper line to make sure P::sInstance is looked-up
67  // and that compiler complains about missing static sInstance of type P
68  // volatile void* ptr = (void*)&P::sInstance;
69  // static assert on type of sInstance:
70  static_assert(std::is_same<decltype(P::sInstance), P>::value, "static instance must of same type as class");
71 
72  // obtain the TClass for P and delegate further
73  auto cl = TClass::GetClass(typeid(P));
74  if (!cl) {
75  _ParamHelper::printWarning(typeid(P));
76  return;
77  }
78  _ParamHelper::printParametersImpl(getName(), cl, (void*)this, showprov ? sValueProvenanceMap : nullptr);
79  }

92  void initFrom(TFile* file) final
93  {
94  // switch off auto registering since the readback object is

96  setRegisterMode(false);
97  P* readback = nullptr;
98  file->GetObject(getName().c_str(), readback);

105  }
