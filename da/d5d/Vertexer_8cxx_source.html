42 Vertexer::Vertexer(const Event& event) : mEvent{ event }, mAverageClustersRadii{ std::array<float, 3>{ 0.f, 0.f, 0.f } }
43 {
44  for (int iLayer{ 0 }; iLayer < Constants::ITS::LayersNumberVertexer; ++iLayer) {
45  const Layer& currentLayer{ event.getLayer(iLayer) };
46  const int clustersNum{ currentLayer.getClustersSize() };
47  mClusters[iLayer].clear();
48  if (clustersNum > mClusters[iLayer].capacity()) {
49  mClusters[iLayer].reserve(clustersNum);
50  }
51  for (int iCluster{ 0 }; iCluster < clustersNum; ++iCluster) {
52  mClusters[iLayer].emplace_back(iLayer, currentLayer.getCluster(iCluster));
53  }
54  if (mClusters[iLayer].size() != 0) {
55  const float inverseNumberOfClusters{ 1.f / mClusters[iLayer].size() };
56  for (auto& cluster : mClusters[iLayer]) {
57  mAverageClustersRadii[iLayer] += cluster.rCoordinate;
58  }
59  mAverageClustersRadii[iLayer] *= inverseNumberOfClusters;
60  } else {
61  mAverageClustersRadii[iLayer] = LayersRCoordinate()[iLayer];
62  }
63  }
64  mDeltaRadii10 = mAverageClustersRadii[1] - mAverageClustersRadii[0];
65  mDeltaRadii21 = mAverageClustersRadii[2] - mAverageClustersRadii[1];
66  mMaxDirectorCosine3 =
67  LayersZCoordinate()[2] / std::sqrt(LayersZCoordinate()[2] * LayersZCoordinate()[2] +
68  (mDeltaRadii10 + mDeltaRadii21) * (mDeltaRadii10 + mDeltaRadii21));
69 }

73 void Vertexer::initialise(const float zCut, const float phiCut, const float pairCut, const float clusterCut,
74  const int clusterContributorsCut)
75 {
76  for (int iLayer{ 0 }; iLayer < Constants::ITS::LayersNumberVertexer; ++iLayer) {
77  std::sort(mClusters[iLayer].begin(), mClusters[iLayer].end(), [](Cluster& cluster1, Cluster& cluster2) {

81  for (int iLayer{ 0 }; iLayer < Constants::ITS::LayersNumberVertexer; ++iLayer) {
82  const int clustersNum = static_cast<int>(mClusters[iLayer].size());
83  int previousBinIndex{ 0 };
84  mIndexTables[iLayer][0] = 0;
85  for (int iCluster{ 0 }; iCluster < clustersNum; ++iCluster) {
86  const int currentBinIndex{ mClusters[iLayer][iCluster].indexTableBinIndex };

89  mIndexTables[iLayer][iBin] = iCluster;

91  previousBinIndex = currentBinIndex;
92  }
93  }

95  mIndexTables[iLayer][iBin] = clustersNum;

99  mZCut = zCut > Constants::ITS::LayersZCoordinate()[0] ? LayersZCoordinate()[0] : zCut;
100  mPhiCut = phiCut > TwoPi ? TwoPi : phiCut;

163 void Vertexer::findTracklets(const bool useMCLabel)
164 {
165  if (mVertexerInitialised) {
166  std::vector<std::pair<int, int>> clusters0, clusters2;
167  std::vector<bool> usedCluster2Flags, usedCluster0Flags;
168  usedCluster2Flags.resize(mClusters[2].size(), false);
169  usedCluster0Flags.resize(mClusters[0].size(), false);
170 
171  for (int iBin1{ 0 }; iBin1 < ZBins * PhiBins; ++iBin1) {
172 
173  int ZBinLow0{ std::max(0, ZBins - static_cast<int>(std::ceil((ZBins - iBin1 % ZBins + 1) *
174  (mDeltaRadii21 + mDeltaRadii10) / mDeltaRadii10))) };
175  int ZBinHigh0{ std::min(
176  static_cast<int>(std::ceil((iBin1 % ZBins + 1) * (mDeltaRadii21 + mDeltaRadii10) / mDeltaRadii21)),
177  ZBins - 1) };
178  // int ZBinLow0 { 0 };
179  // int ZBinHigh0 { ZBins -1 };
180 
181  int PhiBin1{ static_cast<int>(iBin1 / ZBins) };
182  clusters0 = selectClusters(
183  mIndexTables[0],
184  std::array<int, 4>{ ZBinLow0, (PhiBin1 - mPhiSpan < 0) ? PhiBins + (PhiBin1 - mPhiSpan) : PhiBin1 - mPhiSpan,
185  ZBinHigh0,
186  (PhiBin1 + mPhiSpan > PhiBins) ? PhiBin1 + mPhiSpan - PhiBins : PhiBin1 + mPhiSpan });
187 
188  for (int iCluster1{ mIndexTables[1][iBin1] }; iCluster1 < mIndexTables[1][iBin1 + 1]; ++iCluster1) {
189  bool trackFound = false;
190  for (int iRow0{ 0 }; iRow0 < clusters0.size(); ++iRow0) {
191  for (int iCluster0{ std::get<0>(clusters0[iRow0]) };
192  iCluster0 < std::get<0>(clusters0[iRow0]) + std::get<1>(clusters0[iRow0]); ++iCluster0) {
193  if (usedCluster0Flags[iCluster0])
194  continue;
195  if (std::abs(mClusters[0][iCluster0].phiCoordinate - mClusters[1][iCluster1].phiCoordinate) < mPhiCut ||
196  std::abs(mClusters[0][iCluster0].phiCoordinate - mClusters[1][iCluster1].phiCoordinate) >
197  TwoPi - mPhiCut) {
198  float ZProjection{ mClusters[0][iCluster0].zCoordinate +
199  (mAverageClustersRadii[2] - mClusters[0][iCluster0].rCoordinate) *
200  (mClusters[1][iCluster1].zCoordinate - mClusters[0][iCluster0].zCoordinate) /
201  (mClusters[1][iCluster1].rCoordinate - mClusters[0][iCluster0].rCoordinate) };
202  float ZProjectionInner{ mClusters[0][iCluster0].zCoordinate +
203  (mClusters[0][iCluster0].rCoordinate) *
204  (mClusters[1][iCluster1].zCoordinate - mClusters[0][iCluster0].zCoordinate) /
205  (mClusters[1][iCluster1].rCoordinate - mClusters[0][iCluster0].rCoordinate) };
206  if (std::abs(ZProjection) > (LayersZCoordinate()[0] + mZCut))
207  continue;
208  int ZProjectionBin{ (ZProjection < -LayersZCoordinate()[0])
209  ? 0
210  : (ZProjection > LayersZCoordinate()[0]) ? ZBins - 1
211  : getZBinIndex(2, ZProjection) };
212  int ZBinLow2{ (ZProjectionBin - mZSpan < 0) ? 0 : ZProjectionBin - mZSpan };
213  int ZBinHigh2{ (ZProjectionBin + mZSpan > ZBins - 1) ? ZBins - 1 : ZProjectionBin + mZSpan };
214  // int ZBinLow2 { 0 };
215  // int ZBinHigh2 { ZBins - 1 };
216  int PhiBinLow2{ (PhiBin1 - mPhiSpan < 0) ? PhiBins + PhiBin1 - mPhiSpan : PhiBin1 - mPhiSpan };
217  int PhiBinHigh2{ (PhiBin1 + mPhiSpan > PhiBins - 1) ? PhiBin1 + mPhiSpan - PhiBins : PhiBin1 + mPhiSpan };
218  // int PhiBinLow2{ 0 };
219  // int PhiBinHigh2{ PhiBins - 1 };
220  clusters2 =
221  selectClusters(mIndexTables[2], std::array<int, 4>{ ZBinLow2, PhiBinLow2, ZBinHigh2, PhiBinHigh2 });
222  for (int iRow2{ 0 }; iRow2 < clusters2.size(); ++iRow2) {
223  for (int iCluster2{ std::get<0>(clusters2[iRow2]) };
224  iCluster2 < std::get<0>(clusters2[iRow2]) + std::get<1>(clusters2[iRow2]); ++iCluster2) {
225  if (usedCluster2Flags[iCluster2])
226  continue;
227  float ZProjectionRefined{
228  mClusters[0][iCluster0].zCoordinate +
229  (mClusters[2][iCluster2].rCoordinate - mClusters[0][iCluster0].rCoordinate) *
230  (mClusters[1][iCluster1].zCoordinate - mClusters[0][iCluster0].zCoordinate) /
231  (mClusters[1][iCluster1].rCoordinate - mClusters[0][iCluster0].rCoordinate)
232  };
233  bool testMC{ !useMCLabel ||
234  (mEvent.getLayer(0).getClusterLabel(mClusters[0][iCluster0].clusterId).getTrackID() ==
235  mEvent.getLayer(2).getClusterLabel(mClusters[2][iCluster2].clusterId).getTrackID() &&
236  mEvent.getLayer(0).getClusterLabel(mClusters[0][iCluster0].clusterId).getTrackID() ==
237  mEvent.getLayer(1).getClusterLabel(mClusters[1][iCluster1].clusterId).getTrackID()) };
238  float absDeltaPhi{ std::abs(mClusters[2][iCluster2].phiCoordinate -
239  mClusters[1][iCluster1].phiCoordinate) };
240  float absDeltaZ{ std::abs(mClusters[2][iCluster2].zCoordinate - ZProjectionRefined) };
241  if (absDeltaZ < mZCut && ((absDeltaPhi < mPhiCut || std::abs(absDeltaPhi - TwoPi) < mPhiCut) && testMC)) {
242  mTracklets.emplace_back(Line{
243  std::array<float, 3>{ mClusters[0][iCluster0].xCoordinate, mClusters[0][iCluster0].yCoordinate,
244  mClusters[0][iCluster0].zCoordinate },
245  std::array<float, 3>{ mClusters[1][iCluster1].xCoordinate, mClusters[1][iCluster1].yCoordinate,
246  mClusters[1][iCluster1].zCoordinate } });
247  if (std::abs(mTracklets.back().cosinesDirector[2]) < mMaxDirectorCosine3) {
248  usedCluster0Flags[iCluster0] = true;
249  usedCluster2Flags[iCluster2] = true;
250  trackFound = true;
251  break;
252  } else {
253  mTracklets.pop_back();
254  }
255  }
256  }
257  if (trackFound)
258  break;
259  }
260  }
261  if (trackFound)
262  break;
263  }
264  if (trackFound)
265  break;
266  }
267  }
268  }
269  mTrackletsFound = true;
270  }

273 void Vertexer::findVertices()
274 {
275  if (mTrackletsFound) {
276  const int numTracklets{ static_cast<int>(mTracklets.size()) };
277  std::vector<bool> usedTracklets{};
278  usedTracklets.resize(mTracklets.size(), false);
279  for (int tracklet1{ 0 }; tracklet1 < numTracklets; ++tracklet1) {
280  if (usedTracklets[tracklet1])
281  continue;
282  for (int tracklet2{ tracklet1 + 1 }; tracklet2 < numTracklets; ++tracklet2) {
283  if (usedTracklets[tracklet2])
284  continue;
285  if (Line::getDCA(mTracklets[tracklet1], mTracklets[tracklet2]) <= mPairCut) {
286  mTrackletClusters.emplace_back(tracklet1, mTracklets[tracklet1], tracklet2, mTracklets[tracklet2]);
287  std::array<float, 3> tmpVertex{ mTrackletClusters.back().getVertex() };
288  if (tmpVertex[0] * tmpVertex[0] + tmpVertex[1] * tmpVertex[1] > 4.f) {
289  mTrackletClusters.pop_back();
290  break;
291  }
292  usedTracklets[tracklet1] = true;
293  usedTracklets[tracklet2] = true;
294  for (int tracklet3{ 0 }; tracklet3 < numTracklets; ++tracklet3) {
295  if (usedTracklets[tracklet3])
296  continue;
297  if (Line::getDistanceFromPoint(mTracklets[tracklet3], tmpVertex) < mPairCut) {
298  mTrackletClusters.back().add(tracklet3, mTracklets[tracklet3]);
299  usedTracklets[tracklet3] = true;
300  tmpVertex = mTrackletClusters.back().getVertex();
301  }
302  }
303  break;
304  }
305  }
306  }
307  std::sort(mTrackletClusters.begin(), mTrackletClusters.end(),
308  [](ClusterLines& cluster1, ClusterLines& cluster2) { return cluster1.getSize() > cluster2.getSize(); });
309  int noClusters{ static_cast<int>(mTrackletClusters.size()) };
310  for (int iCluster1{ 0 }; iCluster1 < noClusters; ++iCluster1) {
311  std::array<float, 3> vertex1{ mTrackletClusters[iCluster1].getVertex() };
312  std::array<float, 3> vertex2{};
313  for (int iCluster2{ iCluster1 + 1 }; iCluster2 < noClusters; ++iCluster2) {
314  vertex2 = mTrackletClusters[iCluster2].getVertex();
315  if (std::abs(vertex1[2] - vertex2[2]) < mClusterCut) {
316 
317  float distance{ (vertex1[0] - vertex2[0]) * (vertex1[0] - vertex2[0]) +
318  (vertex1[1] - vertex2[1]) * (vertex1[1] - vertex2[1]) +
319  (vertex1[2] - vertex2[2]) * (vertex1[2] - vertex2[2]) };
320  if (distance <= mPairCut * mPairCut) {
321  for (auto label : mTrackletClusters[iCluster2].getLabels()) {
322  mTrackletClusters[iCluster1].add(label, mTracklets[label]);
323  vertex1 = mTrackletClusters[iCluster1].getVertex();
324  }
325  }
326  mTrackletClusters.erase(mTrackletClusters.begin() + iCluster2);
327  --iCluster2;
328  --noClusters;
329  }
330  }
331  }
332  for (int iCluster{ 0 }; iCluster < noClusters; ++iCluster) {
333  if (mTrackletClusters[iCluster].getSize() < mClusterContributorsCut && noClusters > 1) {
334  mTrackletClusters.erase(mTrackletClusters.begin() + iCluster);
335  noClusters--;
336  continue;
337  }
338  float dist{ 0. };
339  for (auto& line : mTrackletClusters[iCluster].mLines) {
340  dist += Line::getDistanceFromPoint(line, mTrackletClusters[iCluster].getVertex()) /
341  mTrackletClusters[iCluster].getSize();
342  }
343  if (mTrackletClusters[iCluster].getVertex()[0] * mTrackletClusters[iCluster].getVertex()[0] +
344  mTrackletClusters[iCluster].getVertex()[1] * mTrackletClusters[iCluster].getVertex()[1] <
345  1.98 * 1.98) {
346  mVertices.emplace_back(
347  Point3D<float>{ mTrackletClusters[iCluster].getVertex()[0], mTrackletClusters[iCluster].getVertex()[1],
348  mTrackletClusters[iCluster].getVertex()[2] },
349  mTrackletClusters[iCluster].getRMS2(), // Symm matrix. Diagonal: RMS2 components,
350  // off-diagonal: square mean of projections on planes.
351  mTrackletClusters[iCluster].getSize(), // Contributors
352  mTrackletClusters[iCluster].getAvgDistance2() // In place of chi2
353  );
354  mVertices.back().setTimeStamp(mROFrame);
355  mEvent.addPrimaryVertex(mVertices.back().getX(), mVertices.back().getY(), mVertices.back().getZ());
356  }
357  }
358  }
