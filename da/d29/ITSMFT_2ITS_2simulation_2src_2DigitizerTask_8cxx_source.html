99 void DigitizerTask::FinishTask()
100 {
101  // finalize digitization, if needed, flash remaining digits
102 
103  if (!mDigitizer.getParams().isContinuous()) {
104  return;
105  }
106 
107  // is there QED backgroung provided? Fill QED slots up to the end of reserved ROFrames
108  if (mQEDBranch) {
109  processQEDBackground(mDigitizer.getEndTimeOfROFMax());
110  }
111 
112  FairRootManager* mgr = FairRootManager::Instance();
113  mgr->SetLastFill(kTRUE);
114  mDigitsArray.clear();
115  mMCTruthArray.clear();
116  mDigitizer.fillOutputContainer();

120 void DigitizerTask::processQEDBackground(double tMax)
121 {
122  // process QED time-slots until provided collision time (in ns)
123 
124  double tQEDNext = mLastQEDTimeNS + mQEDEntryTimeBinNS;
125 
126  while (tQEDNext < tMax) {
127  mLastQEDTimeNS = tQEDNext; // time used for current QED slot
128  tQEDNext += mQEDEntryTimeBinNS; // prepare time for next QED slot
129  if (++mLastQEDEntry >= mQEDBranch->GetEntries()) {
130  mLastQEDEntry = 0; // wrapp if needed
131  }
132  mQEDBranch->GetEntry(mLastQEDEntry);
133  mDigitizer.setEventTime(mLastQEDTimeNS);
134  mDigitizer.process(mHitsArrayQED, mLastQEDEntry, mQEDSourceID);
135  //
136  }
137 }

140 void DigitizerTask::setQEDInput(TBranch* qed, float timebin, UChar_t srcID)
141 {
142  // assign the branch containing hits from QED electrons, whith every entry integrating
143  // timebin ns of collisions
144 
145  LOG(INFO) << "Attaching QED ITS hits as sourceID=" << int(srcID) << ", entry integrates "

147 
148  mQEDBranch = qed;
149  mQEDEntryTimeBinNS = timebin;
150  if (mQEDBranch) {
151  assert(mQEDEntryTimeBinNS >= 1.0);
152  mLastQEDTimeNS = -mQEDEntryTimeBinNS / 2; // time will be assigned to the middle of the bin
153  mQEDBranch->SetAddress(&mHitsArrayQED);
154  mLastQEDEntry = -1;
155  mQEDSourceID = srcID;
156  assert(mHitsArrayQED);
157  assert(srcID < o2::MCCompLabel::maxSourceID());
158  }
