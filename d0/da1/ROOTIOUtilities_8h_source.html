132  {
133  if (mTTreeOpened) {
134  // check whether branch is available
135  if (mTTree->GetBranch(branchname.c_str())) {
136  mTTree->SetBranchAddress(branchname.c_str(), address);
137  return true;
138  }
139  // if update or recreate, the branch might not be there yet, create it
140  else if (mTFileMode == ETFileMode::kRECREATE || mTFileMode == ETFileMode::kUPDATE) {
141  mTTree->Branch(branchname.c_str(), address);
142  return true;
143  }
144  }
145  return false;
