133 inline float SAMPAProcessing::makeSignal(float ADCcounts, const PadSecPos& padSecPos,
134  float& pedestal, float& noise) const
135 {
136  static Baseline baseline;
137  float signal = ADCcounts;
139  // pedestal = baseline.getPedestal(padSecPos);
140  noise = baseline.getNoise(padSecPos);
141  switch (mEleParam->getDigitizationMode()) {
142  case DigitzationMode::FullMode: {
143  signal += noise;
144  signal += pedestal;
145  return getADCSaturation(signal);
146  break;
147  }
148  case DigitzationMode::SubtractPedestal: {
149  signal += noise;
150  signal += pedestal;
151  float signalSubtractPedestal = getADCSaturation(signal) - pedestal;
152  return signalSubtractPedestal;
153  break;
154  }
155  case DigitzationMode::NoSaturation: {
156  signal += noise;
157  signal += pedestal;
158  return signal;
159  break;
160  }
161  case DigitzationMode::PropagateADC: {
162  return signal;
163  break;
164  }
165  }
166 }

168 inline float SAMPAProcessing::getADCSaturation(const float signal) const
169 {
171  const float saturatedSignal = mSaturationSpline->Eval(signal);
172  const float adcSaturation = mEleParam->getADCSaturation();
173  if (saturatedSignal > adcSaturation - 1)
174  return adcSaturation - 1;
175  return saturatedSignal;
176 }

178 template <typename T>
179 inline T SAMPAProcessing::getGamma4(T time, T startTime, T ADC) const
180 {
181  Vc::float_v tmp0 = (time - startTime) / mEleParam->getPeakingTime();
182  Vc::float_m cond = (tmp0 > 0);
183  Vc::float_v tmp;
184  tmp(cond) = tmp0;
185  Vc::float_v tmp2 = tmp * tmp;
186  return 55.f * ADC * Vc::exp(-4.f * tmp) * tmp2 * tmp2;
187 }
