78 bool DataSourceDevice::ConditionalRun() {
79  auto& monitoring = mServiceRegistry.get<o2::monitoring::Monitoring>();
80  monitoring.send({ 1, "dpl/in_handle_data" });
81  ScopedExit metricFlusher([&monitoring] {
82  monitoring.send({ 1, "dpl/in_handle_data" });
83  monitoring.send({ 0, "dpl/in_handle_data" });
84  monitoring.flushBuffer(); });
85  static const auto reftime = std::chrono::system_clock::now();
86  if (mRate > 0.001) {
87  auto timeSinceRef = std::chrono::duration_cast<TimeScale>(std::chrono::system_clock::now() - reftime);
88  auto timespan = timeSinceRef.count() - mLastTime;
89  TimeScale::rep period = (float)TimeScale::period::den / mRate;
90  if (timespan < period) {
91  TimeScale sleepfor(period - timespan);
92  std::this_thread::sleep_for(sleepfor);
93  }
94  mLastTime = std::chrono::duration_cast<TimeScale>(std::chrono::system_clock::now() - reftime).count();
95  }
96  LOG(DEBUG) << "DataSourceDevice::Processing::START";
97  LOG(DEBUG) << "ConditionalRun thread" << pthread_self();
98  // This is dummy because a source does not really have inputs.
99  // However, in order to be orthogonal between sources and
100  // processing code, we still specify it.
101  InputRecord dummyInputs{ {}, { [](size_t) { return nullptr; }, 0 } };
102  try {
103  mTimingInfo.timeslice = mCurrentTimeslice;
104  mContextRegistry.get<MessageContext>()->clear();
105  mContextRegistry.get<RootObjectContext>()->clear();
106  mContextRegistry.get<StringContext>()->clear();
107  mContextRegistry.get<ArrowContext>()->clear();
108  mCurrentTimeslice += 1;
109 
110  // Avoid runaway process in case we have nothing to do.
111  if ((!mStatefulProcess) && (!mStatelessProcess)) {
112  LOG(ERROR) << "No callback defined";
113  sleep(1);
114  }
115 
116  ProcessingContext processingContext{dummyInputs, mServiceRegistry, mAllocator};
117  if (mStatelessProcess) {
118  LOG(DEBUG) << "Has stateless process callback";
119  mStatelessProcess(processingContext);
120  }
121  if (mStatefulProcess) {
122  LOG(DEBUG) << "Has stateful process callback";
123  mStatefulProcess(processingContext);
124  }
125  size_t nMsg = mContextRegistry.get<MessageContext>()->size();
126  nMsg += mContextRegistry.get<RootObjectContext>()->size();
127  nMsg += mContextRegistry.get<StringContext>()->size();
128  nMsg += mContextRegistry.get<ArrowContext>()->size();
129  monitoring.send({ (int)nMsg, "dpl/output_messages" });
130  LOG(DEBUG) << "Process produced " << nMsg << " messages";
131  DataProcessor::doSend(*this, *mContextRegistry.get<MessageContext>());
132  DataProcessor::doSend(*this, *mContextRegistry.get<RootObjectContext>());
133  DataProcessor::doSend(*this, *mContextRegistry.get<StringContext>());
134  DataProcessor::doSend(*this, *mContextRegistry.get<ArrowContext>());
135  } catch(std::exception &e) {
136  if (mError) {
137  ErrorContext errorContext{dummyInputs, mServiceRegistry, e};
138  mError(errorContext);
139  } else {
140  LOG(ERROR) << "Uncaught exception: " << e.what();
141  }
142  } catch(...) {
143  LOG(ERROR) << "Unknown exception type.";
144  LOG(DEBUG) << "DataSourceDevice::Processing::END";
145  return false;
146  }
147  LOG(DEBUG) << "DataSourceDevice::Processing::END";
148  return true;
149 }
