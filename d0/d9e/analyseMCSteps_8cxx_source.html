40 void registerAnalyses(const std::string& analysisDir)
41 {
42  // try to read analyses from analyis directory
43  // \todo try-catch to only read analysis macros and skip other files or non-conforming
44  // analysis classes ==> print warnings/errors
45  TSystemDirectory macroDir(analysisDir.c_str(), analysisDir.c_str());
46  if (macroDir.IsDirectory()) {
47  TList* macros = macroDir.GetListOfFiles();
48  TIter next(macros);
49  TSystemFile* file = nullptr;

61 int analyze(const bpo::variables_map& vm, std::string& errorMessage)
62 {
64  // if external analyses are desired but no analysis directory is passed, fail
65  if (vm.count("analyses") && !vm.count("analysis-dir") && !vm.count("list-analyses")) {
66  errorMessage += "Analysis names but no analysis directory passed.\n";
67  }
69  // now the check for ROOT files
70  if (!vm.count("root-file") && !vm.count("list-analyses")) {
71  errorMessage += "Need ROOT file from MCStepLogger.\n";
72  }
74  // is there an output directory?
75  if (!vm.count("output-dir") && !vm.count("list-analyses")) {
76  errorMessage += "Need an output directory.\n";
77  }
79  // a label needs to be given to identify this analysis run
80  if (!vm.count("label") && !vm.count("list-analyses")) {
81  errorMessage += "Need a label for this analysis.\n";
82  }
83  if (!errorMessage.empty()) {
84  return 1;
85  }
87  // try to read analyses from analyis directory
88  // \todo try-catch to only read analysis macros and skip other files or non-conforming
89  // analysis classes ==> print warnings/errors
90  if (vm.count("analysis-dir") && vm.count("analyses")) {
91  const std::string analysisDir = vm["analysis-dir"].as<std::string>().c_str();
92  registerAnalyses(analysisDir);
93  }
94  // create basic analysis by default, is registered automaticallyt to AnalysisManager
95  new BasicMCAnalysis();
97  // List analyses and quit. This has to be done here after analyses are read from analysis-dir
98  // the first time the AnalysisManager is needed
99  auto& anamgr = MCAnalysisManager::Instance();
100  if (vm.count("list-analyses")) {
101  anamgr.printAnalyses();
102  return 0;
103  }
105  // set a label and the input file from MCStepLogger
106  anamgr.setLabel(vm["label"].as<std::string>());
107  anamgr.setInputFilepath(vm["root-file"].as<std::string>());
108  // if ready, run
109  if (!anamgr.checkReadiness()) {
110  return 1;
111  }
112  anamgr.run(vm["number-events"].as<int>());
113  // save what the AnalysisFileHandler has gotten during the analysis run
114  anamgr.write(vm["output-dir"].as<std::string>());
115  return 0;
116 }

118 // check type and sanity of a file
119 // \note this is a beta version which will always complain unless an analysis file was passed.
120 // In that case, the meta info is printed
121 int checkFile(const bpo::variables_map& vm, std::string& errorMessage)
122 {
123  if (!vm.count("root-file")) {
124  errorMessage += "ROOT file required.\n";
125  }
126  if (!errorMessage.empty()) {
127  return 1;
128  }
129  LOG(INFO) << "Check type and sanity of the input file " << vm["root-file"].as<std::string>();
130  MCAnalysisFileWrapper fileWrapper;
131  if (fileWrapper.read(vm["root-file"].as<std::string>())) {
132  fileWrapper.printAnalysisMetaInfo();
133  fileWrapper.printHistogramInfo();
134  return 0;
135  }
136  auto& anamgr = MCAnalysisManager::Instance();
137  anamgr.setInputFilepath(vm["root-file"].as<std::string>());
138  if (anamgr.dryrun()) {
139  return 0;
140  }
141  LOG(INFO) << "This ROOT file is neither an MCStepLogger nor an MCAnalysis file.";
142  return 1;
143 }

157 int main(int argc, char* argv[])
158 {
159  // Global variables map mapping the command line options to their values
160  bpo::variables_map vm;
161  // Description of the available top-level commands/options
162  bpo::options_description desc("Available commands/options");
163  desc.add_options()("help,h", "show this help message and exit")("command", bpo::value<std::string>(), "command to be executed (\"analyze\", \"checkFile\"")("positional", bpo::value<std::vector<std::string>>(), "positional arguments");
164  // Dedicated description for positional arguments
165  bpo::positional_options_description pos;
166  // First positional argument is actually the command, all others are real positional arguments "( "positional", -1 )"
167  pos.add("command", 1).add("positional", -1);
168  // Parse and store in variables map, allow unregistered since these are not yet specified but will be so as soon as
169  // the command name is checked
170  bpo::parsed_options parsed = bpo::command_line_parser(argc, argv).options(desc).positional(pos).allow_unregistered().run();
171  bpo::store(parsed, vm);
172 
173  // check whether 'command' is missing or if there is even no option at all; show global help message in that case
174  // and return
175  if (!vm.count("command") || vm.empty()) {
176  if (!vm.count("help")) {
177  LOG(FATAL) << "Need command to execute.";
178  }
179  // Print global help message in any case...
180  helpMessage(desc);
181  // ...but check whther it was explicitly asked for it and return accordingly
182  return (!vm.count("help"));
183  }
184 
185  // Extract command
186  const std::string& cmd = vm["command"].as<std::string>();
187  if (std::find(availableCommands.begin(), availableCommands.end(), cmd) == availableCommands.end()) {
188  LOG(ERROR) << "Command \"" << cmd << "\" unknown.";
189  helpMessage(desc);
190  return 1;
191  }
192 
193  // extract all positional arguments passed which are unrecognized by desc
194  std::vector<std::string> opts = bpo::collect_unrecognized(parsed.options, bpo::include_positional);
195  // since 'command' as positional argument is unknown, it needs to be erased explicitly
196  opts.erase(opts.begin());
197 
198  // Preprocessing done.
199  // Track return value
200  int returnValue = 0;
201  // function to be run
202  std::function<int(const bpo::variables_map&, std::string&)> cmdFunction;
203  // corresponding options with description
204  bpo::options_description cmdOptionsDescriptions;
205  // initialize command function and corresponding options description
206  initializeForRun(cmd, cmdOptionsDescriptions, cmdFunction);
207  // Parse again to get positional arguments correctly labelled for analysis run
208  bpo::store(bpo::command_line_parser(opts).options(cmdOptionsDescriptions).run(), vm);
209  // Notify and fix variables map
210  bpo::notify(vm);
211  // To collect error messsages
212  std::string errorMessage("");
213  if (vm.size() == 1) {
214  errorMessage += "Options required...\n";
215  returnValue = 1;
216  }
217  // check if specific help message is desired
218  else if (!vm.count("help")) {
219  returnValue = cmdFunction(vm, errorMessage);
220  }
221  // check return value
222  if (returnValue > 0) {
223  LOG(ERROR) << "Errors occured:";
224  std::cerr << errorMessage << "\n";
225  }
226  if (returnValue > 0 || vm.count("help")) {
227  helpMessage(cmdOptionsDescriptions);
228  }
229  return returnValue;
230 }
