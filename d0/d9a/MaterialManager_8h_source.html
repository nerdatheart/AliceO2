161  // returns global material ID given a "local" material ID for this detector
162  // returns -1 in case local ID not found
163  int getMaterialID(const char* modname, int imat) const
164  {
165  auto lookupiter = mMaterialMap.find(modname);
166  if (lookupiter == mMaterialMap.end()) {
167  return -1;
168  }
169  auto lookup = lookupiter->second;
170 
171  auto iter = lookup.find(imat);
172  if (iter != lookup.end()) {
173  return iter->second;
174  }
175  return -1;
176  }

199  // fill the medium index mapping into a standard vector
200  // the vector gets sized properly and will be overridden
201  void getMediumIDMappingAsVector(const char* modname, std::vector<int>& mapping)
202  {
203  mapping.clear();
204 
205  auto lookupiter = mMediumMap.find(modname);
206  if (lookupiter == mMediumMap.end()) {
207  return;
208  }
209  auto lookup = lookupiter->second;
210 
211  // get the biggest mapped value (maps are sorted in keys)
212  auto maxkey = lookup.rbegin()->first;
213  // resize mapping and initialize with -1 by default
214  mapping.resize(maxkey + 1, -1);
215  // fill vector with entries from map
216  for (auto& p : lookup) {
217  mapping[p.first] = p.second;
218  }
219  }

237  const char* getModuleFromMediumID(int globalindex) const
238  {
239  // loop over module names and corresponding local<->global mapping
240  for (auto& m : mMediumMap) {
241  for (auto& i : m.second) {
242  // is the global index there?
243  if (i.second == globalindex) {
244  // \note maybe unsafe in case mMediumMap is altered in the same scope where the returned C string is used
245  // since that points to memory of string it was derived from.
246  return m.first.c_str();
247  }
248  }
249  }
250  // module is UNKNOWN if global medium ID could not be found.
251  return "UNKNOWN";
252  }

268  void getMediaWithSpecialProcess(EProc process, std::vector<int>& mediumProcessVector)
269  {
270  // clear
271  mediumProcessVector.clear();
272  // resize to maximum number of global IDs for which special processes are set. In case process is not
273  // implemented for a certain medium, value is -1
274  mediumProcessVector.resize(mMediumProcessMap.size(), -1);
275  // find media
276  for (auto& m : mMediumProcessMap) {
277  // loop over processes in medium
278  for (auto& p : m.second) {
279  // push medium ID if process is there
280  if (p.first == process) {
281  mediumProcessVector[m.first] = p.second;
282  break;
283  }
284  }
285  }
