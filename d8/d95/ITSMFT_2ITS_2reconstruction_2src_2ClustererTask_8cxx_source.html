48 InitStatus ClustererTask::Init()
49 {
51 
52  if (mReader) {
53  return kSUCCESS; // already initialized
54  }
55 
56  // create reader according to requested raw of MC mode
57  if (mRawDataMode) {
58  mReaderRaw = std::make_unique<o2::ITSMFT::RawPixelReader<o2::ITSMFT::ChipMappingITS>>();
59  mReader = mReaderRaw.get();
60  } else { // clusterizer of digits needs input from the FairRootManager (at the moment)
61  mReaderMC = std::make_unique<o2::ITSMFT::DigitPixelReader>();
62  mReader = mReaderMC.get();
63 
64  if (!isSelfManagedMode()) {
65  attachFairManagerIO();
66  }
67  }
68 
69  GeometryTGeo* geom = GeometryTGeo::Instance();
70  geom->fillMatrixCache(o2::utils::bit2Mask(o2::TransformType::T2L)); // make sure T2L matrices are loaded
71  mGeometry = geom;
72  mClusterer.setGeometry(geom);
73 
74  mClusterer.print();
75 
76  return kSUCCESS;
77 }

80 void ClustererTask::attachFairManagerIO()
81 {
82  // attachs input/output containers from the FairManager
83 
84  if (mRawDataMode) {
85  LOG(FATAL) << "FairRootManager I/O is not supported in raw data mode" << FairLogger::endl;
86  }
87 
88  FairRootManager* mgr = FairRootManager::Instance();
89  if (!mgr) {
90  LOG(FATAL) << "Could not instantiate FairRootManager. Exiting ..." << FairLogger::endl;
91  }
92 
93  const auto arrDig = mgr->InitObjectAs<const std::vector<o2::ITSMFT::Digit>*>("ITSDigit");
94  if (!arrDig) {
95  LOG(FATAL) << "ITS digits are not registered in the FairRootManager. Exiting ..." << FairLogger::endl;
96  }
97  mReaderMC->setDigits(arrDig);
98 
99  if (mUseMCTruth && !(mClusterer.getWantFullClusters() || mClusterer.getWantCompactClusters())) {
100  mUseMCTruth = false;
101  LOG(WARNING) << "ITS clusters storage is not requested, suppressing MCTruth storage" << FairLogger::endl;
102  }
103 
104  const auto arrDigLbl = mUseMCTruth ? mgr->InitObjectAs<const MCTruth*>("ITSDigitMCTruth") : nullptr;
105 
106  if (!arrDigLbl && mUseMCTruth) {
107  LOG(WARNING) << "ITS digits labels are not registered in the FairRootManager. Continue w/o MC truth ..."
108  << FairLogger::endl;
109  }
110  mReaderMC->setDigitsMCTruth(arrDigLbl);
111 
112  // Register output containers
113  if (mClusterer.getWantFullClusters()) {
114  mFullClusPtr = &mFullClus;
115  mgr->RegisterAny("ITSCluster", mFullClusPtr, kTRUE);
116  LOG(INFO) << Class()->GetName() << " output of full clusters is requested " << FairLogger::endl;
117  } else {
118  LOG(INFO) << Class()->GetName() << " output of full clusters is not requested " << FairLogger::endl;
119  }
120 
121  if (mClusterer.getWantCompactClusters()) {
122  mCompClusPtr = &mCompClus;
123  mgr->RegisterAny("ITSClusterComp", mCompClusPtr, kTRUE);
124  LOG(INFO) << Class()->GetName() << " output of compact clusters is requested " << FairLogger::endl;
125  } else {
126  LOG(INFO) << Class()->GetName() << " output of compact clusters is not requested " << FairLogger::endl;
127  }
128 
129  // Register output MC Truth container if there is an MC truth for digits
130  if (arrDigLbl) {
131  mClsLabelsPtr = &mClsLabels;
132  mgr->RegisterAny("ITSClusterMCTruth", mClsLabelsPtr, kTRUE);
133  }
134  LOG(INFO) << Class()->GetName() << " | MCTruth: " << (arrDigLbl ? "ON" : "OFF") << FairLogger::endl;
135 }

156 void ClustererTask::run(const std::string inpName, const std::string outName, bool entryPerROF)
157 {
158  // standalone execution
159  setSelfManagedMode(true); // to prevent interference with FairRootManager io
160  Init(); // create reader, clusterer
161 
162  std::unique_ptr<TFile> outFile(TFile::Open(outName.data(), "recreate"));
163  if (!outFile || outFile->IsZombie()) {
164  LOG(FATAL) << "Failed to open output file " << outName << FairLogger::endl;
165  }
166  std::unique_ptr<TTree> outTree = std::make_unique<TTree>("o2sim", "ITS Clusters");
167 
168  if (mClusterer.getWantFullClusters()) {
169  mFullClusPtr = &mFullClus;
170  outTree->Branch("ITSCluster", &mFullClusPtr);
171  LOG(INFO) << Class()->GetName() << " output of full clusters is requested " << FairLogger::endl;
172  } else {
173  LOG(INFO) << Class()->GetName() << " output of full clusters is not requested " << FairLogger::endl;
174  }
175 
176  if (mClusterer.getWantCompactClusters()) {
177  mCompClusPtr = &mCompClus;
178  outTree->Branch("ITSClusterComp", &mCompClusPtr);
179  LOG(INFO) << Class()->GetName() << " output of compact clusters is requested " << FairLogger::endl;
180  } else {
181  LOG(INFO) << Class()->GetName() << " output of compact clusters is not requested " << FairLogger::endl;
182  }
183 
184  if (entryPerROF) {
185  mClusterer.setOutputTree(outTree.get()); // this will force flushing at every ROF
186  }
187 
188  if (mRawDataMode) {
189  mReaderRaw->openInput(inpName);
190  mClusterer.process(*mReaderRaw.get(), mFullClusPtr, mCompClusPtr);
191  } else {
192  mReaderMC->openInput(inpName, o2::detectors::DetID("ITS"));
193 
194  if (mUseMCTruth && !(mClusterer.getWantFullClusters() || mClusterer.getWantCompactClusters())) {
195  mUseMCTruth = false;
196  LOG(WARNING) << "ITS clusters storage is not requested, suppressing MCTruth storage" << FairLogger::endl;
197  }
198 
199  if (mUseMCTruth && mReaderMC->getDigitsMCTruth()) {
200  // digit labels are provided directly to clusterer
201  mClsLabelsPtr = &mClsLabels;
202  outTree->Branch("ITSClusterMCTruth", &mClsLabelsPtr);
203  } else {
204  mUseMCTruth = false;
205  }
206  LOG(INFO) << Class()->GetName() << " | MCTruth: " << (mUseMCTruth ? "ON" : "OFF") << FairLogger::endl;
207 
208  // loop over entries of the input tree
209  while (mReaderMC->readNextEntry()) {
210  mClusterer.process(*mReaderMC.get(), mFullClusPtr, mCompClusPtr, mClsLabelsPtr);
211  }
212  }
213 
214  if (!entryPerROF) { // Clustered was not managing the output, do it here
215  outTree->Fill(); // in this mode all ROF will go to single entry of the tree
216  }
217  outTree->Write();
218  outTree.reset(); // tree should be destroyed before the file is closed
219  outFile->Close();
220 
221  mClusterer.clear();
222 }
