62  auto initFunction = [sendMC, fanSpec](InitContext& ic) {
63  // FIXME: the clusterer needs to be initialized with the sector number, so we need one
64  // per sector. Taking a closer look to the HwClusterer, the sector number is only used
65  // for calculating the CRU id. This could be achieved by passing the current sector as
66  // parameter to the clusterer processing function.
67  auto processAttributes = std::make_shared<ProcessAttributes>();
68 
69  auto processingFct = [processAttributes, sendMC, fanSpec](ProcessingContext& pc) {
70  auto& clusterArray = processAttributes->clusterArray;
71  auto& mctruthArray = processAttributes->mctruthArray;
72  auto& clusterers = processAttributes->clusterers;
73  auto& verbosity = processAttributes->verbosity;
74  auto const* sectorHeader = DataRefUtils::getHeader<o2::TPC::TPCSectorHeader*>(pc.inputs().get("digits"));
75  if (sectorHeader == nullptr) {
76  LOG(ERROR) << "sector header missing on header stack";
77  return;
78  }
79  const auto sector = sectorHeader->sector;
80  if (sector < 0) {
81  // forward the control information
82  // FIXME define and use flags in TPCSectorHeader
83  o2::TPC::TPCSectorHeader header{ sector };
84  pc.outputs().snapshot(OutputRef{ "clusters", fanSpec, { header } }, fanSpec);
85  if (sendMC) {
86  pc.outputs().snapshot(OutputRef{ "clusterlbl", fanSpec, { header } }, fanSpec);
87  }
88  return;
89  }
90  auto inMCLabels = pc.inputs().get<const MCLabelContainer*>("mclabels");
91  auto inDigits = pc.inputs().get<const std::vector<o2::TPC::Digit>>("digits");
92  if (verbosity > 0) {
93  LOG(INFO) << "received " << inDigits.size() << " digits";
94  }
95  if (!clusterers[sector]) {
96  // create the clusterer for this sector, take the same target arrays for all clusterers
97  // as they are not invoked in parallel
98  // the cost of creating the clusterer should be small so we do it in the processing
99  clusterers[sector] = std::make_shared<o2::TPC::HwClusterer>(&clusterArray, sector, &mctruthArray);
100  }
101  auto& clusterer = clusterers[sector];
102 
103  if (verbosity > 0) {
104  LOG(INFO) << "processing " << inDigits.size() << " digit object(s) of sector " << sectorHeader->sector;
105  }
106  clusterArray.clear(); // this would also be done in the HwClusterer if the clearContainerFirst of process() would be set to true instead of false
107  mctruthArray.clear(); // this would also be done in the HwClusterer if the clearContainerFirst of process() would be set to true instead of false
108  clusterer->process(inDigits, inMCLabels.get(), false);
109  const std::vector<o2::TPC::Digit> emptyDigits;
110  clusterer->finishProcess(emptyDigits, nullptr, false); // keep here the false, otherwise the clusters are lost of they are not stored in the meantime
111  if (verbosity > 0) {
112  LOG(INFO) << "clusterer produced " << clusterArray.size() << " cluster container";
113  }
114  pc.outputs().snapshot(OutputRef{ "clusters", fanSpec, { *sectorHeader } }, clusterArray);
115  if (sendMC) {
116  pc.outputs().snapshot(OutputRef{ "clusterlbl", fanSpec, { *sectorHeader } }, mctruthArray);
117  }
118  };
119 
120  return processingFct;
121  };
