78 void MCAnalysisManager::initialize()
79 {
80  if (mIsInitialized) {
81  LOG(ERROR) << "Already initialized ==> not initialized again...";
82  return;
83  }
84  // get rid of all analyses not needed anymore
85  for (auto& a : mAnalysesToDump) {
86  delete a;
87  }
88  mAnalysesToDump.clear();
89 
90  for (auto& a : mAnalyses) {
91  // prepare an analysis file and set first analysis meta info
92  mAnalysisFiles.emplace_back();
93  mAnalysisFiles.back().getAnalysisMetaInfo().label = mLabel;
94  mAnalysisFiles.back().getAnalysisMetaInfo().analysisName = a->name();
95  a->setAnalysisFile(mAnalysisFiles.back());
96  a->initialize();
97  a->isInitialized(true);
98  }
99  mIsInitialized = true;
100 }

102 bool MCAnalysisManager::analyze(int nEvents, bool isDryrun)
103 {
104  if (!mIsInitialized && !isDryrun) {
105  LOG(ERROR) << "Not yet initialized ==> nothing to analyze...";
106  return false;
107  }
108  // taking only the first entry is a hack! \todo change this by enabling also for TChains
109  ROOTIOUtilities rootutil(mInputFilepath);
110 
111  // this is by default opening the file in READ mode
112  if (!rootutil.changeToTTree(mAnalysisTreename)) {
113  rootutil.close();
114  if (isDryrun) {
115  return false;
116  }
117  LOG(FATAL) << "Tree " << mAnalysisTreename << " could not be found in file " << mInputFilepath;
118  exit(1);
119  }
120  // print warning if desired number of entries is bigger than number of present entries
121  if (nEvents > rootutil.nEntries()) {
122  LOG(WARNING) << "You want to process " << nEvents << ", however only " << rootutil.nEntries() << " are present.";
123  }
124 
125  // prepare variables and connect to branches
126  // \todo align branch names with MCStepLogger and get rid of hard coded names
127  if (!rootutil.setBranch("Steps", &mCurrentStepInfo) || !rootutil.setBranch("Calls", &mCurrentMagCallInfo) || !rootutil.setBranch("Lookups", &mCurrentLookups)) {
128  rootutil.close();
129  if (isDryrun) {
130  return false;
131  }
132  LOG(FATAL) << "Cannot find required branches in TTree " << mAnalysisTreename;
133  exit(1);
134  }
135  // process tree and analyze
136  while (rootutil.processTTree()) {
137  if (nEvents <= mCurrentEventNumber && nEvents > 0) {
138  break;
139  }
140  // check whether all pointers to MCStepLogger branches are set...
141  if (mCurrentStepInfo == nullptr || mCurrentMagCallInfo == nullptr || mCurrentLookups == nullptr) {
142  rootutil.close();
143  if (isDryrun) {
144  return false;
145  }
146  LOG(FATAL) << "Obtained nullptrs while processing TTree " << mAnalysisTreename;
147  exit(1);
148  }
149  // ... if so, next event
150  mCurrentEventNumber++;
151  mNSteps += mCurrentStepInfo->size();
152 
153  std::cout << "---> Event " << mCurrentEventNumber << " <---\n";
154  std::cout << "#steps: " << mCurrentStepInfo->size() << "\n";
155  std::cout << "#mag field calls: " << mCurrentMagCallInfo->size() << "\n";
156  if (!isDryrun) {
157  std::cout << "\nStart..." << std::endl;
158  for (auto& a : mAnalyses) {
159  std::cout << "\t\tCall analysis " << a->name() << std::endl;
160  a->analyze(mCurrentStepInfo, mCurrentMagCallInfo);
161  }
162  std::cout << "Done\n";
163  }
164  }
165  rootutil.close();
166  if (!isDryrun) {
167  LOG(INFO) << "Analysis run on file " << mInputFilepath << " done.";
168  mIsAnalyzed = true;
169  } else {
170  mCurrentEventNumber = 0;
171  mNSteps = 0;
172  }
173  return true;
174 }
