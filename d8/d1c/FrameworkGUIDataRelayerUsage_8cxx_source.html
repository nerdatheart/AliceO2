64  for (size_t mi = 0, me = getNumItems(record); mi < me; mi++) {
65  ImVec2 yOffSet{ 0, (mi * boxSizeY) + padding };
66  ImVec2 ySize{ 0, boxSizeY - 2 * padding };
67  drawList->AddQuadFilled(
68  xOffset + yOffSet + winPos,
69  xOffset + xSize + yOffSet + winPos,
70  xOffset + xSize + yOffSet + ySize + winPos,
71  xOffset + yOffSet + ySize + winPos,
72  getColor(getValue(getItem(record, mi))));
73  }
74  }
75 
76  ImGui::SetCursorScreenPos(winPos + size);
77  }

80 void displayDataRelayer(DeviceMetricsInfo const& metrics,
81  DeviceInfo const& info,
82  ImVec2 const &size)
83 {
84  auto& viewIndex = info.dataRelayerViewIndex;
85 
86  auto getNumRecords = [&viewIndex]() -> size_t {
87  if (viewIndex.isComplete()) {
88  return viewIndex.w;
89  }
90  return 0;
91  };
92  auto getRecord = [&metrics](size_t i) -> int {
93  return i;
94  };
95  auto getNumItems = [&viewIndex](int record) -> int {
96  if (viewIndex.isComplete()) {
97  return viewIndex.h;
98  }
99  return 0;
100  };
101  auto getItem = [&metrics, &viewIndex](int const& record, size_t i) -> int const& {
102  // Calculate the index in the viewIndex.
103  auto idx = record * viewIndex.h + i;
104  assert(viewIndex.indexes.size() > idx);
105  MetricInfo const& metricInfo = metrics.metrics[viewIndex.indexes[idx]];
106  assert(metrics.intMetrics.size() > metricInfo.storeIdx);
107  auto& data = metrics.intMetrics[metricInfo.storeIdx];
108  return data[(metricInfo.pos - 1) % data.size()];
109  };
110  auto getValue = [](int const& item) -> int { return item; };
111  auto getColor = [](int value) {
112  const ImU32 SLOT_EMPTY = ImColor(70, 70, 70, 255);
113  const ImU32 SLOT_FULL = ImColor(0xf9, 0xcd, 0xad, 255);
114  const ImU32 SLOT_DISPATCHED = ImColor(0xc8, 0xc8, 0xa9, 255);
115  const ImU32 SLOT_DONE = ImColor(0x83, 0xaf, 0, 255);
116  const ImU32 SLOT_ERROR = ImColor(0xfe, 0x43, 0x65, 255);
117  switch (value) {
118  case 0:
119  return SLOT_EMPTY;
120  case 1:
121  return SLOT_FULL;
122  case 2:
123  return SLOT_DISPATCHED;
124  case 3:
125  return SLOT_DONE;
126  }
127  return SLOT_ERROR;
128  };
129 
130  if (getNumRecords()) {
131  HeatMapHelper::draw<int, int>("DataRelayer",
132  size,
133  getNumRecords,
134  getRecord,
135  getNumItems,
136  getItem,
137  getValue,
138  getColor);
139  }
140 }
