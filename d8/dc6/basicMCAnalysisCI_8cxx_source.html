59 // checking meta info and whether there is a value container the analysis can be conpared with
60 BOOST_AUTO_TEST_CASE(meta_mcbasicanalysis_test)
61 {
62  // a path to a ROOT file containing the analysis histograms and a JSON file containing reference values, hence 3 arguments
63  BOOST_REQUIRE(boost::unit_test::framework::master_test_suite().argc == 3);
64  // read the file to a file wrapper
65  MCAnalysisFileWrapper fileWrapper;
66  BOOST_REQUIRE(fileWrapper.read(boost::unit_test::framework::master_test_suite().argv[1]));
67  // read the reference JSON file
68  rj::Document doc;
69  std::FILE* referenceFile = std::fopen(boost::unit_test::framework::master_test_suite().argv[2], "r");
70  // make sure a file was read
71  BOOST_REQUIRE(referenceFile != nullptr);
72  char readBuffer[65536];
73  rj::FileReadStream is(referenceFile, readBuffer, sizeof(readBuffer));
74  // read the JSON assuming that it's same
75  doc.ParseStream(is);
76  //std::cerr << doc.Size() << std::endl;
77  // meta info from step logging and analysis run
78  MCAnalysisMetaInfo& anaMetaInfo = fileWrapper.getAnalysisMetaInfo();
79  // check only for the right analysis name, considering that the user knows which histograms to expect from this analysis
80  BOOST_REQUIRE(doc.HasMember("analysisName"));
81  BOOST_REQUIRE(anaMetaInfo.analysisName.compare(doc["analysisName"].GetString()) == 0);
82 
83  // require histograms, so fail if not there
84  BOOST_REQUIRE(fileWrapper.hasHistogram("nSteps"));
85  BOOST_REQUIRE(fileWrapper.hasHistogram("nTracks"));
86 
87  // have checked that histograms are there, so get them
88  TH1& histNSteps = fileWrapper.getHistogram("nSteps");
89  TH1& histNTracks = fileWrapper.getHistogram("nTracks");
90 
91  // again assuming the user passed JSON containing what is needed
92  const rj::Value& nSteps = doc["nSteps"];
93  const rj::Value& nTracks = doc["nTracks"];
94  // ...and perform tests, assuming the JSON values are arrays
95  BOOST_TEST(nSteps[0].GetFloat() == float(histNSteps.GetEntries()), tt::tolerance(nSteps[1].GetFloat()));
96  BOOST_TEST(nTracks[0].GetFloat() == float(histNTracks.GetEntries()), tt::tolerance(nTracks[1].GetFloat()));
97 }
