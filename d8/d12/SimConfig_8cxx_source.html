42 bool SimConfig::resetFromParsedMap(boost::program_options::variables_map const& vm)
43 {
44  using o2::detectors::DetID;
45  mConfigData.mMCEngine = vm["mcEngine"].as<std::string>();
46  mConfigData.mActiveDetectors = vm["modules"].as<std::vector<std::string>>();
47  auto& active = mConfigData.mActiveDetectors;
48  if (active.size() == 1 && active[0] == "all") {
49  active.clear();
50  for (int d = DetID::First; d <= DetID::Last; ++d) {
51  active.emplace_back(DetID::getName(d));
52  }
53  }
54  // now we take out detectors listed as skipped
55  auto& skipped = vm["skipModules"].as<std::vector<std::string>>();
56  for (auto& s : skipped) {
57  auto iter = std::find(active.begin(), active.end(), s);
58  if (iter != active.end()) {
59  // take it out
60  active.erase(iter);
61  }
62  }
63 
64  mConfigData.mGenerator = vm["generator"].as<std::string>();
65  mConfigData.mNEvents = vm["nEvents"].as<unsigned int>();
66  mConfigData.mExtKinFileName = vm["extKinFile"].as<std::string>();
67  mConfigData.mExtGenFileName = vm["extGenFile"].as<std::string>();
68  mConfigData.mExtGenFuncName = vm["extGenFunc"].as<std::string>();
69  mConfigData.mStartEvent = vm["startEvent"].as<unsigned int>();
70  mConfigData.mBMax = vm["bMax"].as<float>();
71  mConfigData.mIsMT = vm["isMT"].as<bool>();
72  mConfigData.mOutputPrefix = vm["outPrefix"].as<std::string>();
73  mConfigData.mLogSeverity = vm["logseverity"].as<std::string>();
74  mConfigData.mLogVerbosity = vm["logverbosity"].as<std::string>();
75  mConfigData.mKeyValueTokens = vm["configKeyValues"].as<std::string>();
76  return true;
77 }
