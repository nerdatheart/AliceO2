57 int AlpideCoder::encodeChip(short chipInModule, short framestartdata, short roflags)
58 {
59  //print();

61  int nfound = 0;
62  // chip header
63  addToBuffer(makeChipHeader(chipInModule, framestartdata));
64  for (int ir = NRegions; ir--;) {
65  nfound += procRegion(ir);
66  }
67  if (nfound) {
68  addToBuffer(makeChipTrailer(roflags, 0));
69  } else {
70  eraseInBuffer(SizeChipHeader);

73  //
74  resetMap();

79 int AlpideCoder::procDoubleCol(short reg, short dcol)
80 {
81  // process double column: encoding
82  short hits[2 * NRows];
83  int nHits = 0, nData = 0;
84  //
85  int nr = mFirstInRow.size();
86  short col0 = ((reg * NDColInReg + dcol) << 1), col1 = col0 + 1; // 1st,2nd column of double column
87  int prevRow = -1;
88  for (int ir = 0; ir < nr; ir++) {
89  int linkID = mFirstInRow[ir];
90  if (linkID == -1 || mPix2Encode[linkID].col > col1) { // no pixels left on this row or higher column IDs
91  continue;
92  }
93  short rowID = mPix2Encode[linkID].row;
94  // process fired pixels
95  bool left = 0, right = 0;
96  if (mPix2Encode[linkID].col == col0) { // pixel in left column
97  left = 1;
98  linkID = mPix2Encode[linkID].nextInRow; // unlink processed pixel
99  }
100  if (linkID != -1 && mPix2Encode[linkID].col == col1) { // pixel in right column
101  right = 1;
102  linkID = mPix2Encode[linkID].nextInRow; // unlink processed pixel
103  }
104  short addr0 = rowID << 1;
105  if (rowID & 0x1) { // odd rows: right to left numbering
106  if (right) {
107  hits[nHits++] = addr0;
108  }
109  if (left) {
110  hits[nHits++] = addr0 + 1;
111  }
112  } else { // even rows: left to right numbering
113  if (left) {
114  hits[nHits++] = addr0;
115  }
116  if (right) {
117  hits[nHits++] = addr0 + 1;
118  }
119  }
120  if (linkID == -1) {
121  mFirstInRow[ir] = -1;
122  continue;
123  } // this row is finished
124  mFirstInRow[ir] = linkID; // link remaining hit pixels to row
125  }
126  //
127  int ih = 0;
128  while ((ih < nHits)) {
129  short addrE, addrW = hits[ih++]; // address of the reference hit
130  UChar_t mask = 0, npx = 0;
131  short addrLim = addrW + HitMapSize + 1; // 1+address of furthest hit can be put in the map
132  while ((ih < nHits && (addrE = hits[ih]) < addrLim)) {
133  mask |= 0x1 << (addrE - addrW - 1);
134  ih++;
135  }
136  if (mask) { // flag DATALONG
137  addToBuffer(makeDataLong(dcol, addrW));
138  addToBuffer(mask);
139  } else {
140  addToBuffer(makeDataShort(dcol, addrW));
141  }
142  nData++;
143  }
144  //
145  return nData;
146 }

200 bool AlpideCoder::flushBuffer()
201 {
202  // flush current content of buffer
203  if (!mWrBufferFill) {
204  return false;
205  }
206  if (!mIOFile) {
207  LOG(FATAL) << "Output handler is not created" << FairLogger::endl;
208  }
209  mTimerIO.Start(false);
210  fwrite(mWrBuffer, sizeof(char), mWrBufferFill, mIOFile);
211  mTimerIO.Stop();
212  mWrBufferFill = 0; // reset the counter
213  return true;
214 }

217 int AlpideCoder::loadInBuffer(int chunk)
218 {
219  // upload next chunk of data to buffer
220  int save = mBufferEnd - mBufferPointer;
221  // move unprocessed part to the beginning of the buffer
222  if (save > 0) {
223  memmove(mWrBuffer, mBufferPointer, save);
224  } else {
225  save = 0;
226  }
227  //
228  if (mWrBufferSize < (chunk + save)) {
229  expandBuffer(chunk + save + 1000);
230  }
231  mBufferPointer = mWrBuffer;
232 #ifdef _DEBUG_PIX_CONV_
233  printf("loadInBuffer: %d bytes placed with offset %d\n", chunk, save);
234 #endif
235  mTimerIO.Start(false);
236  int nc = (int)fread(mWrBuffer + save, sizeof(char), chunk, mIOFile);
237  mTimerIO.Stop();
238  mWrBufferFill = save + nc;
239  mBufferEnd = mWrBuffer + mWrBufferFill;
240  return mWrBufferFill;
241 }

252 int AlpideCoder::readChipData(std::vector<AlpideCoder::HitsRecord>& hits,
253  UShort_t& chip, // updated on change, don't modify returned value
254  UShort_t& module, // updated on change, don't modify returned value
255  UShort_t& cycle) // updated on change, don't modify returned value
256 {
257  // read record for single non-empty chip, updating on change module and cycle.
258  // return number of records filled (>0), EOFFlag or Error
259  //
260  hits.clear();
261  UChar_t dataC = 0;
262  UChar_t region = 0;
263  UChar_t framestartdata = 0;
264  UShort_t dataS = 0;
265  UInt_t dataI = 0;
266  //
267  int nHitsRec = 0;
268  //
269  while (1) {
270  //
271  if (!getFromBuffer(dataC)) {
272  return EOFFlag;
273  }
274  //
275  if (mExpectInp & ExpectModuleHeader && dataC == MODULEHEADER) { // new module header
276  if (!getFromBuffer(module) || !getFromBuffer(cycle)) {
277  return unexpectedEOF("MODULE_HEADER");
278  }
279  mExpectInp = ExpectModuleTrailer | ExpectChipHeader | ExpectChipEmpty;
280  continue;
281  }
282  //
283  if (mExpectInp & ExpectModuleTrailer && dataC == MODULETRAILER) { // module trailer
284  UShort_t moduleT, cycleT;
285  if (!getFromBuffer(moduleT) || !getFromBuffer(cycleT)) {
286  return unexpectedEOF("MODULE_HEADER");
287  }
288  if (moduleT != module || cycleT != cycle) {
289  LOG(ERROR) << "Error: expected module trailer for module " << module << "/cycle " << cycle << ", got for module "
290  << moduleT << "/cycle " << cycleT << FairLogger::endl;
291  return Error;
292  }
293  mExpectInp = ExpectModuleHeader;
294  continue;
295  }
296  // ---------- chip info ?
297  UChar_t dataCM = dataC & (~MaskChipID);
298  //
299  if ((mExpectInp & ExpectChipHeader) && dataCM == CHIPHEADER) { // chip header was expected
300  chip = dataC & MaskChipID;
301  if (!getFromBuffer(framestartdata)) {
302  return unexpectedEOF("CHIP_HEADER");
303  }
304  mExpectInp = ExpectRegion; // now expect region info
305  continue;
306  }
307  //
308  if ((mExpectInp & ExpectChipEmpty) && dataCM == CHIPEMPTY) { // chip trailer was expected
309  chip = dataC & MaskChipID;
310  if (!getFromBuffer(framestartdata)) {
311  return unexpectedEOF("CHIP_EMPTY:FrameStartData");
312  }
313  if (!getFromBuffer(dataC)) {
314  return unexpectedEOF("CHIP_EMPTY:ReservedWord");
315  }
316  mExpectInp = ExpectModuleTrailer | ExpectChipHeader | ExpectChipEmpty;
317  continue;
318  }
319  //
320  if ((mExpectInp & ExpectChipTrailer) && dataCM == CHIPTRAILER) { // chip trailer was expected
321  if (!getFromBuffer(framestartdata)) {
322  return unexpectedEOF("CHIP_TRAILER:FrameStartData");
323  }
324  mExpectInp = ExpectModuleTrailer | ExpectChipHeader | ExpectChipEmpty;
325  return hits.size();
326  }
327  // region info ?
328  if ((mExpectInp & ExpectRegion) && (dataC & REGION) == REGION) { // chip header was seen, or hit data read
329  region = dataC & MaskRegion;
330  mExpectInp = ExpectData;
331  continue;
332  }
333  // hit info ?
334  if ((mExpectInp & ExpectData)) { // region header was seen, expect data
335  stepBackInBuffer(); // need to reinterpred as short
336  if (!getFromBuffer(dataS)) {
337  return unexpectedEOF("CHIPDATA");
338  }
339  UShort_t dataSM = dataS & (~MaskDColID); // check hit data mask
340  if (dataSM == DATASHORT) { // single hit
341  UChar_t dColID = (dataS & MaskEncoder) >> 10;
342  UShort_t pixID = dataS & MaskPixID;
343  hits.emplace_back(region, dColID, pixID, 0);
344  mExpectInp = ExpectData | ExpectRegion | ExpectChipTrailer;
345  continue;
346  } else if (dataSM == DATALONG) { // multiple hits
347  UChar_t dColID = (dataS & MaskEncoder) >> 10;
348  UShort_t pixID = dataS & MaskPixID;
349  UChar_t hitsPattern = 0;
350  if (!getFromBuffer(hitsPattern)) {
351  return unexpectedEOF("CHIP_DATA_LONG:Pattern");
352  }
353  hits.emplace_back(region, dColID, pixID, hitsPattern);
354  mExpectInp = ExpectData | ExpectRegion | ExpectChipTrailer;
355  continue;
356  } else {
357  LOG(ERROR) << "Expected DataShort or DataLong mask, got : " << dataSM << FairLogger::endl;
358  return Error;
359  }
360  }
361  }
362 }
