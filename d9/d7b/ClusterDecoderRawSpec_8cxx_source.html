66  auto processingFct = [verbosity, decoder, sendMC, fanSpec](ProcessingContext& pc) {
67  // this will return a span of TPC clusters
68  const auto& ref = pc.inputs().get("rawin");
69  auto size = o2::framework::DataRefUtils::getPayloadSize(ref);
70  if (ref.payload == nullptr) {
71  return;
72  }

74  // init the stacks for forwarding the sector header
75  // FIXME check if there is functionality in the DPL to forward the stack
76  // FIXME make one function
77  o2::header::Stack rawHeaderStack;
78  o2::header::Stack mcHeaderStack;
79  o2::TPC::TPCSectorHeader const* sectorHeaderMC = nullptr;
80  if (sendMC) {
81  sectorHeaderMC = DataRefUtils::getHeader<o2::TPC::TPCSectorHeader*>(pc.inputs().get("mclblin"));
82  if (sectorHeaderMC) {
83  o2::header::Stack actual{ *sectorHeaderMC };
84  std::swap(mcHeaderStack, actual);
85  if (sectorHeaderMC->sector < 0) {
86  pc.outputs().snapshot(OutputRef{ "mclblout", fanSpec, std::move(mcHeaderStack) }, fanSpec);
87  }

89  }

91  if (sectorHeader) {
92  o2::header::Stack actual{ *sectorHeader };
93  std::swap(rawHeaderStack, actual);
94  if (sectorHeader->sector < 0) {
95  pc.outputs().snapshot(OutputRef{ "clusterout", fanSpec, std::move(rawHeaderStack) }, fanSpec);
96  return;
97  }
98  }
99  assert(sectorHeaderMC == nullptr || sectorHeader->sector == sectorHeaderMC->sector);

101  // input to the decoder is a vector of raw pages description ClusterHardwareContainer,
102  // each specified as a pair of pointer to ClusterHardwareContainer and the number
103  // of pages in that buffer
104  // FIXME: better description of the raw page
105  size_t nPages = size / 8192;
106  std::vector<std::pair<const ClusterHardwareContainer*, std::size_t>> inputList;
107 
108  // MC labels are received as one container of labels in the sequence matching clusters
109  // in the raw pages
110  std::vector<MCLabelContainer> mcinCopies;
111  std::unique_ptr<const MCLabelContainer> mcin;
112  if (sendMC) {
113  mcin = std::move(pc.inputs().get<MCLabelContainer*>("mclblin"));
114  mcinCopies.resize(nPages);
115  if (verbosity > 0) {
116  LOG(INFO) << "Decoder input: " << size << ", " << nPages << " pages, " << mcin->getIndexedSize() << " MC label sets";
117  }
118  }
119 
120  // FIXME: the decoder takes vector of MCLabelContainers as input and the retreived
121  // input can not be moved because the input is const, so we have to copy
122  // Furthermore, the current decoder implementation supports handling of MC labels
123  // only for single 8kb pages. So we have to add the raw pages individually and create
124  // MC label containers for the corresponding clusters.
125  size_t mcinPos = 0;
126  for (size_t page = 0; page < nPages; page++) {
127  inputList.emplace_back(reinterpret_cast<const ClusterHardwareContainer*>(ref.payload + page * 8192), 1);
128  const ClusterHardwareContainer& container = *(inputList.back().first);
129  if (verbosity > 0) {
130  LOG(INFO) << "Decoder input in page " << std::setw(2) << page << ": " //
131  << "CRU " << std::setw(3) << container.CRU << " " //
132  << std::setw(3) << container.numberOfClusters << " cluster(s)"; //
133  }
134  if (mcin) {
135  for (size_t mccopyPos = 0; mccopyPos < container.numberOfClusters; mccopyPos++, mcinPos++) {
136  for (auto const& label : mcin->getLabels(mcinPos)) {
137  mcinCopies[page].addElement(mccopyPos, label);
138  }
139  }
140  }
141  }
142  assert(!mcin || mcinPos == mcin->getIndexedSize());
143  // output of the decoder is sorted in (sector,globalPadRow) coordinates, individual
144  // containers are created for clusters and MC labels per (sector,globalPadRow) address
145  std::vector<ClusterNativeContainer> cont;
146  std::vector<MCLabelContainer> mcoutList;
147  decoder->decodeClusters(inputList, cont, (mcin ? &mcinCopies : nullptr), &mcoutList);
148 
149  // The output of clusters involves a copy to flatten the list of buffers and extend the
150  // ClusterGroupAttribute struct containing sector and globalPadRow by the size of the collection.
151  // FIXME: provide allocator to decoder to do the allocation in place, create ClusterGroupHeader
152  // directly
153  // The vector of MC label containers is simply serialized
154  size_t totalSize = 0;
155  for (const auto& coll : cont) {
156  const auto& groupAttribute = static_cast<const ClusterGroupAttribute>(coll);
157  totalSize += coll.getFlatSize() + sizeof(ClusterGroupHeader) - sizeof(ClusterGroupAttribute);
158  }
159 
160  auto* target = pc.outputs().newChunk(OutputRef{ "clusterout", fanSpec, std::move(rawHeaderStack) }, totalSize).data;
161 
162  for (const auto& coll : cont) {
163  if (verbosity > 0) {
164  LOG(INFO) << "decoder " << std::setw(2) << sectorHeader->sector //
165  << ": decoded " << std::setw(4) << coll.clusters.size() << " clusters on sector " //
166  << std::setw(2) << (int)coll.sector << "[" << (int)coll.globalPadRow << "]"; //
167  }
168  ClusterGroupHeader groupHeader(coll, coll.clusters.size());
169  memcpy(target, &groupHeader, sizeof(groupHeader));
170  target += sizeof(groupHeader);
171  memcpy(target, coll.data(), coll.getFlatSize() - sizeof(ClusterGroupAttribute));
172  target += coll.getFlatSize() - sizeof(ClusterGroupAttribute);
173  }
174  if (sendMC) {
175  if (verbosity > 0) {
176  LOG(INFO) << "sending " << mcoutList.size() << " MC label containers" << std::endl;
177  }
178  // serialize the complete list of MC label containers
179  pc.outputs().snapshot(OutputRef{ "mclblout", fanSpec, std::move(mcHeaderStack) }, mcoutList);
180  }
181  };

186  };
