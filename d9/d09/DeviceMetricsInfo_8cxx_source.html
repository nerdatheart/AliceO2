35 bool DeviceMetricsHelper::processMetric(const std::smatch& match,
36  DeviceMetricsInfo& info,
37  DeviceMetricsHelper::NewMetricCallback newMetricsCallback)
38 {
39  auto type = match[2];
40  auto name = match[1];
41  char *ep = nullptr;
42  auto timestamp = strtol(match[4].str().c_str(), &ep, 10);
43  if (ep == nullptr || *ep != '\0') {
44  return false;
45  }
46  auto stringValue = match[3];
47  size_t metricIndex = -1;
48 
49  auto metricType = MetricType::Unknown;
50  if (type.str() == "0") {
51  metricType = MetricType::Int;
52  } else if (type.str() == "2") {
53  metricType = MetricType::Float;
54  }
55 
56  int intValue = 0;
57  float floatValue = 0;
58  switch (metricType) {
59  case MetricType::Int:
60  intValue = strtol(stringValue.str().c_str(), &ep, 10);
61  if (!ep || *ep != '\0') {
62  return false;
63  }
64  break;
65  case MetricType::Float:
66  floatValue = strtof(stringValue.str().c_str(), &ep);
67  if (!ep || *ep != '\0') {
68  return false;
69  }
70  break;
71  default:
72  return false;
73  break;
74  };
75 
76  // Find the metric based on the label. Create it if not found.
77  using IndexElement = std::pair<std::string, size_t>;
78  auto cmpFn = [](const IndexElement &a, const IndexElement &b) -> bool {
79  return std::tie(a.first, a.second) < std::tie(b.first, b.second);
80  };
81  IndexElement metricLabelIdx = std::make_pair(name.str(), 0);
82  auto mi = std::lower_bound(info.metricLabelsIdx.begin(),
83  info.metricLabelsIdx.end(),
84  metricLabelIdx,
85  cmpFn);
86 
87  // We could not find the metric, lets insert a new one.
88  if (mi == info.metricLabelsIdx.end()
89  || mi->first != metricLabelIdx.first) {
90  // Create a new metric
91  MetricInfo metricInfo;
92  metricInfo.pos = 0;
93  metricInfo.type = metricType;
94  // Add a new empty buffer for it of the correct kind
95  switch(metricType) {
96  case MetricType::Int:
97  metricInfo.storeIdx = info.intMetrics.size();
98  info.intMetrics.emplace_back(std::array<int, 1024>{});
99  break;
100  case MetricType::Float:
101  metricInfo.storeIdx = info.floatMetrics.size();
102  info.floatMetrics.emplace_back(std::array<float, 1024>{});
103  break;
104  default:
105  return false;
106  };
107  // Add the timestamp buffer for it
108  info.timestamps.emplace_back(std::array<size_t, 1024>{});
109  info.max.push_back(std::numeric_limits<float>::lowest());
110  info.min.push_back(std::numeric_limits<float>::max());
111 
112  // Add the index by name in the correct position
113  // this will require moving the tail of the index,
114  // but inserting should happen only once for each metric,
115  // so who cares.
116  metricLabelIdx.second = info.metrics.size();
117  info.metricLabelsIdx.insert(mi, metricLabelIdx);
118  // Add the the actual Metric info to the store
119  metricIndex = info.metrics.size();
120  assert(metricInfo.storeIdx != -1);
121  assert(metricLabelIdx.first.empty() == false);
122  if (newMetricsCallback != nullptr) {
123  newMetricsCallback(metricLabelIdx.first, metricInfo, intValue);
124  }
125  info.metrics.push_back(metricInfo);
126  } else {
127  metricIndex = mi->second;
128  }
129  assert(metricIndex != -1);
130  // We are now guaranteed our metric is present at metricIndex.
131  MetricInfo &metricInfo = info.metrics[metricIndex];
132 
133  auto mod = info.timestamps[metricIndex].size();
134 
135  switch(metricInfo.type) {
136  case MetricType::Int: {
137  intValue = strtol(stringValue.str().c_str(), &ep, 10);
138  if (!ep || *ep != '\0') {
139  return false;
140  }
141  info.intMetrics[metricInfo.storeIdx][metricInfo.pos] = intValue;
142  info.max[metricIndex] = std::max(info.max[metricIndex], (float)intValue);
143  info.min[metricIndex] = std::min(info.min[metricIndex], (float)intValue);
144  } break;
145  case MetricType::Float: {
146  floatValue = strtof(stringValue.str().c_str(), &ep);
147  if (!ep || *ep != '\0') {
148  return false;
149  }
150  info.floatMetrics[metricInfo.storeIdx][metricInfo.pos] = floatValue;
151  info.max[metricIndex] = std::max(info.max[metricIndex], floatValue);
152  info.min[metricIndex] = std::min(info.min[metricIndex], floatValue);
153  } break;
154  default:
155  return false;
156  break;
157  };
158 
159  // Save the timestamp for the current metric we do it here
160  // so that we do not update timestamps for broken metrics
161  info.timestamps[metricIndex][metricInfo.pos] = timestamp;
162  // Update the position where to write the next metric
163  metricInfo.pos = (metricInfo.pos + 1) % mod;
164  return true;
165 }

168 size_t
169 DeviceMetricsHelper::metricIdxByName(const std::string& name, const DeviceMetricsInfo& info)
170 {
171  size_t i = 0;
172  while (i < info.metricLabelsIdx.size()) {
173  auto &metricName = info.metricLabelsIdx[i];
174  if (metricName.first == name) {
175  return metricName.second;
176  }
177  ++i;
178  }
179  return i;
180 }
